```javascript
function saveConversation(message, timestamp) {
  // 1. 解析时间
  const date = new Date(timestamp);
  const dateStr = formatDate(date, 'YYYYMMDD');
  const timeStr = formatDate(date, 'HH:mm');
  
  // 2. 构建文件路径
  const filePath = `docs/Conversation/${dateStr}.md`;
  
  // 3. 构建内容
  const content = `\n## ${timeStr}\n陛下：${message}\n`;
  
  // 4. 追加到文件
  if (!fs.existsSync(filePath)) {
    // 创建新文件（带头部）
    const header = `# ${dateStr} 陛下对话记录\n\n**记录部门**: [部门名称]\n**开始时间**: ${dateStr}\n\n---\n`;
    fs.writeFileSync(filePath, header);
  }
  
  fs.appendFileSync(filePath, content);
}
```

```javascript
function saveConversation(message, timestamp) {
```
