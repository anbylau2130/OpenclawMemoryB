namespace YunTianYou.Workflow.Engine
{
    public interface IWorkflowEngine
    {
        Task<WorkflowInstance> StartAsync(string workflowId, Dictionary<string, object> variables);
        Task<WorkflowInstance> GetInstanceAsync(Guid instanceId);
        Task<bool> SubmitAsync(Guid instanceId, string action, Dictionary<string, object> data);
        Task<bool> ApproveAsync(Guid instanceId, string approver, string comment);
        Task<bool> RejectAsync(Guid instanceId, string approver, string comment);
    }
    
    public class WorkflowEngine : IWorkflowEngine
    {
        private readonly IWorkflowRepository _repository;
        private readonly IWorkflowExecutionService _executionService;
        
        public async Task<WorkflowInstance> StartAsync(string workflowId, Dictionary<string, object> variables)
        {
            var definition = await _repository.GetDefinitionAsync(workflowId);
            var instance = new WorkflowInstance
            {
                Id = Guid.NewGuid(),
                WorkflowId = workflowId,
                Status = WorkflowStatus.Running,
                CurrentNode = definition.Nodes.First(n => n is StartNode).NodeId,
                Variables = variables,
                StartTime = DateTime.UtcNow
            };
            
            await _repository.SaveInstanceAsync(instance);
            await _executionService.ExecuteAsync(instance);
            
            return instance;
        }
    }
}

    }
    
    public class DataSyncService : IDataSyncService
    {
        public async Task<SyncResult> SyncAsync(SyncTask task)
        {
            var source = await ConnectAsync(task.SourceDataSource);
            var target = await ConnectAsync(task.TargetDataSource);
            
            var data = await FetchDataAsync(source, task.Query);
            var transformed = TransformData(data, task.Mapping);

            return new SyncResult { Success = true, RecordCount = data.Count };
        }
    }
}

-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 角色表
CREATE TABLE roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_name VARCHAR(50) UNIQUE NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 用户角色关联表
CREATE TABLE user_roles (
    user_id UUID REFERENCES users(id),
    role_id UUID REFERENCES roles(id),
    PRIMARY KEY (user_id, role_id)

namespace YunTianYou.Infrastructure.Caching
{
    public class CacheService
    {
        private readonly IDatabase _redis;
        
        public async Task<T> GetOrSetAsync<T>(
            string key, 
            Func<Task<T>> factory, 
            TimeSpan? expiry = null)
        {
            var cached = await _redis.StringGetAsync(key);
            if (cached.HasValue)
            {
                return JsonSerializer.Deserialize<T>(cached);
            }
            
            var value = await factory();
            await _redis.StringSetAsync(
                key, 
                JsonSerializer.Serialize(value), 
                expiry ?? TimeSpan.FromMinutes(30)
            );
            
            return value;
        }
    }
}

public class EncryptionService
{
    private readonly byte[] _key;
    
    public string Encrypt(string plainText)
    {
        using var aes = Aes.Create();
        aes.Key = _key;
        aes.GenerateIV();
        
        var encryptor = aes.CreateEncryptor();
        var plainBytes = Encoding.UTF8.GetBytes(plainText);
        var cipherBytes = encryptor.TransformFinalBlock(plainBytes, 0, plainBytes.Length);
        
        return Convert.ToBase64String(aes.IV.Concat(cipherBytes).ToArray());
    }
}
