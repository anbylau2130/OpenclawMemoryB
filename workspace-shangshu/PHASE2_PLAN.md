- **位置**：`backend/routes/auth.js`
- **问题**：`error.message`直接返回给客户端，可能泄露敏感信息
- **修复方案**：
  ```javascript
  // 修改前
  res.status(500).json({ error: '登录失败', message: error.message });
  
  // 修改后
  console.error('登录失败:', error); // 仅记录日志
  res.status(500).json({ error: '登录失败' }); // 不返回详细信息
  ```
- **预估时间**：1小时
- **截止时间**：2026-03-26 18:00
- **责任人**：兵部（待分配）

### 🟡 中优先级（下周完成）

#### P2-01: 输入验证增强
- **位置**：`backend/routes/auth.js`
- **问题**：缺少email格式、username长度、密码复杂度验证
- **修复方案**：
  ```javascript
  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailRegex.test(email)) {
    return res.status(400).json({ error: '邮箱格式不正确' });
  }
  if (username.length < 3 || username.length > 20) {
    return res.status(400).json({ error: '用户名长度需为3-20位' });
  }
  if (!/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d).{8,}$/.test(password)) {
    return res.status(400).json({ error: '密码需包含大小写字母和数字，至少8位' });
  }
  ```
- **预估时间**：2小时
- **截止时间**：2026-03-27 18:00
- **责任人**：兵部（待分配）

#### P2-06: Rate Limiting
- **位置**：全局
- **问题**：缺少API访问频率限制，易受暴力破解攻击
- **修复方案**：
  ```javascript
  const rateLimit = require('express-rate-limit');
  
  const loginLimiter = rateLimit({
    windowMs: 15 * 60 * 1000, // 15分钟
    max: 5, // 最多5次尝试
    message: { error: '登录尝试次数过多，请稍后再试' }
  });
  
  app.use('/api/auth/login', loginLimiter);
  ```
- **预估时间**：1小时
- **截止时间**：2026-03-29 18:00
- **责任人**：兵部（待分配）
