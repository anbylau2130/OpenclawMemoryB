## 🗄️ 数据库优化

### 1. 索引优化

**索引设计原则**:
```sql
-- 1. 主键索引（自动创建）
CREATE TABLE t_bill (
    bill_id BIGSERIAL PRIMARY KEY,
    bill_no VARCHAR(50) NOT NULL,
    bill_date DATE NOT NULL,
    customer_id BIGINT,
    total_amount DECIMAL(18,2),
    status VARCHAR(20),
    created_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. 唯一索引
CREATE UNIQUE INDEX idx_bill_no ON t_bill(bill_no);

-- 3. 复合索引（注意顺序）
CREATE INDEX idx_bill_date_status ON t_bill(bill_date, status);
CREATE INDEX idx_customer_date ON t_bill(customer_id, bill_date);

-- 4. 部分索引（条件索引）
CREATE INDEX idx_active_bills ON t_bill(bill_date) WHERE status = 'ACTIVE';

-- 5. 函数索引
CREATE INDEX idx_bill_date_month ON t_bill(EXTRACT(YEAR FROM bill_date), EXTRACT(MONTH FROM bill_date));

-- 6. 表达式索引
CREATE INDEX idx_lower_bill_no ON t_bill(LOWER(bill_no));

-- 7. 包含索引（覆盖索引）
CREATE INDEX idx_bill_covering ON t_bill(customer_id) INCLUDE (bill_no, total_amount, status);
```

**索引使用建议**:
```sql
-- 好的查询（使用索引）
SELECT * FROM t_bill WHERE bill_no = 'B20260320001';
SELECT * FROM t_bill WHERE bill_date = '2026-03-20' AND status = 'ACTIVE';
SELECT * FROM t_bill WHERE customer_id = 1001 ORDER BY bill_date DESC;

-- 不好的查询（无法使用索引）
SELECT * FROM t_bill WHERE YEAR(bill_date) = 2026;  -- 函数导致索引失效
SELECT * FROM t_bill WHERE bill_no LIKE '%B2026%';  -- 前缀模糊匹配
SELECT * FROM t_bill WHERE customer_id + 1 = 1002;  -- 表达式导致索引失效
```

**索引维护**:

### 2. 查询优化

**查询优化技巧**:
```sql
-- 1. 使用EXPLAIN分析查询计划
EXPLAIN ANALYZE SELECT * FROM t_bill WHERE bill_date = '2026-03-20';

-- 2. 避免SELECT *
-- 不好
SELECT * FROM t_bill WHERE customer_id = 1001;

-- 好
SELECT bill_id, bill_no, total_amount FROM t_bill WHERE customer_id = 1001;

-- 3. 使用LIMIT分页
-- 不好（深度分页）
SELECT * FROM t_bill ORDER BY bill_id LIMIT 10000 OFFSET 90000;

-- 好（游标分页）
SELECT * FROM t_bill WHERE bill_id > 90000 ORDER BY bill_id LIMIT 10000;

-- 4. 避免子查询，使用JOIN
-- 不好
SELECT * FROM t_bill 
WHERE customer_id IN (SELECT customer_id FROM t_customer WHERE region = '华东');

-- 好
SELECT b.* FROM t_bill b
INNER JOIN t_customer c ON b.customer_id = c.customer_id
WHERE c.region = '华东';

-- 5. 批量操作
-- 不好（循环插入）
INSERT INTO t_bill_item (bill_id, item_no, quantity) VALUES (1, 'I001', 10);
INSERT INTO t_bill_item (bill_id, item_no, quantity) VALUES (1, 'I002', 20);

-- 好（批量插入）
INSERT INTO t_bill_item (bill_id, item_no, quantity) VALUES 
(1, 'I001', 10),
(1, 'I002', 20);

-- 6. 使用CTE提高可读性
WITH active_customers AS (
    SELECT customer_id FROM t_customer WHERE status = 'ACTIVE'
),
recent_bills AS (
    SELECT bill_id, customer_id, total_amount 
    FROM t_bill 
    WHERE bill_date >= CURRENT_DATE - INTERVAL '30 days'
)
SELECT c.customer_name, COUNT(r.bill_id) as bill_count, SUM(r.total_amount) as total
FROM active_customers ac
JOIN t_customer c ON ac.customer_id = c.customer_id
LEFT JOIN recent_bills r ON ac.customer_id = r.customer_id
GROUP BY c.customer_id, c.customer_name;
```

## 🚀 缓存策略

### 1. Redis缓存架构

**多级缓存架构**:
```
客户端 → 本地缓存(Caffeine) → Redis缓存 → 数据库
```

**缓存配置**:
```java
@Configuration
@EnableCaching
public class MultiLevelCacheConfig {
    
    // 本地缓存（一级缓存）
    @Bean
    public CacheManager localCacheManager() {
        CaffeineCacheManager cacheManager = new CaffeineCacheManager();
        cacheManager.setCaffeine(Caffeine.newBuilder()
            .initialCapacity(100)
            .maximumSize(1000)
            .expireAfterWrite(5, TimeUnit.MINUTES)
            .recordStats());
        return cacheManager;
    }
    
    // Redis缓存（二级缓存）
    @Bean
    @Primary
    public CacheManager redisCacheManager(RedisConnectionFactory connectionFactory) {
        RedisCacheConfiguration config = RedisCacheConfiguration.defaultCacheConfig()
            .entryTtl(Duration.ofHours(1))
            .serializeValuesWith(RedisSerializationContext.SerializationPair
                .fromSerializer(new GenericJackson2JsonRedisSerializer()));
        
        return RedisCacheManager.builder(connectionFactory)
            .cacheDefaults(config)
            .transactionAware()
            .build();
    }
}
```

### 2. 缓存策略选择

**缓存策略对比**:

| 策略 | 适用场景 | 优点 | 缺点 |
|-----|---------|------|------|
| Cache-Aside | 读多写少 | 实现简单 | 数据可能不一致 |
| Read-Through | 读密集 | 简化读取逻辑 | 需要缓存中间件支持 |
| Write-Through | 写密集 | 数据一致性好 | 写性能较低 |
| Write-Behind | 写密集 | 写性能高 | 数据可能丢失 |
| Refresh-Ahead | 热点数据 | 避免缓存失效 | 实现复杂 |

**Cache-Aside模式实现**:
```java
@Service
public class CacheService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private BillMapper billMapper;
    
    private static final String CACHE_PREFIX = "k3cloud:bill:";
    private static final Duration CACHE_TTL = Duration.ofHours(1);
    
    // 读缓存
    public Bill getBillById(Long billId) {
        String key = CACHE_PREFIX + billId;
        
        // 1. 尝试从缓存读取
        Bill bill = (Bill) redisTemplate.opsForValue().get(key);
        
        if (bill != null) {
            return bill;
        }
        
        // 2. 缓存未命中，从数据库读取
        bill = billMapper.selectById(billId);
        
        if (bill != null) {
            // 3. 写入缓存
            redisTemplate.opsForValue().set(key, bill, CACHE_TTL);
        }
        
        return bill;
    }
    
    // 更新缓存
    @Transactional
    public void updateBill(Bill bill) {
        String key = CACHE_PREFIX + bill.getBillId();
        
        // 1. 先删除缓存
        redisTemplate.delete(key);
        
        try {
            // 2. 更新数据库
            billMapper.updateById(bill);
        } catch (Exception e) {
            // 3. 数据库更新失败，恢复缓存
            redisTemplate.opsForValue().set(key, bill, CACHE_TTL);
            throw e;
        }
    }
    
    // 删除缓存
    public void deleteBill(Long billId) {
        String key = CACHE_PREFIX + billId;
        
        // 1. 删除缓存
        redisTemplate.delete(key);
        
        // 2. 删除数据库
        billMapper.deleteById(billId);
    }
}
```

**Write-Behind模式实现**:
```java
@Service
public class WriteBehindCacheService {
    
    @Autowired
    private RedisTemplate<String, Object> redisTemplate;
    
    @Autowired
    private BillMapper billMapper;
    
    private final BlockingQueue<Bill> writeQueue = new LinkedBlockingQueue<>(10000);
    private final ExecutorService executor = Executors.newFixedThreadPool(5);
    
    @PostConstruct
    public void init() {
        // 启动后台写入线程
        for (int i = 0; i < 5; i++) {
            executor.submit(this::processWriteQueue);
        }
    }
    
    // 写入缓存（异步写数据库）
    public void writeBill(Bill bill) {
        String key = CACHE_PREFIX + bill.getBillId();
        
        // 1. 立即写入缓存
        redisTemplate.opsForValue().set(key, bill, CACHE_TTL);
        
        // 2. 异步写入数据库
        writeQueue.offer(bill);
    }
    
    // 后台处理队列
    private void processWriteQueue() {
        while (!Thread.currentThread().isInterrupted()) {
            try {
                Bill bill = writeQueue.take();
                
                // 批量写入
                List<Bill> batch = new ArrayList<>();
                batch.add(bill);
                writeQueue.drainTo(batch, 99);
                
                // 批量更新数据库
                billMapper.batchUpdate(batch);
                
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                break;
            } catch (Exception e) {
                logger.error("写入数据库失败", e);
            }
        }
    }
}
```

```java
@Service
public class CachePenetrationSolution {
    
    // 方案1：缓存空值
    public Bill getBillById(Long billId) {
        String key = CACHE_PREFIX + billId;
        
        Bill bill = (Bill) redisTemplate.opsForValue().get(key);
        
        if (bill != null) {
            return bill;
        }
        
        // 检查是否是空值标记
        if (redisTemplate.hasKey(key + ":null")) {
            return null;
        }
        
        bill = billMapper.selectById(billId);
        
        if (bill != null) {
            redisTemplate.opsForValue().set(key, bill, CACHE_TTL);
        } else {
            // 缓存空值，防止穿透
            redisTemplate.opsForValue().set(key + ":null", "1", Duration.ofMinutes(5));
        }
        
        return bill;
    }
    
    // 方案2：布隆过滤器
    @Autowired
    private BloomFilter<Long> billIdBloomFilter;
    
    public Bill getBillByIdWithBloom(Long billId) {
        // 先检查布隆过滤器
        if (!billIdBloomFilter.mightContain(billId)) {
            return null;  // 一定不存在
        }
        
        // 可能存在，继续查询
        return getBillById(billId);
    }
}
```

**缓存击穿**:
```java
@Service
public class CacheBreakdownSolution {
    
    // 方案1：互斥锁
    public Bill getBillByIdWithLock(Long billId) {
        String key = CACHE_PREFIX + billId;
        String lockKey = "lock:" + key;
        
        Bill bill = (Bill) redisTemplate.opsForValue().get(key);
        
        if (bill != null) {
            return bill;
        }
        
        // 尝试获取锁
        Boolean locked = redisTemplate.opsForValue().setIfAbsent(
            lockKey, "1", Duration.ofSeconds(10));
        
        if (Boolean.TRUE.equals(locked)) {
            try {
                // 获取锁成功，查询数据库
                bill = billMapper.selectById(billId);
                
                if (bill != null) {
                    redisTemplate.opsForValue().set(key, bill, CACHE_TTL);
                }
                
                return bill;
            } finally {
                // 释放锁
                redisTemplate.delete(lockKey);
            }
        } else {
            // 获取锁失败，等待并重试
            try {
                Thread.sleep(50);
                return getBillByIdWithLock(billId);
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                throw new RuntimeException(e);
            }
        }
    }
    
    // 方案2：逻辑过期
    public Bill getBillByIdWithLogicalExpire(Long billId) {
        String key = CACHE_PREFIX + billId;
        
        RedisData redisData = (RedisData) redisTemplate.opsForValue().get(key);
        
        if (redisData == null) {
            return null;
        }
        
        // 检查逻辑过期时间
        if (redisData.getExpireTime().isAfter(LocalDateTime.now())) {
            return redisData.getData();
        }
        
        // 已过期，异步刷新
        String lockKey = "lock:" + key;
        Boolean locked = redisTemplate.opsForValue().setIfAbsent(
            lockKey, "1", Duration.ofSeconds(10));
        
        if (Boolean.TRUE.equals(locked)) {
            // 获取锁成功，异步刷新缓存
            CompletableFuture.runAsync(() -> {
                try {
                    Bill bill = billMapper.selectById(billId);
                    RedisData newData = new RedisData(bill, LocalDateTime.now().plusHours(1));
                    redisTemplate.opsForValue().set(key, newData);
                } finally {
                    redisTemplate.delete(lockKey);
                }
            });
        }
        
        // 返回旧数据
        return redisData.getData();
    }
}
```

```java
@Service
public class CacheAvalancheSolution {
    
    // 方案1：随机过期时间
    public void setCacheWithRandomExpire(String key, Object value) {
        // 基础过期时间 + 随机时间
        long baseExpire = 3600;  // 1小时
        long randomExpire = ThreadLocalRandom.current().nextLong(0, 600);  // 0-10分钟
        
        redisTemplate.opsForValue().set(key, value, 
            Duration.ofSeconds(baseExpire + randomExpire));
    }
    
    // 方案2：多级缓存
    @Autowired
    private Cache localCache;  // Caffeine本地缓存
    
    public Bill getBillWithMultiLevel(Long billId) {
        String key = CACHE_PREFIX + billId;
        
        // 1. 查询本地缓存
        Bill bill = localCache.get(key, () -> null);
        if (bill != null) {
            return bill;
        }
        
        // 2. 查询Redis缓存
        bill = (Bill) redisTemplate.opsForValue().get(key);
        if (bill != null) {
            // 写入本地缓存
            localCache.put(key, bill);
            return bill;
        }
        
        // 3. 查询数据库
        bill = billMapper.selectById(billId);
        
        if (bill != null) {
            // 写入Redis和本地缓存
            setCacheWithRandomExpire(key, bill);
            localCache.put(key, bill);
        }
        
        return bill;
    }
```

### 2. 负载均衡算法

**算法对比**:

| 算法 | 说明 | 适用场景 |
|-----|------|---------|
| Round Robin | 轮询（默认） | 服务器性能相近 |
| Weighted | 加权轮询 | 服务器性能不同 |
| Least Connections | 最少连接 | 长连接场景 |
| IP Hash | 基于IP哈希 | 需要会话保持 |
| URL Hash | 基于URL哈希 | 缓存场景 |
| Random | 随机选择 | 简单场景 |

```java
        "192.168.1.103:8080"
    );
    
    private final Map<String, Integer> serverConnections = new ConcurrentHashMap<>();
    private final AtomicInteger counter = new AtomicInteger(0);
    
    // 加权轮询
    public String getServerByWeight() {
        int totalWeight = servers.stream()
            .mapToInt(this::getServerWeight)
            .sum();
        
        int currentWeight = counter.getAndIncrement() % totalWeight;
        
        for (String server : servers) {
            currentWeight -= getServerWeight(server);
            if (currentWeight < 0) {
                return server;
            }
        }
        
        return servers.get(0);
    }
    
    // 最少连接
    public String getServerByLeastConnection() {
        return serverConnections.entrySet().stream()
            .min(Map.Entry.comparingByValue())
            .map(Map.Entry::getKey)
            .orElse(servers.get(0));
    }
    
    // 一致性哈希
    public String getServerByConsistentHash(String key) {
        TreeMap<Integer, String> ring = new TreeMap<>();
        
        for (String server : servers) {
            for (int i = 0; i < 160; i++) {  // 虚拟节点
                int hash = Hashing.murmur3_128()
                    .hashString(server + "#" + i, StandardCharsets.UTF_8)
                    .asInt();
                ring.put(hash, server);
            }
        }
        
        int keyHash = Hashing.murmur3_128()
            .hashString(key, StandardCharsets.UTF_8)
            .asInt();
        
        Map.Entry<Integer, String> entry = ring.ceilingEntry(keyHash);
        if (entry == null) {
            entry = ring.firstEntry();
        }
        
        return entry.getValue();
    }
    
    private int getServerWeight(String server) {
        // 根据服务器配置返回权重
        return 3;  // 默认权重
    }
}
```
