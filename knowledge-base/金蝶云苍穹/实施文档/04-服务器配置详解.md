## 🔧 操作系统配置

### 1. Linux系统优化

**内核参数调优** (`/etc/sysctl.conf`):
```bash
# 网络优化
net.core.somaxconn = 65535
net.core.netdev_max_backlog = 65535
net.ipv4.tcp_max_syn_backlog = 65535
net.ipv4.tcp_syncookies = 1
net.ipv4.tcp_tw_reuse = 1
net.ipv4.tcp_tw_recycle = 0
net.ipv4.tcp_fin_timeout = 30
net.ipv4.tcp_keepalive_time = 1200
net.ipv4.tcp_keepalive_probes = 5
net.ipv4.tcp_keepalive_intvl = 30
net.ipv4.tcp_max_tw_buckets = 10000
net.ipv4.ip_local_port_range = 1024 65535

# 内存优化
vm.swappiness = 10
vm.dirty_ratio = 80
vm.dirty_background_ratio = 5
vm.overcommit_memory = 1

# 文件描述符
fs.file-max = 2097152
fs.nr_open = 2097152

# 共享内存
kernel.shmmax = 68719476736
kernel.shmall = 4294967296
```

**应用配置**:
```bash
# 使配置生效
sysctl -p

# 查看当前配置
sysctl -a | grep <参数名>
```

### 2. 用户限制配置

**文件描述符限制** (`/etc/security/limits.conf`):
```bash
# 软限制和硬限制

* hard nofile 65535
* soft nproc 65535
* hard nproc 65535

# 针对k3cloud用户
k3cloud soft nofile 100000
k3cloud hard nofile 100000
k3cloud soft nproc 100000
k3cloud hard nproc 100000
k3cloud soft memlock unlimited
k3cloud hard memlock unlimited
```

**Systemd服务限制** (`/etc/systemd/system/k3cloud.service`):
```ini
[Service]
LimitNOFILE=100000
LimitNPROC=100000
LimitMEMLOCK=infinity
```

### 3. 磁盘I/O优化

**调度算法设置**:
```bash
# 查看当前调度算法
cat /sys/block/sda/queue/scheduler

# SSD使用noop或mq-deadline
echo noop > /sys/block/sda/queue/scheduler

# HDD使用deadline或cfq
echo deadline > /sys/block/sda/queue/scheduler

# 永久设置（添加到/etc/rc.local）
echo 'echo noop > /sys/block/sda/queue/scheduler' >> /etc/rc.local
```

**文件系统优化**:
```bash
# XFS挂载选项
mount -o noatime,nodiratime,logbufs=8,logbsize=256k /dev/sda1 /data

# Ext4挂载选项
mount -o noatime,nodiratime,data=writeback /dev/sda1 /data

# /etc/fstab配置
/dev/sda1  /data  xfs  noatime,nodiratime,logbufs=8,logbsize=256k  0 0
```

### 2. 负载均衡配置

**Nginx配置示例**:
```nginx
upstream k3cloud_backend {
    # 负载均衡算法
    least_conn;
    
    # 应用服务器列表
    server 192.168.1.101:8080 weight=3 max_fails=3 fail_timeout=30s;
    server 192.168.1.102:8080 weight=3 max_fails=3 fail_timeout=30s;
    server 192.168.1.103:8080 weight=2 max_fails=3 fail_timeout=30s;
    
    # 健康检查
    keepalive 32;
}

server {
    listen 80;
    server_name k3cloud.example.com;
    
    # 重定向到HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name k3cloud.example.com;
    
    # SSL配置
    ssl_certificate /etc/nginx/ssl/k3cloud.crt;
    ssl_certificate_key /etc/nginx/ssl/k3cloud.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;
    ssl_session_cache shared:SSL:10m;
    ssl_session_timeout 10m;
    
    # 安全头
    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    
    # 代理配置
    location / {
        proxy_pass http://k3cloud_backend;
        proxy_http_version 1.1;
        
        # 请求头
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # WebSocket支持
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        
        # 超时设置
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
        
        # 缓冲设置
        proxy_buffering on;
        proxy_buffer_size 4k;
        proxy_buffers 8 4k;
    }
    
    # 静态文件
    location /static/ {
        alias /data/k3cloud/static/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }
    
    # 文件上传大小限制
    client_max_body_size 100M;
}
```

**HAProxy配置示例**:
```haproxy
global
    log /dev/log local0
    log /dev/log local1 notice

defaults
    log     global

backend k3cloud_app

backend static
    balance roundrobin
    server static1 192.168.1.201:80 check
```

### 3. 防火墙配置

**iptables规则示例**:
```bash
# 清除现有规则
iptables -F
iptables -X

# 默认策略
iptables -P INPUT DROP
iptables -P FORWARD DROP
iptables -P OUTPUT ACCEPT

# 允许回环接口
iptables -A INPUT -i lo -j ACCEPT
iptables -A OUTPUT -o lo -j ACCEPT

# 允许已建立的连接
iptables -A INPUT -m state --state ESTABLISHED,RELATED -j ACCEPT

# SSH访问（限制IP）
iptables -A INPUT -p tcp -s 192.168.1.0/24 --dport 22 -j ACCEPT

# HTTP/HTTPS访问
iptables -A INPUT -p tcp --dport 80 -j ACCEPT
iptables -A INPUT -p tcp --dport 443 -j ACCEPT

# 应用端口（仅内网）
iptables -A INPUT -p tcp -s 192.168.1.0/24 --dport 8080 -j ACCEPT

# 数据库端口（仅应用服务器访问）
iptables -A INPUT -p tcp -s 192.168.1.101 --dport 5432 -j ACCEPT
iptables -A INPUT -p tcp -s 192.168.1.102 --dport 5432 -j ACCEPT
iptables -A INPUT -p tcp -s 192.168.1.103 --dport 5432 -j ACCEPT

# Redis端口（仅应用服务器访问）
iptables -A INPUT -p tcp -s 192.168.1.0/24 --dport 6379 -j ACCEPT

# 防止SYN攻击
iptables -A INPUT -p tcp --syn -m limit --limit 1/s --limit-burst 3 -j ACCEPT

# 防止ping攻击
iptables -A INPUT -p icmp --icmp-type echo-request -m limit --limit 1/s -j ACCEPT

# 保存规则
service iptables save
# 或
iptables-save > /etc/iptables/rules.v4
```

**firewalld配置示例**:
```bash
# 启用防火墙
systemctl enable firewalld
systemctl start firewalld

# 默认区域
firewall-cmd --set-default-zone=public

# 允许服务
firewall-cmd --permanent --add-service=http
firewall-cmd --permanent --add-service=https
firewall-cmd --permanent --add-port=8080/tcp

# 允许特定IP访问
firewall-cmd --permanent --add-rich-rule='rule family="ipv4" source address="192.168.1.0/24" port protocol="tcp" port="22" accept'

# 重载配置
firewall-cmd --reload

# 查看规则
firewall-cmd --list-all
```

**告警规则** (`rules/k3cloud.yml`):
```yaml
groups:
  - name: k3cloud_alerts
    rules:
      # CPU使用率告警
      - alert: HighCPUUsage
        expr: 100 - (avg by(instance) (irate(node_cpu_seconds_total{mode="idle"}[5m])) * 100) > 80
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "高CPU使用率: {{ $labels.instance }}"
          description: "CPU使用率 {{ $value }}%"

      # 内存使用率告警
      - alert: HighMemoryUsage
        expr: (1 - (node_memory_MemAvailable_bytes / node_memory_MemTotal_bytes)) * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "高内存使用率: {{ $labels.instance }}"
          description: "内存使用率 {{ $value }}%"

      # 磁盘使用率告警
      - alert: HighDiskUsage
        expr: (1 - (node_filesystem_avail_bytes{fstype!="tmpfs"} / node_filesystem_size_bytes{fstype!="tmpfs"})) * 100 > 85
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "高磁盘使用率: {{ $labels.instance }}"
          description: "磁盘 {{ $labels.mountpoint }} 使用率 {{ $value }}%"

      # 应用响应时间告警
      - alert: HighResponseTime
        expr: histogram_quantile(0.95, rate(http_server_requests_seconds_bucket[5m])) > 2
        for: 5m
        labels:
          severity: warning
        annotations:
          summary: "应用响应时间过长"
          description: "P95响应时间 {{ $value }}秒"

      # 应用错误率告警
      - alert: HighErrorRate
        expr: sum(rate(http_server_requests_seconds_count{status=~"5.."}[5m])) / sum(rate(http_server_requests_seconds_count[5m])) * 100 > 5
        for: 5m
        labels:
          severity: critical
        annotations:
          summary: "应用错误率过高"
          description: "错误率 {{ $value }}%"
```

### 2. SSL/TLS证书配置

**生成自签名证书**:
```bash
# 生成私钥
openssl genrsa -out k3cloud.key 2048

# 生成证书签名请求
openssl req -new -key k3cloud.key -out k3cloud.csr \
  -subj "/C=CN/ST=Guangdong/L=Shenzhen/O=Kingdee/CN=k3cloud.example.com"

# 生成自签名证书
openssl x509 -req -days 365 -in k3cloud.csr -signkey k3cloud.key \
  -out k3cloud.crt

# 转换为PKCS12格式
openssl pkcs12 -export -in k3cloud.crt -inkey k3cloud.key \
  -out k3cloud.p12 -name k3cloud

# 生成DH参数
openssl dhparam -out dhparam.pem 2048
```

**Let's Encrypt证书**:
```bash
# 安装certbot
yum install -y certbot python3-certbot-nginx

# 获取证书
certbot certonly --nginx -d k3cloud.example.com

# 自动续期

0 0 1 * * /usr/bin/certbot renew --quiet
```
