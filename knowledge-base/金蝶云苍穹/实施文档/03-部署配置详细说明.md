## 🐳 Dockerfile配置

### 1. 应用服务Dockerfile

**多阶段构建Dockerfile**:
```dockerfile
# ==================== 第一阶段：构建阶段 ====================
FROM maven:3.8.6-openjdk-11 AS builder

# 设置工作目录
WORKDIR /build

# 复制pom.xml并下载依赖（利用Docker缓存）
COPY pom.xml .
RUN mvn dependency:go-offline -B

# 复制源代码
COPY src ./src

# 编译应用
RUN mvn clean package -DskipTests -B

# ==================== 第二阶段：运行阶段 ====================
FROM openjdk:11-jre-slim

# 安装必要的工具
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
    curl \
    ca-certificates \
    && rm -rf /var/lib/apt/lists/*

# 创建应用用户
RUN groupadd -r k3cloud && useradd -r -g k3cloud k3cloud

# 设置工作目录
WORKDIR /app

# 从构建阶段复制jar包
COPY --from=builder /build/target/*.jar app.jar

# 复制启动脚本
COPY scripts/docker-entrypoint.sh /app/
RUN chmod +x /app/docker-entrypoint.sh

# 创建日志和数据目录
RUN mkdir -p /app/logs /app/data && \
    chown -R k3cloud:k3cloud /app

# 切换到非root用户
USER k3cloud

# 设置环境变量
ENV JAVA_OPTS="-Xms2g -Xmx4g -XX:+UseG1GC -XX:MaxGCPauseMillis=200"
ENV SPRING_PROFILES_ACTIVE=production
ENV LOGGING_LEVEL_ROOT=INFO

# 暴露端口
EXPOSE 8080 8081

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=60s --retries=3 \
    CMD curl -f http://localhost:8081/actuator/health || exit 1

# 启动应用
ENTRYPOINT ["./docker-entrypoint.sh"]
```

**启动脚本** (`scripts/docker-entrypoint.sh`):
```bash
#!/bin/bash
set -e

# JVM参数
JAVA_OPTS="${JAVA_OPTS:-}"
# Spring Profile
SPRING_PROFILES_ACTIVE="${SPRING_PROFILES_ACTIVE:-production}"
# 数据库配置
DB_HOST="${DB_HOST:-localhost}"
DB_PORT="${DB_PORT:-5432}"
DB_NAME="${DB_NAME:-k3cloud_db}"
DB_USER="${DB_USER:-k3cloud}"
DB_PASSWORD="${DB_PASSWORD:-}"
# Redis配置
REDIS_HOST="${REDIS_HOST:-localhost}"
REDIS_PORT="${REDIS_PORT:-6379}"
REDIS_PASSWORD="${REDIS_PASSWORD:-}"

# 设置Java选项
export JAVA_OPTS="${JAVA_OPTS} -Dspring.profiles.active=${SPRING_PROFILES_ACTIVE}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.datasource.url=jdbc:postgresql://${DB_HOST}:${DB_PORT}/${DB_NAME}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.datasource.username=${DB_USER}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.datasource.password=${DB_PASSWORD}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.redis.host=${REDIS_HOST}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.redis.port=${REDIS_PORT}"
export JAVA_OPTS="${JAVA_OPTS} -Dspring.redis.password=${REDIS_PASSWORD}"

# 启动应用
echo "Starting K3Cloud Application..."
echo "JAVA_OPTS: ${JAVA_OPTS}"
exec java ${JAVA_OPTS} -jar /app/app.jar
```

### 2. Nginx Dockerfile

```dockerfile
FROM nginx:1.24-alpine

# 复制nginx配置
COPY nginx/nginx.conf /etc/nginx/nginx.conf

    chmod -R 755 /var/log/nginx

# 暴露端口

# 启动nginx
CMD ["nginx", "-g", "daemon off;"]
```

### 3. PostgreSQL Dockerfile

```dockerfile
FROM postgres:14-alpine

# 安装必要工具
RUN apk add --no-cache \
    postgresql-client \
    bash

# 复制初始化脚本
COPY postgres/init /docker-entrypoint-initdb.d

# 创建数据目录
RUN mkdir -p /var/lib/postgresql/data && \
    chown -R postgres:postgres /var/lib/postgresql/data

# 切换到postgres用户
USER postgres

# 暴露端口
EXPOSE 5432

# 健康检查
HEALTHCHECK --interval=10s --timeout=5s --start-period=10s --retries=5 \
    CMD pg_isready -U k3cloud

# 启动PostgreSQL
CMD ["postgres"]
```

**初始化脚本** (`postgres/init/01-init-db.sh`):
```bash
#!/bin/bash
set -e

# 创建数据库
psql -v ON_ERROR_STOP=1 <<-EOSQL
    CREATE DATABASE k3cloud_db OWNER k3cloud ENCODING 'UTF8';
EOSQL

# 连接到数据库
psql -v ON_ERROR_STOP=1 -d k3cloud_db <<-EOSQL
    -- 创建扩展
    CREATE EXTENSION IF NOT EXISTS "uuid-ossp";
    CREATE EXTENSION IF NOT EXISTS "pgcrypto";
    CREATE EXTENSION IF NOT EXISTS "pg_trgm";
EOSQL

echo "Database initialization completed!"
```

### 3. 部署脚本

**启动脚本** (`deploy.sh`):
```bash
#!/bin/bash
set -e

# 颜色定义
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# 日志函数
log_info() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

log_warn() {
    echo -e "${YELLOW}[WARN]${NC} $1"
}

log_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# 检查环境变量文件
if [ ! -f .env.production ]; then
    log_error ".env.production 文件不存在"
    exit 1
fi

# 加载环境变量
export $(cat .env.production | grep -v '^#' | xargs)

# 创建数据目录
log_info "创建数据目录..."
mkdir -p /data/k3cloud/{postgres,redis,logs,data}

# 检查Docker是否运行
if ! docker info > /dev/null 2>&1; then
    log_error "Docker未运行，请先启动Docker"
    exit 1
fi

# 拉取镜像
log_info "拉取Docker镜像..."
docker-compose pull

# 构建镜像
log_info "构建Docker镜像..."
docker-compose build

# 启动服务
log_info "启动服务..."
docker-compose up -d

# 等待服务启动
log_info "等待服务启动..."
sleep 30

# 检查服务状态
log_info "检查服务状态..."
docker-compose ps

# 显示日志
log_info "显示应用日志..."
docker-compose logs -f k3cloud-app-1
```

**备份脚本** (`backup.sh`):
```bash
#!/bin/bash
set -e

# 备份目录
BACKUP_DIR="/backup/k3cloud/$(date +%Y%m%d_%H%M%S)"
mkdir -p "$BACKUP_DIR"

log_info "开始备份..."
log_info "备份目录: $BACKUP_DIR"

# 备份PostgreSQL
log_info "备份PostgreSQL..."
docker exec postgres-primary pg_dump -U k3cloud k3cloud_db > "$BACKUP_DIR/postgres_backup.sql"

# 备份Redis
log_info "备份Redis..."
docker exec redis-primary redis-cli --rdb /data/dump.rdb
docker cp redis-primary:/data/dump.rdb "$BACKUP_DIR/redis_backup.rdb"

# 备份应用数据
log_info "备份应用数据..."
tar -czf "$BACKUP_DIR/app_data.tar.gz" /data/k3cloud/data
tar -czf "$BACKUP_DIR/app_logs.tar.gz" /data/k3cloud/logs

log_info "备份完成: $BACKUP_DIR"
```
