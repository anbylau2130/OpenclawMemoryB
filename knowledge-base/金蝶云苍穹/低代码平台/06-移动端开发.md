## 一、移动端开发概述

### 1.1 移动端特性

苍穹移动端开发具有以下特性：
- **一次开发，多端运行**: 支持iOS、Android、H5

    }
    
    // 字段变更

        // 处理字段变更
    }
    
    // 按钮点击
    click(buttonKey) {
        switch (buttonKey) {
            case "scan":
                this.doScan();

            case "photo":
                this.doPhoto();
                break;
            case "location":
                this.doLocation();
                break;
        }
    }

### 3.2 调用原生能力

#### 扫码功能
```javascript
// 调用扫码
doScan() {
    var that = this;
    
    // 调用扫码组件
    this.getView().scanCode({
        type: ["qrCode", "barCode"],
        success: function(result) {
            // 扫码成功
            that.getModel().setValue("barcode", result.code);
            that.handleScanResult(result);
        },
        fail: function(error) {
            // 扫码失败
            that.getView().showErrorNotification("扫码失败: " + error);
        }
    });
}
```

#### 拍照功能
```javascript
// 调用拍照
doPhoto() {
    var that = this;
    
    this.getView().takePhoto({
        quality: 80,
        maxSize: 1024,
        success: function(result) {
            // 拍照成功，result为图片路径
            that.uploadPhoto(result.path);
        },
        fail: function(error) {
            that.getView().showErrorNotification("拍照失败");
        }
    });
}
```

#### 定位功能
```javascript
// 获取位置
doLocation() {
    var that = this;
    
    this.getView().getLocation({
        type: "gcj02",
        success: function(result) {
            // 获取位置成功
            var address = result.address;
            var lat = result.latitude;
            var lng = result.longitude;
            
            that.getModel().setValue("address", address);
            that.getModel().setValue("latitude", lat);
            that.getModel().setValue("longitude", lng);
        },
        fail: function(error) {
            that.getView().showErrorNotification("定位失败");
        }
    });
}
```

## 四、移动端列表开发

### 4.1 列表配置

```javascript
// 移动端列表插件
mlist {
    // 初始化
    onLoad() {
        // 配置列表
        this.configList();
    }
    
    // 配置列表
    configList() {
        var config = {
            columns: [
                { field: "billNo", title: "单号", width: "40%" },
                { field: "date", title: "日期", width: "30%" },
                { field: "amount", title: "金额", width: "30%", format: "currency" }
            ],
            filters: [
                { field: "status", title: "状态", type: "select", options: ["待审批", "已审批"] }
            ],
            sortField: "date",
            sortOrder: "desc"
        };
        
        this.setListConfig(config);
    }
    
    // 列表项点击
    itemClick(item) {
        // 打开详情
        this.openDetail(item.id);
    }
    
    // 下拉刷新
    onRefresh() {
        this.reloadData();
        this.getView().stopRefresh();
    }
    
    // 上拉加载
    onLoadMore() {
        this.loadMoreData();
    }
}
```

## 六、离线功能开发

### 6.1 离线数据配置

```javascript
// 配置离线数据
offline {
    // 需要离线的数据表
    tables: ["customer", "material", "warehouse"],
    
    // 同步策略
    syncStrategy: {
        type: "incremental",  // 增量同步
        interval: 300000,     // 5分钟
        wifiOnly: true        // 仅WiFi同步
    },
    
    // 数据冲突处理
    conflictResolve: "server"  // 服务端优先
}
```

### 6.2 离线操作

```javascript
// 检查网络状态
checkNetwork() {
    if (!this.isOnline()) {
        this.getView().showWarningNotification("当前处于离线状态，数据将在联网后同步");
    }
}

// 离线保存数据
saveOffline(data) {
    // 保存到本地
    this.localStore.save("order_" + data.id, data);
    
    // 标记待同步
    this.syncQueue.add({
        type: "order",
        id: data.id,
        action: "save",
        timestamp: new Date()
    });
}
```
