#### 1.2 Service层测试

```java
@ExtendWith(MockitoExtension.class)
@DisplayName("单据服务测试")
class BillServiceTest {
    
    @Mock
    private BillRepository billRepository;
    
    @Mock
    private CustomerService customerService;
    
    @Mock
    private MessageService messageService;
    
    @InjectMocks
    private BillServiceImpl billService;
    
    private Bill testBill;
    private Customer testCustomer;
    
    @BeforeEach
    void setUp() {
        // 准备测试数据
        testCustomer = new Customer();
        testCustomer.setId(1001L);
        testCustomer.setCode("C001");
        testCustomer.setName("测试客户");
        
        testBill = new Bill();
        testBill.setId(1L);
        testBill.setBillNo("SO20260320001");
        testBill.setCustomerId(1001L);
        testBill.setStatus(BillStatus.DRAFT);
        testBill.setTotalAmount(new BigDecimal("1000.00"));
    }
    
    @Test
    @DisplayName("创建单据-成功")
    void testCreateBill_Success() {
        // Given
        BillCreateInput input = new BillCreateInput();
        input.setCustomerId(1001L);
        input.setBillType(BillType.SALES_ORDER);
        
        when(customerService.getCustomerById(1001L)).thenReturn(testCustomer);
        when(billRepository.save(any(Bill.class))).thenReturn(testBill);
        
        // When
        Bill result = billService.createBill(input);
        
        // Then
        assertNotNull(result);
        assertEquals("SO20260320001", result.getBillNo());
        assertEquals(BillStatus.DRAFT, result.getStatus());
        
        // 验证方法调用
        verify(customerService, times(1)).getCustomerById(1001L);
        verify(billRepository, times(1)).save(any(Bill.class));
        verify(messageService, times(1)).sendCreateNotification(any(Bill.class));
    }
    
    @Test
    @DisplayName("创建单据-客户不存在")
    void testCreateBill_CustomerNotFound() {
        // Given
        BillCreateInput input = new BillCreateInput();
        input.setCustomerId(9999L);
        
        when(customerService.getCustomerById(9999L))
            .thenThrow(new BusinessException("客户不存在"));
        
        // When & Then
        assertThrows(BusinessException.class, () -> {
            billService.createBill(input);
        });
        
        // 验证未保存
        verify(billRepository, never()).save(any(Bill.class));
    }
    
    @ParameterizedTest
    @CsvSource({
        "DRAFT, SUBMITTED",
        "SUBMITTED, AUDITED",
        "AUDITED, CLOSED"
    })
    @DisplayName("单据状态流转测试")
    void testBillStatusTransition(BillStatus from, BillStatus to) {
        testBill.setStatus(from);
        when(billRepository.findById(1L)).thenReturn(Optional.of(testBill));
        
        Bill result = null;
        
        switch (to) {
            case SUBMITTED:
                result = billService.submitBill(1L);
                break;
            case AUDITED:
                result = billService.auditBill(1L, "PASS", "审核通过");
                break;
            case CLOSED:
                result = billService.closeBill(1L);
                break;
        }
        
        assertEquals(to, result.getStatus());
    }
}
```

```csharp
[TestClass]
public class BillServiceTest
{
    private readonly Mock<IBillRepository> _repositoryMock;
    private readonly BillService _billService;
    
    public BillServiceTest()
    {
        _repositoryMock = new Mock<IBillRepository>();
        _billService = new BillService(_repositoryMock.Object);
    }
    
    [TestMethod]
    public async Task CreateBill_ShouldReturnNewBill()
    {
        // Arrange
        var bill = new Bill
        {
            BillNo = "TEST001",
            Amount = 1000m,
            Status = BillStatus.Draft
        };
        
        _repositoryMock
            .Setup(r => r.SaveAsync(It.IsAny<Bill>()))
            .ReturnsAsync((Bill b) => { b.BillId = Guid.NewGuid().ToString(); return b; });
        
        // Act
        var result = await _billService.CreateBillAsync(bill);
        
        // Assert
        Assert.IsNotNull(result.BillId);
        Assert.AreEqual("TEST001", result.BillNo);
        Assert.AreEqual(BillStatus.Draft, result.Status);
    }
    
    [TestMethod]
    [ExpectedException(typeof(BusinessException))]
    public async Task CreateBill_WithDuplicateNumber_ShouldThrowException()
    {
        // Arrange
        var bill = new Bill { BillNo = "EXISTING" };
        _repositoryMock.Setup(r => r.ExistsByBillNoAsync("EXISTING")).ReturnsAsync(true);
        
        // Act
        await _billService.CreateBillAsync(bill);
    }
}
```

### 2. 集成测试

#### 2.1 Spring Boot集成测试

```java
@SpringBootTest(webEnvironment = SpringBootTest.WebEnvironment.RANDOM_PORT)
@AutoConfigureMockMvc
@DisplayName("单据API集成测试")
class BillControllerIntegrationTest {
    
    @Autowired
    private MockMvc mockMvc;
    
    @Autowired
    private ObjectMapper objectMapper;
    
    @Autowired
    private BillRepository billRepository;
    
    @Autowired
    private CustomerRepository customerRepository;
    
    private Customer testCustomer;
    private String accessToken;
    
    @BeforeEach
    void setUp() throws Exception {
        // 清理数据
        billRepository.deleteAll();
        customerRepository.deleteAll();
        
        // 创建测试客户
        testCustomer = new Customer();
        testCustomer.setCode("C001");
        testCustomer.setName("测试客户");
        testCustomer = customerRepository.save(testCustomer);
        
        // 获取访问令牌
        accessToken = obtainAccessToken("admin", "admin123");
    }
    
    @Test
    @DisplayName("创建单据-集成测试")
    void testCreateBill() throws Exception {
        // Given
        BillCreateRequest request = new BillCreateRequest();
        request.setCustomerId(testCustomer.getId());
        request.setBillType("SALES_ORDER");
        request.setEntries(List.of(
            createEntryRequest("M001", 10, new BigDecimal("100.00")),
            createEntryRequest("M002", 20, new BigDecimal("50.00"))
        ));
        
        // When & Then
        mockMvc.perform(post("/api/bills")
            .header("Authorization", "Bearer " + accessToken)
            .contentType(MediaType.APPLICATION_JSON)
            .content(objectMapper.writeValueAsString(request)))
            .andExpect(status().isCreated())
            .andExpect(jsonPath("$.billNo").isNotEmpty())
            .andExpect(jsonPath("$.status").value("DRAFT"))
            .andExpect(jsonPath("$.totalAmount").value(2000.00))
            .andExpect(jsonPath("$.customerId").value(testCustomer.getId()));
        
        // 验证数据库
        List<Bill> bills = billRepository.findAll();
        assertEquals(1, bills.size());
        assertEquals(2, bills.get(0).getEntries().size());
    }
    
    @Test
    @DisplayName("单据操作-状态流转")
    void testBillWorkflow() throws Exception {
        // 1. 创建单据
        String billJson = mockMvc.perform(post("/api/bills")
            .header("Authorization", "Bearer " + accessToken)
            .contentType(MediaType.APPLICATION_JSON)
            .content(objectMapper.writeValueAsString(createBillRequest())))
            .andExpect(status().isCreated())
            .andReturn().getResponse().getContentAsString();
        
        Long billId = JsonPath.parse(billJson).read("$.id", Long.class);
        
        // 2. 提交单据
        mockMvc.perform(post("/api/bills/{id}/submit", billId)
            .header("Authorization", "Bearer " + accessToken))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.status").value("SUBMITTED"));
        
        // 3. 审核单据
        mockMvc.perform(post("/api/bills/{id}/audit", billId)
            .header("Authorization", "Bearer " + accessToken)
            .param("result", "PASS")
            .param("opinion", "审核通过"))
            .andExpect(status().isOk())
            .andExpect(jsonPath("$.status").value("AUDITED"));
    }
}
```

### 1. 应用监控

```csharp
// 使用Micrometer记录指标
private readonly Counter _requestCounter;
private readonly Timer _requestTimer;

public BillService(MeterRegistry registry)
{
    _requestCounter = registry.counter("http.requests", "method", "uri");
    _requestTimer = registry.timer("http.requests.duration");
}

public async Task<Bill> GetBillAsync(string billId)
{
    var timer = _requestTimer.record();
    _requestCounter.increment("GET", "/api/bills/{id}");
    
    try
    {
        return await _repository.GetByIdAsync(billId);
    }
    finally
    {
        timer.stop();
    }
}
```

### 2. 注释规范

```csharp
/// <summary>
/// 单据服务
/// 负责单据的创建、更新、删除、查询等操作
/// </summary>
public class BillService
{
    /// <summary>
    /// 创建单据
    /// </summary>
    /// <param name="bill">单据对象</param>
    /// <returns>创建的单据</returns>
    /// <exception cref="BusinessException">当单据编号已存在时抛出</exception>
    public async Task<Bill> CreateBillAsync(Bill bill)
    {
        // 验证单据编号是否已存在
        if (await _repository.ExistsByBillNoAsync(bill.BillNo))
        {
            throw new BusinessException("单据编号已存在");
        }
        
        // TODO: 添加业务验证
        
        return await _repository.SaveAsync(bill);
    }
}
```
