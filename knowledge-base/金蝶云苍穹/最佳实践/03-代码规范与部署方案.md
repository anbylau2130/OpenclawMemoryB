**异常处理**:
```java
// 自定义业务异常
public class BusinessException extends RuntimeException {
    private String errorCode;
    private Object[] args;
    
    public BusinessException(String message) {
        super(message);
    }
    
    public BusinessException(String errorCode, String message) {
        super(message);
        this.errorCode = errorCode;
    }
    
    public BusinessException(String errorCode, String message, Object... args) {
        super(message);
        this.errorCode = errorCode;
        this.args = args;
    }
}

// 全局异常处理
@RestControllerAdvice
@Slf4j
public class GlobalExceptionHandler {
    
    @ExceptionHandler(BusinessException.class)
    public ResponseEntity<ErrorResponse> handleBusinessException(
            BusinessException e) {
        log.warn("业务异常: {}", e.getMessage());
        
        ErrorResponse response = new ErrorResponse();
        response.setCode(e.getErrorCode());
        response.setMessage(e.getMessage());
        response.setTimestamp(LocalDateTime.now());
        
        return ResponseEntity.badRequest().body(response);
    }
    
    @ExceptionHandler(Exception.class)
    public ResponseEntity<ErrorResponse> handleException(Exception e) {
        log.error("系统异常", e);
        
        ErrorResponse response = new ErrorResponse();
        response.setCode("SYSTEM_ERROR");
        response.setMessage("系统异常，请联系管理员");
        response.setTimestamp(LocalDateTime.now());
        
        return ResponseEntity.status(500).body(response);
    }
}

// 服务层异常处理
@Service
@Slf4j
public class BillServiceImpl implements BillService {
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public Bill createBill(BillCreateInput input) {
        try {
            // 业务逻辑
            return doCreateBill(input);
        } catch (BusinessException e) {
            // 业务异常直接抛出
            throw e;
        } catch (Exception e) {
            // 其他异常包装为业务异常
            log.error("创建单据失败", e);
            throw new BusinessException("CREATE_BILL_ERROR", "创建单据失败");
        }
    }
}
```

## 🔄 CI/CD流程

### 1. Git分支管理

**分支策略**:
```
master (生产分支)
  └── release/* (发布分支)
        └── develop (开发分支)
              └── feature/* (功能分支)
              └── bugfix/* (修复分支)
              └── hotfix/* (紧急修复分支)
```

**Git提交规范**:
```
<type>(<scope>): <subject>

<body>

<footer>

type类型：
- feat: 新功能
- fix: 修复bug
- docs: 文档更新
- style: 代码格式
- refactor: 重构
- test: 测试
- chore: 构建/工具

示例：

- 新增批量审核接口
- 新增批量审核页面

Closes #123
```

### 2. Jenkins Pipeline

**Jenkinsfile**:
```groovy
pipeline {
    agent any
    
    environment {
        DOCKER_REGISTRY = 'registry.example.com'
        IMAGE_NAME = 'k3cloud'
        VERSION = "${env.BUILD_ID}"
    }
    
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        
        stage('Build') {
            steps {
                sh 'mvn clean package -DskipTests'
            }
        }
        
        stage('Test') {
            steps {
                sh 'mvn test'
                junit '**/target/surefire-reports/*.xml'
            }
        }
        
        stage('SonarQube') {
            steps {
                withSonarQubeEnv('sonarqube') {
                    sh 'mvn sonar:sonar'
                }
            }
        }
        
        stage('Quality Gate') {
            steps {
                timeout(time: 5, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
        }
        
        stage('Build Image') {
            steps {
                script {
                    docker.withRegistry("https://${DOCKER_REGISTRY}", 
                        'docker-registry-credentials') {
                        def image = docker.build(
                            "${DOCKER_REGISTRY}/${IMAGE_NAME}:${VERSION}")
                        image.push()
                        image.push('latest')
                    }
                }
            }
        }
        
        stage('Deploy to Dev') {
            when {
                branch 'develop'
            }
            steps {
                sh "kubectl set image deployment/k3cloud " +
                   "k3cloud=${DOCKER_REGISTRY}/${IMAGE_NAME}:${VERSION} " +
                   "-n dev"
            }
        }
        
        stage('Deploy to Test') {
            when {
                branch 'release/*'
            }
            steps {
                sh "kubectl set image deployment/k3cloud " +
                   "k3cloud=${DOCKER_REGISTRY}/${IMAGE_NAME}:${VERSION} " +
                   "-n test"
            }
        }
        
        stage('Deploy to Prod') {
            when {
                branch 'master'
            }
            steps {
                input 'Deploy to Production?'
                sh "kubectl set image deployment/k3cloud " +
                   "k3cloud=${DOCKER_REGISTRY}/${IMAGE_NAME}:${VERSION} " +
                   "-n prod"
            }
        }
    }
    
    post {
        always {
            cleanWs()
        }
        success {
            dingtalk(
                robot: 'dingtalk-robot',
                type: 'MARKDOWN',
                title: '构建成功',
                text: ["## 构建成功\n- 项目: ${env.JOB_NAME}\n- 版本: ${VERSION}\n- 构建号: ${env.BUILD_NUMBER}"]
            )
        }
        failure {
            dingtalk(
                robot: 'dingtalk-robot',
                type: 'MARKDOWN',
                title: '构建失败',
                text: ["## 构建失败\n- 项目: ${env.JOB_NAME}\n- 版本: ${VERSION}\n- 构建号: ${env.BUILD_NUMBER}"]
            )
        }
    }
}
```

### 1. 灰度发布策略

```yaml
spec:

  - k3cloud

    - destination:

kind: DestinationRule
metadata:
  name: k3cloud
spec:
  host: k3cloud
  subsets:
  - name: stable
    labels:
      version: stable
  - name: canary
    labels:
      version: canary
```

**灰度发布脚本**:
```bash
#!/bin/bash
# 灰度发布脚本

NAMESPACE="prod"
SERVICE="k3cloud"
STABLE_VERSION="v1.0.0"
CANARY_VERSION="v2.0.0"

# 灰度比例数组
WEIGHTS=(10 30 50 80 100)

# 健康检查
health_check() {
    local version=$1
    local pods=$(kubectl get pods -n $NAMESPACE -l app=$SERVICE,version=$version -o json)
    
    echo "检查 $version 版本健康状态..."
    
    ready=$(echo $pods | jq -r '.items[] | select(.status.phase=="Running") | .status.containerStatuses[0].ready' | grep -c true)
    total=$(echo $pods | jq -r '.items | length')
    
    if [ $ready -eq $total ]; then
        echo "$version 版本健康检查通过"
        return 0
    else
        echo "$version 版本健康检查失败: $ready/$total"
        return 1
    fi
}

# 更新流量比例
update_weight() {
    local weight=$1
    
    echo "更新灰度比例为 $weight%..."
    
    cat <<EOF | kubectl apply -f -
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: $SERVICE
spec:
  hosts:
  - $SERVICE
  http:
  - route:
    - destination:
        host: $SERVICE
        subset: stable
      weight: $((100 - weight))
    - destination:
        host: $SERVICE
        subset: canary
      weight: $weight
EOF
    
    echo "等待流量切换..."
    sleep 60
}

# 监控指标
monitor_metrics() {
    echo "监控灰度版本指标..."
    
    # 调用监控API
    error_rate=$(curl -s "http://prometheus:9090/api/v1/query?query=rate(http_requests_total{status=~\"5..\"}[5m])" | jq -r '.data.result[0].value[1]')
    
    if (( $(echo "$error_rate > 0.05" | bc -l) )); then
        echo "错误率过高: $error_rate"
        return 1
    fi
    
    echo "指标正常，错误率: $error_rate"
    return 0
}

# 回滚
rollback() {
    echo "执行回滚..."
    
    # 删除灰度版本
    kubectl delete deployment $SERVICE-canary -n $NAMESPACE
    
    # 恢复流量到稳定版本
    cat <<EOF | kubectl apply -f -
apiVersion: networking.istio.io/v1alpha3
kind: VirtualService
metadata:
  name: $SERVICE
spec:
  hosts:
  - $SERVICE
  http:
  - route:
    - destination:
        host: $SERVICE
        subset: stable
      weight: 100
EOF
    
    echo "回滚完成"
}

# 主流程
main() {
    echo "开始灰度发布..."
    
    # 1. 部署灰度版本
    kubectl apply -f deployment-canary.yaml
    
    # 2. 等待灰度版本就绪
    if ! health_check $CANARY_VERSION; then
        echo "灰度版本部署失败"
        exit 1
    fi
    
    # 3. 逐步增加流量
    for weight in "${WEIGHTS[@]}"; do
        update_weight $weight
        
        # 监控指标
        if ! monitor_metrics; then
            echo "监控指标异常，执行回滚"
            rollback
            exit 1
        fi
        
        # 确认继续
        if [ $weight -lt 100 ]; then
            read -p "是否继续增加流量? (y/n) " -n 1 -r
            echo
            if [[ ! $REPLY =~ ^[Yy]$ ]]; then
                echo "停止灰度发布"
                exit 0
            fi
        fi
    done
    
    # 4. 灰度完成，更新稳定版本
    kubectl set image deployment/$SERVICE-stable $SERVICE=$CANARY_VERSION -n $NAMESPACE
    
    # 5. 删除灰度版本
    kubectl delete deployment $SERVICE-canary -n $NAMESPACE
    
    echo "灰度发布完成"
}

main
```
