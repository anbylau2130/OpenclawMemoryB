**工作流服务（C#）**:
```csharp
using K3Cloud.Workflow;
using K3Cloud.Workflow.Engine;

public class WorkflowService : IWorkflowService
{
    private readonly IWorkflowEngine _engine;
    private readonly IWorkflowRepository _repository;
    private readonly ITaskRepository _taskRepository;
    private readonly INotificationService _notification;
    private readonly ILogger<WorkflowService> _logger;
    
    public WorkflowService(
        IWorkflowEngine engine,
        IWorkflowRepository repository,
        ITaskRepository taskRepository,
        INotificationService notification,
        ILogger<WorkflowService> logger)
    {
        _engine = engine;
        _repository = repository;
        _taskRepository = taskRepository;
        _notification = notification;
        _logger = logger;
    }
    
    /// <summary>
    /// 启动工作流实例
    /// </summary>
    public async Task<WorkflowInstance> StartWorkflowAsync(
        string workflowId,
        Dictionary<string, object> variables,
        string initiator)
    {
        // 加载工作流定义
        var workflow = await _repository.GetByIdAsync(workflowId);
        if (workflow == null)
        {
            throw new WorkflowNotFoundException($"工作流 {workflowId} 不存在");
        }
        
        // 创建工作流实例
        var instance = new WorkflowInstance
        {
            WorkflowId = workflowId,
            WorkflowName = workflow.WorkflowName,
            Initiator = initiator,
            Status = "running",
            StartTime = DateTime.UtcNow,
            Variables = variables
        };
        
        // 设置初始变量
        instance.SetVariable("initiator", initiator);
        foreach (var variable in variables)
        {
            instance.SetVariable(variable.Key, variable.Value);
        }
        
        // 启动工作流引擎
        await _engine.StartAsync(instance);
        
        // 保存实例
        await _repository.SaveInstanceAsync(instance);
        
        _logger.LogInformation(
            "工作流实例已启动: {InstanceId}, 工作流: {WorkflowId}, 发起人: {Initiator}",
            instance.InstanceId,
            workflowId,
            initiator
        );
        
        return instance;
    }
    
    /// <summary>
    /// 完成用户任务
    /// </summary>
    public async Task<TaskInfo> CompleteTaskAsync(
        string taskId,
        string userId,
        Dictionary<string, object> variables,
        string comment)
    {
        // 加载任务
        var task = await _taskRepository.GetByIdAsync(taskId);
        if (task == null)
        {
            throw new TaskNotFoundException($"任务 {taskId} 不存在");
        }
        
        // 验证任务权限
        if (!task.CanComplete(userId))
        {
            throw new TaskPermissionException("用户没有权限完成此任务");
        }
        
        // 验证任务状态
        if (task.Status != "active")
        {
            throw new TaskStatusException($"任务状态错误: {task.Status}");
        }
        
        // 设置任务变量
        foreach (var variable in variables)
        {
            task.SetVariable(variable.Key, variable.Value);
        }
        
        task.Comment = comment;
        task.Assignee = userId;
        task.CompleteTime = DateTime.UtcNow;
        
        // 完成任务
        await _engine.CompleteTaskAsync(task);
        
        // 更新任务状态
        task.Status = "completed";
        await _taskRepository.UpdateAsync(task);
        
        _logger.LogInformation(
            "任务已完成: {TaskId}, 用户: {UserId}, 变量: {@Variables}, 评论: {Comment}",
            taskId,
            userId,
            variables,
            comment
        );
        
        return task;
    }
    
    /// <summary>
    /// 委派任务
    /// </summary>
    public async Task<TaskInfo> DelegateTaskAsync(
        string taskId,
        string fromUserId,
        string toUserId,
        string comment)
    {
        var task = await _taskRepository.GetByIdAsync(taskId);
        if (task == null)
        {
            throw new TaskNotFoundException($"任务 {taskId} 不存在");
        }
        
        if (!task.CanDelegate(fromUserId))
        {
            throw new TaskPermissionException("用户没有权限委派此任务");
        }
        
        // 创建新任务
        var newTask = new TaskInfo
        {
            TaskId = Guid.NewGuid().ToString(),
            InstanceId = task.InstanceId,
            TaskName = task.TaskName,
            TaskType = task.TaskType,
            Assignee = toUserId,
            Status = "active",
            CreateTime = DateTime.UtcNow,
            Variables = task.Variables
        };
        
        // 完成原任务
        task.Status = "delegated";
        task.DelegateTo = toUserId;
        task.Comment = comment;
        await _taskRepository.UpdateAsync(task);
        
        // 保存新任务
        await _taskRepository.SaveAsync(newTask);
        
        // 发送通知
        await _notification.SendTaskNotificationAsync(newTask);
        
        _logger.LogInformation(
            "任务已委派: {TaskId}, 从用户: {FromUserId} 到用户: {ToUserId}, 评论: {Comment}",
            taskId,
            fromUserId,
            toUserId,
            comment
        );
        
        return newTask;
    }
    
    /// <summary>
    /// 获取用户待办任务
    /// </summary>
    public async Task<List<TaskInfo>> GetUserTasksAsync(
        string userId,
        TaskQuery query)
    {
        return await _taskRepository.GetByAssigneeAsync(userId, query);
    }
    
    /// <summary>
    /// 获取流程实例详情
    /// </summary>
    public async Task<WorkflowInstanceDetail> GetInstanceDetailAsync(
        string instanceId)
    {
        var instance = await _repository.GetInstanceByIdAsync(instanceId);
        var tasks = await _taskRepository.GetByInstanceIdAsync(instanceId);
        
        return new WorkflowInstanceDetail
        {
            Instance = instance,
            Tasks = tasks,
            ProcessDiagram = await _engine.GetProcessDiagramAsync(instanceId)
        };
    }
}
```

```csharp
using K3Cloud.Approval;

public class ApprovalInstance
{
    public string InstanceId { get; set; }
    public string WorkflowId { get; set; }
    public string BusinessKey { get; set; }
    public string Initiator { get; set; }
    public string Status { get; set; }  // running, approved, rejected
    public DateTime StartTime { get; set; }
    public DateTime? EndTime { get; set; }
    public List<ApprovalNode> Nodes { get; set; }
    public List<ApprovalRecord> Records { get; set; }
    public Dictionary<string, object> Variables { get; set; }
}

public class ApprovalNode
{
    public string NodeId { get; set; }
    public string NodeName { get; set; }
    public string NodeType { get; set; }  // start, approval, end
    public string Operator { get; set; }
    public string OperatorType { get; set; }  // user, role
    public string Status { get; set; }  // pending, approved, rejected, skipped
    public DateTime? ApproveTime { get; set; }
    public string Comment { get; set; }
    public List<ApprovalRecord> Records { get; set; }
}

public class ApprovalRecord
{
    public string RecordId { get; set; }
    public string InstanceId { get; set; }
    public string NodeId { get; set; }
    public string Approver { get; set; }
    public string Decision { get; set; }  // approve, reject, delegate
    public string Comment { get; set; }
    public DateTime ApproveTime { get; set; }
}
```

```csharp
using K3Cloud.Workflow.Monitoring;

public class WorkflowMonitorService : IWorkflowMonitorService
{
    private readonly IMetricsCollector _metrics;
    private readonly IAlertService _alert;
    private readonly IWorkflowRepository _repository;
    private readonly ILogger<WorkflowMonitorService> _logger;
    
    public WorkflowMonitorService(
        IMetricsCollector metrics,
        IAlertService alert,
        IWorkflowRepository repository,
        ILogger<WorkflowMonitorService> logger)
    {
        _metrics = metrics;
        _alert = alert;
        _repository = repository;
        _logger = logger;
    }
    
    /// <summary>
    /// 记录工作流启动
    /// </summary>
    public async Task RecordWorkflowStartAsync(string instanceId, string workflowId, string initiator)
    {
        await _metrics.IncrementAsync("workflow_start_count", new Dictionary<string, string>
        {
            { "workflow_id", workflowId },
            { "initiator", initiator },
            { "department", await GetDepartment(initiator) }
        });
        
        _logger.LogInformation(
            "工作流已启动: {InstanceId}, 工作流: {WorkflowId}, 发起人: {Initiator}",
            instanceId,
            workflowId,
            initiator
        );
    }
    
    /// <summary>
    /// 记录工作流完成
    /// </summary>
    public async Task RecordWorkflowCompleteAsync(
        string instanceId,
        string workflowId,
        string status,
        TimeSpan duration)
    {
        await _metrics.RecordAsync("workflow_completion_time", duration.TotalSeconds, new Dictionary<string, string>
        {
            { "workflow_id", workflowId },
            { "instance_id", instanceId },
            { "status", status }
        });
        
        // 检查超时告警
        await CheckWorkflowTimeoutAlert(instanceId, workflowId, duration);
        
        _logger.LogInformation(
            "工作流已完成: {InstanceId}, 工作流: {WorkflowId}, 状态: {Status}, 耗时: {Duration}s",
            instanceId,
            workflowId,
            status,
            duration.TotalSeconds
        );
    }
    
    /// <summary>
    /// 记录任务完成
    /// </summary>
    public async Task RecordTaskCompleteAsync(
        string taskId,
        string taskName,
        string assignee,
        string decision,
        TimeSpan duration)
    {
        await _metrics.RecordAsync("task_pending_time", duration.TotalSeconds, new Dictionary<string, string>
        {
            { "task_id", taskId },
            { "task_name", taskName },
            { "assignee", assignee }
        });
        
        // 检查任务超时告警
        await CheckTaskTimeoutAlert(taskId, taskName, assignee, duration);
        
        _logger.LogInformation(
            "任务已完成: {TaskId}, 任务: {TaskName}, 处理人: {Assignee}, 决策: {Decision}, 耗时: {Duration}s",
            taskId,
            taskName,
            assignee,
            decision,
            duration.TotalSeconds
        );
    }
    
    /// <summary>
    /// 记录审批决策
    /// </summary>
    public async Task RecordApprovalDecisionAsync(
        string instanceId,
        string nodeId,
        string approver,
        string decision)
    {
        await _metrics.IncrementAsync("approval_decision_count", new Dictionary<string, string>
        {
            { "instance_id", instanceId },
            { "node_id", nodeId },
            { "approver", approver },
            { "decision", decision }
        });
        
        // 计算审批通过率
        await CalculateApprovalPassRateAsync(approver, nodeId);
    }
    
    private async Task CheckWorkflowTimeoutAlert(string instanceId, string workflowId, TimeSpan duration)
    {
        var threshold = TimeSpan.FromHours(2);  // 2小时
        
        if (duration > threshold)
        {
            await _alert.SendAlertAsync(new Alert
            {
                AlertId = "workflow_timeout",
                AlertName = "流程超时告警",
                Severity = "warning",
                Message = $"工作流实例 {instanceId} 已超时 {duration.TotalHours:F2} 小时",
                Context = new Dictionary<string, object>
                {
                    { "instance_id", instanceId },
                    { "workflow_id", workflowId },
                    { "duration", duration.TotalSeconds }
                }
            });
        }
    }
    
    private async Task CheckTaskTimeoutAlert(
        string taskId,
        string taskName,
        string assignee,
        TimeSpan duration)
    {
        var threshold = TimeSpan.FromHours(1);  // 1小时
        
        if (duration > threshold)
        {
            await _alert.SendAlertAsync(new Alert
            {
                AlertId = "task_timeout",
                AlertName = "任务超时告警",
                Severity = "warning",
                Message = $"任务 {taskId} ({taskName}) 已等待 {duration.TotalHours:F2} 小时未处理",
                Context = new Dictionary<string, object>
                {
                    { "task_id", taskId },
                    { "task_name", taskName },
                    { "assignee", assignee },
                    { "duration", duration.TotalSeconds }
                }
            });
        }
    }
    
    private async Task CalculateApprovalPassRateAsync(string approver, string nodeId)
    {
        // 获取审批人的所有审批记录
        var records = await _repository.GetApprovalRecordsAsync(approver, nodeId);
        
        if (records.Count == 0)
        {
            return;
        }
        
        // 计算通过率
        var passCount = records.Count(r => r.Decision == "approve");
        var passRate = (double)passCount / records.Count;
        
        await _metrics.RecordAsync("approval_pass_rate", passRate, new Dictionary<string, string>
        {
            { "approver", approver },
            { "node_id", nodeId }
        });
        
        // 检查通过率告警
        if (passRate < 0.5)
        {
            await _alert.SendAlertAsync(new Alert
            {
                AlertId = "low_approval_pass_rate",
                AlertName = "审批通过率低",
                Severity = "warning",
                Message = $"审批人员 {approver} 的通过率低于 50%",
                Context = new Dictionary<string, object>
                {
                    { "approver", approver },
                    { "node_id", nodeId },
                    { "pass_rate", passRate }
                }
            });
        }
    }
}
```

## 📝 最佳实践

### 1. 工作流设计
- 流程简化：避免不必要的审批环节
- 权限控制：明确每个节点的操作权限
- 异常处理：提供流程挂起、撤回等功能
- 流程优化：定期分析流程效率，优化流程

### 2. 审批管理
- 审批规则：设置合理的审批金额阈值
- 委派代理：支持任务委派和代理
- 并行审批：提高审批效率
- 审批记录：完整记录审批过程

### 3. 流程监控
- 关键指标：监控流程启动、完成、耗时
- 异常告警：及时发现和处理异常情况
- 效率分析：分析流程效率，优化流程设计
- 数据分析：定期分析流程数据，持续改进
