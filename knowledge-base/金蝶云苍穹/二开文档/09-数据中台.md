### 2. 数据清洗

**数据清洗规则（C#）**:
```csharp
using K3Cloud.DataWarehouse.Cleaning;

public class DataCleaner : IDataCleaner
{
    private readonly ILogger<DataCleaner> _logger;
    
    public DataCleaner(ILogger<DataCleaner> logger)
    {
        _logger = logger;
    }
    
    public async Task<CleanResult> CleanAsync(DataBatch batch)
    {
        var result = new CleanResult
        {
            OriginalCount = batch.Records.Count,
            CleanedRecords = new List<DataRecord>()
        };
        
        foreach (var record in batch.Records)
        {
            try
            {
                var cleanedRecord = await CleanRecordAsync(record);
                result.CleanedRecords.Add(cleanedRecord);
                result.CleanedCount++;
            }
            catch (Exception ex)
            {
                _logger.LogError(ex, "清洗记录失败: {RecordId}", record.RecordId);
                result.ErrorCount++;
            }
        }
        
        return result;
    }
    
    private async Task<DataRecord> CleanRecordAsync(DataRecord record)
    {
        // 1. 去重
        if (await IsDuplicateAsync(record))
        {
            throw new DuplicateException("记录已存在");
        }
        
        // 2. 数据标准化
        record = StandardizeRecord(record);
        
        // 3. 数据验证
        ValidateRecord(record);
        
        // 4. 数据补全
        record = EnrichRecord(record);
        
        return record;
    }
    
    private async Task<bool> IsDuplicateAsync(DataRecord record)
    {
        // 检查是否重复
        var exists = await CheckExistenceAsync(record);
        return exists;
    }
    
    private DataRecord StandardizeRecord(DataRecord record)
    {
        // 数据标准化处理
        
        // 字符串标准化：去除首尾空格、统一大小写
        foreach (var field in record.Fields.Where(f => f.Type == "string"))
        {
            field.Value = field.Value.ToString()?.Trim();
            
            // 字段名标准化
            if (field.Name.Equals("customer_name", StringComparison.OrdinalIgnoreCase))
            {
                field.Name = "customer_name";
            }
        }
        
        // 日期标准化
        foreach (var field in record.Fields.Where(f => f.Type == "date"))
        {
            if (DateTime.TryParse(field.Value?.ToString(), out var date))
            {
                field.Value = date.ToString("yyyy-MM-dd");
            }
        }
        
        // 金额标准化：保留2位小数
        foreach (var field in record.Fields.Where(f => f.Type == "decimal"))
        {
            if (decimal.TryParse(field.Value?.ToString(), out var amount))
            {
                field.Value = amount.ToString("F2");
            }
        }
        
        return record;
    }
    
    private void ValidateRecord(DataRecord record)
    {
        // 数据验证
        
        // 必填字段验证
        var requiredFields = new[] { "customer_id", "product_id", "date_id", "amount" };
        foreach (var field in requiredFields)
        {
            if (record.Fields.All(f => f.Name != field) || 
                string.IsNullOrEmpty(record.Fields.First(f => f.Name == field).Value?.ToString()))
            {
                throw new ValidationException($"必填字段{field}不能为空");
            }
        }
        
        // 金额验证
        var amountField = record.Fields.First(f => f.Name == "amount");
        if (decimal.TryParse(amountField.Value?.ToString(), out var amount) && amount < 0)
        {
            throw new ValidationException("金额不能为负数");
        }
        
        // 日期验证
        var dateField = record.Fields.First(f => f.Name == "date_id");
        if (int.TryParse(dateField.Value?.ToString(), out var dateId))
        {
            var date = new DateTime(dateId / 10000, dateId % 10000 / 100, dateId % 100);
            if (date > DateTime.Now)
            {
                throw new ValidationException("日期不能晚于当前日期");
            }
        }
    }
    
    private async Task<DataRecord> EnrichRecord(DataRecord record)
    {
        // 数据补全：添加额外的维度和指标
        
        // 添加创建时间
        if (!record.Fields.Any(f => f.Name == "create_time"))
        {
            record.Fields.Add(new DataField
            {
                Name = "create_time",
                Type = "datetime",
                Value = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss")
            });
        }
        
        // 添加更新时间
        record.Fields.Add(new DataField
        {
            Name = "update_time",
            Type = "datetime",
            Value = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss")
        });
        
        // 添加数据源标识
        record.Fields.Add(new DataField
        {
            Name = "data_source",
            Type = "string",
            Value = "k3cloud"
        });
        
        // 如果有利润计算需求
        var amountField = record.Fields.FirstOrDefault(f => f.Name == "amount");
        var costField = record.Fields.FirstOrDefault(f => f.Name == "cost");
        
        if (amountField != null && costField != null)
        {
            if (decimal.TryParse(amountField.Value?.ToString(), out var amount) &&
                decimal.TryParse(costField.Value?.ToString(), out var cost))
            {
                var profit = amount - cost;
                record.Fields.Add(new DataField
                {
                    Name = "profit",
                    Type = "decimal",
                    Value = profit.ToString("F2")
                });
            }
        }
        
        return record;
    }
}
```

### 1. 数据计算

**聚合计算（SQL）**:
```sql
-- 每日销售额
SELECT 
    date_id,
    SUM(amount) AS daily_sales,
    SUM(profit) AS daily_profit,
    COUNT(DISTINCT order_id) AS daily_orders
FROM fact_sales
GROUP BY date_id
ORDER BY date_id DESC;

-- 每月销售额
SELECT 
    year,
    month,
    SUM(amount) AS monthly_sales,
    SUM(profit) AS monthly_profit,
    COUNT(DISTINCT order_id) AS monthly_orders
FROM fact_sales
GROUP BY year, month
ORDER BY year DESC, month DESC;

-- 客户销售额Top10
SELECT 
    c.customer_id,
    c.customer_name,
    SUM(s.amount) AS total_sales,
    SUM(s.profit) AS total_profit,
    COUNT(DISTINCT s.order_id) AS order_count
FROM fact_sales s
JOIN dim_customer c ON s.customer_id = c.customer_id
GROUP BY c.customer_id, c.customer_name
ORDER BY total_sales DESC
LIMIT 10;

-- 产品销售额Top10
SELECT 
    p.product_id,
    p.product_name,
    SUM(s.amount) AS total_sales,
    SUM(s.quantity) AS total_quantity,
    AVG(s.amount) AS avg_order_amount
FROM fact_sales s
JOIN dim_product p ON s.product_id = p.product_id
GROUP BY p.product_id, p.product_name
ORDER BY total_sales DESC
LIMIT 10;

-- 地区销售分析
SELECT 
    c.region,
    SUM(s.amount) AS regional_sales,
    SUM(s.profit) AS regional_profit,
    COUNT(DISTINCT c.customer_id) AS customer_count
FROM fact_sales s
JOIN dim_customer c ON s.customer_id = c.customer_id
GROUP BY c.region
ORDER BY regional_sales DESC;
```

```python
    c.customer_id,
    c.customer_name,
    COUNT(DISTINCT s.order_id) AS purchase_count,
    SUM(s.amount) AS total_amount,
    MAX(s.date_id) AS last_purchase_date,
    MIN(s.date_id) AS first_purchase_date
FROM fact_sales s
JOIN dim_customer c ON s.customer_id = c.customer_id
GROUP BY c.customer_id, c.customer_name
""", engine)

# 计算RFM指标
customer_data['recency'] = (current_date - customer_data['last_purchase_date']).dt.days
customer_data['frequency'] = customer_data['purchase_count']
customer_data['monetary'] = customer_data['total_amount']

# RFM标准化
scaler = StandardScaler()
rfm_normalized = scaler.fit_transform(customer_data[['recency', 'frequency', 'monetary']])

# K-means聚类
kmeans = KMeans(n_clusters=5, random_state=42)
```

### 2. 数据质量

**数据质量规则（C#）:
```csharp
using K3Cloud.DataQuality;

public class DataQualityChecker : IDataQualityChecker
{
    private readonly ILogger<DataQualityChecker> _logger;
    private readonly IDataRepository _repository;
    
    public DataQualityChecker(
        ILogger<DataQualityChecker> logger,
        IDataRepository repository)
    {
        _logger = logger;
        _repository = repository;
    }
    
    public async Task<QualityReport> CheckQualityAsync(string tableName, QualityRules rules)
    {
        var report = new QualityReport
        {
            TableName = tableName,
            CheckTime = DateTime.UtcNow,
            Rules = rules,
            Issues = new List<QualityIssue>()
        };
        
        // 1. 完整性检查
        await CheckCompletenessAsync(report);
        
        // 2. 唯一性检查
        await CheckUniquenessAsync(report);
        
        // 3. 准确性检查
        await CheckAccuracyAsync(report);
        
        // 4. 及时性检查
        await CheckTimelinessAsync(report);
        
        // 5. 一致性检查
        await CheckConsistencyAsync(report);
        
        // 计算质量得分
        report.QualityScore = CalculateQualityScore(report);
        
        return report;
    }
    
    private async Task CheckCompletenessAsync(QualityReport report)
    {
        foreach (var rule in report.Rules.CompletenessRules)
        {
            // 检查空值
            var nullCount = await _repository.GetNullCountAsync(report.TableName, rule.ColumnName);
            var totalCount = await _repository.GetTotalCountAsync(report.TableName);
            var nullRatio = (double)nullCount / totalCount;
            
            if (nullRatio > rule.AllowedNullRatio)
            {
                report.Issues.Add(new QualityIssue
                {
                    IssueType = "Completeness",
                    Severity = nullRatio > rule.AllowedNullRatio * 2 ? "Critical" : "Warning",
                    Description = $"字段{rule.ColumnName}空值比例{nullRatio:P2}超过阈值{rule.AllowedNullRatio:P2}",
                    ColumnName = rule.ColumnName,
                    ActualValue = nullRatio,
                    ExpectedValue = rule.AllowedNullRatio
                });
            }
        }
    }
    
    private async Task CheckUniquenessAsync(QualityReport report)
    {
        foreach (var rule in report.Rules.UniquenessRules)
        {
            // 检查唯一性
            var duplicateCount = await _repository.GetDuplicateCountAsync(report.TableName, rule.ColumnName);
            
            if (duplicateCount > 0)
            {
                report.Issues.Add(new QualityIssue
                {
                    IssueType = "Uniqueness",
                    Severity = "Warning",
                    Description = $"字段{rule.ColumnName}存在{duplicateCount}条重复记录",
                    ColumnName = rule.ColumnName,
                    ActualValue = duplicateCount,
                    ExpectedValue = 0
                });
            }
        }
    }
    
    private async Task CheckAccuracyAsync(QualityReport report)
    {
        foreach (var rule in report.Rules.AccuracyRules)
        {
            // 检查数据格式
            var invalidCount = await _repository.GetInvalidCountAsync(
                report.TableName, 
                rule.ColumnName, 
                rule.Pattern);
            
            if (invalidCount > 0)
            {
                report.Issues.Add(new QualityIssue
                {
                    IssueType = "Accuracy",
                    Severity = "Warning",
                    Description = $"字段{rule.ColumnName}有{invalidCount}条记录不符合格式{rule.Pattern}",
                    ColumnName = rule.ColumnName,
                    ActualValue = invalidCount,
                    ExpectedValue = 0
                });
            }
        }
    }
    
    private async Task CheckTimelinessAsync(QualityReport report)
    {
        // 检查数据及时性
        var maxDate = await _repository.GetMaxDateAsync(report.TableName, "update_time");
        var delay = (DateTime.UtcNow - maxDate).TotalHours;
        
        if (delay > report.Rules.MaxDataDelayHours)
        {
            report.Issues.Add(new QualityIssue
            {
                IssueType = "Timeliness",
                Severity = "Warning",
                Description = $"数据延迟{delay:F1}小时，超过阈值{report.Rules.MaxDataDelayHours}小时",
                ColumnName = "update_time",
                ActualValue = delay,
                ExpectedValue = report.Rules.MaxDataDelayHours
            });
        }
    }
    
    private async Task CheckConsistencyAsync(QualityReport report)
    {
        // 检查数据一致性
        foreach (var rule in report.Rules.ConsistencyRules)
        {
            var invalidCount = await _repository.GetInconsistentCountAsync(
                report.TableName,
                rule.ColumnName,
                rule.ReferencedTable,
                rule.ReferencedColumn);
            
            if (invalidCount > 0)
            {
                report.Issues.Add(new QualityIssue
                {
                    IssueType = "Consistency",
                    Severity = "Critical",
                    Description = $"字段{rule.ColumnName}有{invalidCount}条记录引用不存在的{rule.ReferencedTable}.{rule.ReferencedColumn}",
                    ColumnName = rule.ColumnName,
                    ActualValue = invalidCount,
                    ExpectedValue = 0
                });
            }
        }
    }
    
    private double CalculateQualityScore(QualityReport report)
    {
        // 计算质量得分（0-100）
        var criticalIssues = report.Issues.Count(i => i.Severity == "Critical");
        var warningIssues = report.Issues.Count(i => i.Severity == "Warning");
        
        var score = 100 - criticalIssues * 20 - warningIssues * 5;
        return Math.Max(0, score);
    }
}
```
