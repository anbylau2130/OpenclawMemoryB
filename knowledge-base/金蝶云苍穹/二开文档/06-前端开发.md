**K3Cloud UI Components**（React）:
```jsx
import React from 'react';
import { 
  Button, 
  Input, 
  Form, 
  Table, 
  DatePicker, 
  Select,
  Card,
  Modal
} from 'k3cloud-ui';

const CustomBillForm = () => {
  const [form, setForm] = React.useState({
    billNo: '',
    billDate: null,
    status: 'draft',
    remarks: ''
  });

  const handleSubmit = async (e) => {
    e.preventDefault();
    
    // 表单验证
    if (!form.billNo) {
      alert('单据编号不能为空');
      return;
    }

    try {
      // API调用
      const response = await api.createBill(form);
      
      if (response.success) {
        alert('单据创建成功');
        setForm({
          billNo: '',
          billDate: null,
          status: 'draft',
          remarks: ''
        });
      } else {
        alert(`创建失败: ${response.message}`);
      }
    } catch (error) {
      console.error('创建失败:', error);
      alert('系统错误，请稍后重试');
    }
  };

  return (
    <Card title="自定义单据">
      <Form onSubmit={handleSubmit}>
        <Form.Item label="单据编号" required>
          <Input 
            value={form.billNo}
            onChange={(e) => setForm({...form, billNo: e.target.value})}
            placeholder="请输入单据编号"
            maxLength={50}
          />
        </Form.Item>

        <Form.Item label="单据日期" required>
          <DatePicker 
            value={form.billDate}
            onChange={(date) => setForm({...form, billDate: date})}
            placeholder="请选择日期"
            format="YYYY-MM-DD"
          />
        </Form.Item>

        <Form.Item label="状态">
          <Select 
            value={form.status}
            onChange={(value) => setForm({...form, status: value})}
            placeholder="请选择状态"
            options={[
              { label: '草稿', value: 'draft' },
              { label: '已提交', value: 'submitted' },
              { label: '已审批', value: 'approved' },
              { label: '已驳回', value: 'rejected' }
            ]}
          />
        </Form.Item>

        <Form.Item label="备注">
          <Input.TextArea 
            value={form.remarks}
            onChange={(e) => setForm({...form, remarks: e.target.value})}
            placeholder="请输入备注"
            rows={3}
            maxLength={500}
          />
        </Form.Item>

        <Form.Item>
          <Button type="primary" htmlType="submit">
            提交
          </Button>
          <Button type="default">
            重置
          </Button>
        </Form.Item>
      </Form>
    </Card>
  );
};

export default CustomBillForm;
```

## 🎯 核心功能

### 1. 单据列表

**React组件**:
```jsx
import React, { useEffect, useState } from 'react';
import { Table, Card, Button, Pagination, Input, Space } from 'k3cloud-ui';
import { fetchBills, deleteBill } from '@/services/bill';

const BillList = () => {
  const [loading, setLoading] = useState(false);
  const [bills, setBills] = useState([]);
  const [pagination, setPagination] = useState({
    current: 1,
    pageSize: 20,
    total: 0
  });
  const [filters, setFilters] = useState({
    billNo: '',
    status: '',
    dateRange: null
  });

  // 获取单据列表
  const fetchBillsData = async (page = 1) => {
    setLoading(true);
    try {
      const response = await fetchBills({
        page,
        pageSize: pagination.pageSize,
        ...filters
      });
      
      setBills(response.data.rows);
      setPagination({
        current: response.data.page,
        pageSize: pagination.pageSize,
        total: response.data.total
      });
    } catch (error) {
      console.error('获取失败:', error);
      alert('获取失败，请稍后重试');
    } finally {
      setLoading(false);
    }
  };

  // 删除单据
  const handleDelete = async (billId) => {
    if (!confirm('确认删除该单据？')) {
      return;
    }

    try {
      const response = await deleteBill(billId);
      
      if (response.success) {
        alert('删除成功');
        fetchBillsData(pagination.current);
      } else {
        alert(`删除失败: ${response.message}`);
      }
    } catch (error) {
      console.error('删除失败:', error);
      alert('系统错误，请稍后重试');
    }
  };

  // 初始化加载数据
  useEffect(() => {
    fetchBillsData();
  }, []);

  // 处理页码变化
  const handlePageChange = (page) => {
    fetchBillsData(page);
  };

  // 处理查询
  const handleSearch = () => {
    fetchBillsData(1);
  };

  // 表格列配置
  const columns = [
    {
      title: '单据编号',
      dataIndex: 'billNo',
      key: 'billNo',
      width: 150,
      sorter: true
    },
    {
      title: '单据日期',
      dataIndex: 'billDate',
      key: 'billDate',
      width: 120,
      sorter: true
    },
    {
      title: '状态',
      dataIndex: 'status',
      key: 'status',
      width: 100,
      render: (status) => {
        const statusMap = {
          draft: '草稿',
          submitted: '已提交',
          approved: '已审批',
          rejected: '已驳回'
        };
        return statusMap[status] || status;
      }
    },
    {
      title: '金额',
      dataIndex: 'amount',
      key: 'amount',
      width: 120,
      render: (amount) => `¥${amount.toLocaleString()}`
    },
    {
      title: '操作',
      key: 'action',
      width: 150,
      render: (text, record) => (
        <Space>
          <Button type="link" size="small">查看</Button>
          <Button type="link" size="small">编辑</Button>
          <Button type="link" danger size="small" onClick={() => handleDelete(record.billId)}>删除</Button>
        </Space>
      )
    }
  ];

  return (
    <Card 
      title="单据列表"
      extra={
        <Space>
          <Input.Search
            placeholder="搜索单据编号"
            allowClear
            enterButton
            onSearch={handleSearch}
            style={{ width: 200 }}
          />
          <Button type="primary" onClick={() => fetchBillsData()}>刷新</Button>
        </Space>
      }
    >
      <Table
        columns={columns}
        dataSource={bills}
        loading={loading}
        rowKey="billId"
        pagination={{
          current: pagination.current,
          pageSize: pagination.pageSize,
          total: pagination.total,
          onChange: handlePageChange,
          showSizeChanger: true
          showTotal: (total) => `共 ${total} 条`
        }}
        scroll={{ x: 1200 }}
      />
    </Card>
  );
};

export default BillList;
```

### 2. 单据详情

**React组件**:
```jsx
import React, { useState, useEffect } from 'react';
import { Card, Button, Descriptions, Tabs, Table, Space, Spin, Message } from 'k3cloud-ui';
import { fetchBillDetail, submitBill, approveBill } from '@/services/bill';

const BillDetail = ({ billId, onCancel }) => {
  const [loading, setLoading] = useState(false);
  const [bill, setBill] = useState(null);
  const [activeTab, setActiveTab] = useState('basic');

  useEffect(() => {
    fetchBillDetail();
  }, [billId]);

  const fetchBillDetail = async () => {
    setLoading(true);
    try {
      const response = await fetchBillDetail(billId);
      setBill(response.data);
    } catch (error) {
      console.error('获取详情失败:', error);
      Message.error('获取详情失败');
    } finally {
      setLoading(false);
    }
  };

  const handleSubmit = async () => {
    try {
      const response = await submitBill(billId);
      
      if (response.success) {
        Message.success('提交成功');
        fetchBillDetail();
      } else {
        Message.error(`提交失败: ${response.message}`);
      }
    } catch (error) {
      console.error('提交失败:', error);
      Message.error('系统错误');
    }
  };

  const handleApprove = async () => {
    try {
      const response = await approveBill(billId);
      
      if (response.success) {
        Message.success('审批成功');
        fetchBillDetail();
      } else {
        Message.error(`审批失败: ${response.message}`);
      }
    } catch (error) {
      console.error('审批失败:', error);
      Message.error('系统错误');
    }
  };

  if (loading || !bill) {
    return (
      <Card>
        <Spin spinning={loading}>
          <div style={{ textAlign: 'center', padding: '50px' }}>加载中...</div>
        </Spin>
      </Card>
    );
  }

  return (
    <Card 
      title="单据详情"
      extra={
        <Space>
          <Button onClick={() => onCancel()}>返回</Button>
          <Button type="primary" onClick={handleSubmit}>提交</Button>
          <Button type="primary" onClick={handleApprove}>审批</Button>
        </Space>
      }
    >
      <Tabs activeKey={activeTab} onChange={setActiveTab}>
        <Tabs.TabPane tabKey="basic" key="basic" title="基本信息">
          <Descriptions bordered column={2}>
            <Descriptions.Item label="单据编号">{bill.billNo}</Descriptions.Item>
            <Descriptions.Item label="单据日期">{bill.billDate}</Descriptions.Item>
            <Descriptions.Item label="创建时间">{bill.createTime}</Descriptions.Item>
            <Descriptions.Item label="创建人">{bill.creator}</Descriptions.Item>
            <Descriptions.Item label="状态">
              <span style={{ 
                color: bill.status === 'approved' ? '#52c41a' : 
                       bill.status === 'rejected' ? '#ff4d4f' : '#1890ff'
              }}>
                {bill.status === 'draft' && '草稿' ||
                 bill.status === 'submitted' && '已提交' ||
                 bill.status === 'approved' && '已审批' ||
                 bill.status === 'rejected' && '已驳回'}
              </span>
            </Descriptions.Item>
            <Descriptions.Item label="金额">
              <span style={{ fontWeight: 'bold' }}>
                ¥{bill.amount?.toLocaleString() || '0.00'}
              </span>
            </Descriptions.Item>
            <Descriptions.Item label="备注" span={2}>{bill.remarks || '-'}</Descriptions.Item>
          </Descriptions>
        </Tabs.TabPane>

        <Tabs.TabPane tabKey="entries" key="entries" title="明细信息">
          <Table
            columns={[
              { title: '行号', dataIndex: 'entryNo', width: 80 },
              { title: '产品ID', dataIndex: 'productId', width: 120 },
              { title: '数量', dataIndex: 'quantity', width: 100 },
              { title: '单价', dataIndex: 'price', width: 100 },
              { title: '金额', dataIndex: 'amount', width: 100 }
            ]}
            dataSource={bill.entries || []}
            rowKey="entryId"
            pagination={false}
            scroll={{ y: 300 }}
          />
        </Tabs.TabPane>

        <Tabs.TabPane tabKey="history" key="history" title="操作历史">
          <Timeline>
            {(bill.history || []).map((item, index) => (
              <Timeline.Item key={index} color={item.action === 'approve' ? 'green' : 'blue'}>
                <p>{item.action === 'approve' ? '审批' : item.action === 'submit' ? '提交' : item.action}</p>
                <p>{item.operator}</p>
                <p>{item.timestamp}</p>
                <p>{item.remark || ''}</p>
              </Timeline.Item>
            ))}
          </Timeline>
        </Tabs.TabPane>
      </Tabs>
    </Card>
  );
};

export default BillDetail;
```

### 3. 缓存优化

**React数据缓存**:
```jsx
import { useQuery, useMutation, QueryClient, QueryClientProvider } from '@tanstack/react-query';

const BillsPage = () => {
  const queryClient = new QueryClient();
  
  const { data, isLoading } = useQuery(
    ['bills', { page: 1, pageSize: 20 }],
    () => fetchBills({ page: 1, pageSize: 20 }),
    {
      staleTime: 5 * 60 * 1000,  // 5分钟内不重新请求
      cacheTime: 10 * 60 * 1000  // 10分钟内使用缓存数据
    }
  );

  const submitMutation = useMutation({
    mutationFn: submitBill,
    onSuccess: () => {
      queryClient.invalidateQueries(['bills']);  // 刷新数据
      Message.success('提交成功');
    },
    onError: (error) => {
      Message.error('提交失败');
    }
  });

  return (
    <Table
      loading={isLoading}
      dataSource={data?.rows || []}
      pagination={data?.pagination}
    />
  );
};
```
