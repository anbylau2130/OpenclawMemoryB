**用户服务（C#）**:
```csharp
using K3Cloud.System.User;

public class UserService : IUserService
{
    private readonly IUserRepository _repository;
    private readonly IPasswordHasher _passwordHasher;
    private readonly IRoleRepository _roleRepository;
    private readonly ILogger<UserService> _logger;
    
    public UserService(
        IUserRepository repository,
        IPasswordHasher passwordHasher,
        IRoleRepository roleRepository,
        ILogger<UserService> logger)
    {
        _repository = repository;
        _passwordHasher = passwordHasher;
        _roleRepository = roleRepository;
        _logger = logger;
    }
    
    /// <summary>
    /// 创建用户
    /// </summary>
    public async Task<User> CreateUserAsync(CreateUserRequest request)
    {
        // 检查用户名是否已存在
        if (await _repository.ExistsByUsernameAsync(request.Username))
        {
            throw new UserExistsException("用户名已存在");
        }
        
        // 检查邮箱是否已存在
        if (await _repository.ExistsByEmailAsync(request.Email))
        {
            throw new UserExistsException("邮箱已被注册");
        }
        
        // 密码哈希
        var (passwordHash, passwordSalt) = _passwordHasher.Hash(request.Password);
        
        // 创建用户
        var user = new User
        {
            UserId = Guid.NewGuid().ToString(),
            Username = request.Username,
            PasswordHash = passwordHash,
            PasswordSalt = passwordSalt,
            FullName = request.FullName,
            Email = request.Email,
            Phone = request.Phone,
            Gender = request.Gender,
            Status = "ACTIVE",
            CreateTime = DateTime.UtcNow,
            CreateBy = request.CreatorId
        };
        
        // 保存用户
        await _repository.SaveAsync(user);
        
        // 分配默认角色
        var defaultRoles = await _roleRepository.GetDefaultRolesAsync();
        foreach (var role in defaultRoles)
        {
            await _roleRepository.AssignRoleAsync(user.UserId, role.RoleId);
        }
        
        _logger.LogInformation(
            "用户已创建: {UserId}, 用户名: {Username}, 邮箱: {Email}",
            user.UserId,
            user.Username,
            user.Email
        );
        
        return user;
    }
    
    /// <summary>
    /// 更新用户信息
    /// </summary>
    public async Task<User> UpdateUserAsync(string userId, UpdateUserRequest request)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 更新基本信息
        user.FullName = request.FullName;
        user.Email = request.Email;
        user.Phone = request.Phone;
        user.Gender = request.Gender;
        user.Avatar = request.Avatar;
        user.UpdateTime = DateTime.UtcNow;
        user.UpdateBy = request.UpdaterId;
        
        // 更新用户资料
        if (request.Profile != null)
        {
            user.Profile = user.Profile ?? new UserProfile();
            user.Profile.BirthDate = request.Profile.BirthDate;
            user.Profile.Address = request.Profile.Address;
            user.Profile.City = request.Profile.City;
            user.Profile.Province = request.Profile.Province;
            user.Profile.Country = request.Profile.Country;
            user.Profile.PostCode = request.Profile.PostCode;
            user.Profile.EmergencyContact = request.Profile.EmergencyContact;
            user.Profile.EmergencyPhone = request.Profile.EmergencyPhone;
            user.Profile.Education = request.Profile.Education;
        }
        
        // 更新用户设置
        if (request.Settings != null)
        {
            user.Settings = user.Settings ?? new UserSettings();
            user.Settings.Language = request.Settings.Language;
            user.Settings.Timezone = request.Settings.Timezone;
            user.Settings.DateFormat = request.Settings.DateFormat;
            user.Settings.TimeFormat = request.Settings.TimeFormat;
            user.Settings.Theme = request.Settings.Theme;
            user.Settings.PageSize = request.Settings.PageSize;
            user.Settings.NotificationPreferences = request.Settings.NotificationPreferences;
        }
        
        // 保存用户
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户已更新: {UserId}, 用户名: {Username}",
            user.UserId,
            user.Username
        );
        
        return user;
    }
    
    /// <summary>
    /// 修改密码
    /// </summary>
    public async Task ChangePasswordAsync(string userId, ChangePasswordRequest request)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 验证旧密码
        var passwordValid = await _passwordHasher.VerifyAsync(
            request.OldPassword,
            user.PasswordHash,
            user.PasswordSalt
        );
        
        if (!passwordValid)
        {
            throw new InvalidPasswordException("旧密码错误");
        }
        
        // 检查新密码强度
        if (!IsPasswordStrong(request.NewPassword))
        {
            throw new WeakPasswordException("新密码强度不足");
        }
        
        // 检查新密码是否与旧密码相同
        if (request.NewPassword == request.OldPassword)
        {
            throw new InvalidPasswordException("新密码不能与旧密码相同");
        }
        
        // 更新密码
        var (passwordHash, passwordSalt) = _passwordHasher.Hash(request.NewPassword);
        user.PasswordHash = passwordHash;
        user.PasswordSalt = passwordSalt;
        user.UpdateTime = DateTime.UtcNow;
        
        // 保存用户
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户密码已修改: {UserId}, 用户名: {Username}",
            user.UserId,
            user.Username
        );
    }
    
    /// <summary>
    /// 重置密码（管理员操作）
    /// </summary>
    public async Task<string> ResetPasswordAsync(string userId, string operatorId)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 生成临时密码
        var tempPassword = GenerateTemporaryPassword();
        
        // 更新密码
        var (passwordHash, passwordSalt) = _passwordHasher.Hash(tempPassword);
        user.PasswordHash = passwordHash;
        user.PasswordSalt = passwordSalt;
        user.Status = "ACTIVE";  // 解锁账户
        user.UpdateTime = DateTime.UtcNow;
        user.UpdateBy = operatorId;
        
        // 保存用户
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户密码已重置: {UserId}, 用户名: {Username}, 操作人: {OperatorId}",
            user.UserId,
            user.Username,
            operatorId
        );
        
        return tempPassword;
    }
    
    /// <summary>
    /// 锁定用户
    /// </summary>
    public async Task LockUserAsync(string userId, string reason, string operatorId)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        user.Status = "LOCKED";
        user.LockReason = reason;
        user.UpdateTime = DateTime.UtcNow;
        user.UpdateBy = operatorId;
        
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户已锁定: {UserId}, 用户名: {Username}, 原因: {Reason}, 操作人: {OperatorId}",
            user.UserId,
            user.Username,
            reason,
            operatorId
        );
    }
    
    /// <summary>
    /// 解锁用户
    /// </summary>
    public async Task UnlockUserAsync(string userId, string operatorId)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        user.Status = "ACTIVE";
        user.LockReason = null;
        user.UpdateTime = DateTime.UtcNow;
        user.UpdateBy = operatorId;
        
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户已解锁: {UserId}, 用户名: {Username}, 操作人: {OperatorId}",
            user.UserId,
            user.Username,
            operatorId
        );
    }
    
    /// <summary>
    /// 删除用户（软删除）
    /// </summary>
    public async Task DeleteUserAsync(string userId, string operatorId)
    {
        var user = await _repository.GetByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 不能删除超级管理员
        if (user.Username == "admin")
        {
            throw new DeleteForbiddenException("不能删除超级管理员");
        }
        
        // 检查用户是否有未完成任务
        var hasPendingTasks = await _repository.HasPendingTasksAsync(userId);
        if (hasPendingTasks)
        {
            throw new DeleteForbiddenException("用户有待办任务，无法删除");
        }
        
        // 软删除用户
        user.Status = "INACTIVE";
        user.DeletedAt = DateTime.UtcNow;
        user.DeletedBy = operatorId;
        
        await _repository.UpdateAsync(user);
        
        _logger.LogInformation(
            "用户已删除: {UserId}, 用户名: {Username}, 操作人: {OperatorId}",
            user.UserId,
            user.Username,
            operatorId
        );
    }
    
    private bool IsPasswordStrong(string password)
    {
        // 密码强度规则
        // 至少8个字符
        // 包含大小写字母
        // 包含数字
        // 包含特殊字符
        return password.Length >= 8 &&
               password.Any(char.IsUpper) &&
               password.Any(char.IsLower) &&
               password.Any(char.IsDigit) &&
               password.Any(char.IsSymbol);
    }
    
    private string GenerateTemporaryPassword()
    {
        const string chars = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789!@#$%^&*";
        var random = new Random();
        return new string(Enumerable.Repeat(chars, 12)
            .Select(s => s[random.Next(s.Length)]).ToArray());
    }
}
```

**角色服务（C#）**:
```csharp
using K3Cloud.System.Role;

public class RoleService : IRoleService
{
    private readonly IRoleRepository _repository;
    private readonly IPermissionRepository _permissionRepository;
    private readonly ILogger<RoleService> _logger;
    
    public RoleService(
        IRoleRepository repository,
        IPermissionRepository permissionRepository,
        ILogger<RoleService> logger)
    {
        _repository = repository;
        _permissionRepository = permissionRepository;
        _logger = logger;
    }
    
    /// <summary>
    /// 创建角色
    /// </summary>
    public async Task<Role> CreateRoleAsync(CreateRoleRequest request)
    {
        // 检查角色编码是否已存在
        if (await _repository.ExistsByCodeAsync(request.RoleCode))
        {
            throw new RoleExistsException("角色编码已存在");
        }
        
        // 创建角色
        var role = new Role
        {
            RoleId = Guid.NewGuid().ToString(),
            RoleName = request.RoleName,
            RoleCode = request.RoleCode,
            Description = request.Description,
            RoleType = "CUSTOM",
            Status = "ACTIVE",
            CreateTime = DateTime.UtcNow,
            CreateBy = request.CreatorId
        };
        
        // 保存角色
        await _repository.SaveAsync(role);
        
        // 分配权限
        if (request.PermissionIds != null && request.PermissionIds.Any())
        {
            foreach (var permissionId in request.PermissionIds)
            {
                await AssignPermissionAsync(role.RoleId, permissionId, request.CreatorId);
            }
        }
        
        _logger.LogInformation(
            "角色已创建: {RoleId}, 角色名称: {RoleName}, 角色编码: {RoleCode}",
            role.RoleId,
            role.RoleName,
            role.RoleCode
        );
        
        return role;
    }
    
    /// <summary>
    /// 分配权限到角色
    /// </summary>
    public async Task AssignPermissionAsync(string roleId, string permissionId, string operatorId)
    {
        var role = await _repository.GetByIdAsync(roleId);
        if (role == null)
        {
            throw new RoleNotFoundException($"角色 {roleId} 不存在");
        }
        
        var permission = await _permissionRepository.GetByIdAsync(permissionId);
        if (permission == null)
        {
            throw new PermissionNotFoundException($"权限 {permissionId} 不存在");
        }
        
        // 检查是否已分配
        var exists = await _repository.PermissionAssignedAsync(roleId, permissionId);
        if (exists)
        {
            return;
        }
        
        // 分配权限
        await _repository.AssignPermissionAsync(roleId, permissionId, operatorId);
        
        _logger.LogInformation(
            "权限已分配到角色: {RoleId}, 权限: {PermissionId}, 操作人: {OperatorId}",
            roleId,
            permissionId,
            operatorId
        );
    }
    
    /// <summary>
    /// 撤销角色权限
    /// </summary>
    public async Task RevokePermissionAsync(string roleId, string permissionId, string operatorId)
    {
        var role = await _repository.GetByIdAsync(roleId);
        if (role == null)
        {
            throw new RoleNotFoundException($"角色 {roleId} 不存在");
        }
        
        // 撤销权限
        await _repository.RevokePermissionAsync(roleId, permissionId);
        
        _logger.LogInformation(
            "权限已从角色撤销: {RoleId}, 权限: {PermissionId}, 操作人: {OperatorId}",
            roleId,
            permissionId,
            operatorId
        );
    }
    
    /// <summary>
    /// 分配角色到用户
    /// </summary>
    public async Task AssignRoleAsync(string userId, string roleId, string operatorId)
    {
        var user = await _repository.GetUserByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        var role = await _repository.GetByIdAsync(roleId);
        if (role == null)
        {
            throw new RoleNotFoundException($"角色 {roleId} 不存在");
        }
        
        // 检查是否已分配
        var exists = await _repository.RoleAssignedAsync(userId, roleId);
        if (exists)
        {
            return;
        }
        
        // 分配角色
        await _repository.AssignRoleAsync(userId, roleId, operatorId);
        
        _logger.LogInformation(
            "角色已分配到用户: {UserId}, 角色: {RoleId}, 操作人: {OperatorId}",
            userId,
            roleId,
            operatorId
        );
    }
    
    /// <summary>
    /// 撤销用户角色
    /// </summary>
    public async Task RevokeRoleAsync(string userId, string roleId, string operatorId)
    {
        var user = await _repository.GetUserByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 不能撤销超级管理员角色
        if (roleId == "super_admin")
        {
            throw new OperationForbiddenException("不能撤销超级管理员角色");
        }
        
        // 撤销角色
        await _repository.RevokeRoleAsync(userId, roleId);
        
        _logger.LogInformation(
            "角色已从用户撤销: {UserId}, 角色: {RoleId}, 操作人: {OperatorId}",
            userId,
            roleId,
            operatorId
        );
    }
    
    /// <summary>
    /// 获取用户权限
    /// </summary>
    public async Task<List<Permission>> GetUserPermissionsAsync(string userId)
    {
        var userRoles = await _repository.GetUserRolesAsync(userId);
        var permissions = new List<Permission>();
        
        foreach (var roleId in userRoles)
        {
            var rolePermissions = await _repository.GetRolePermissionsAsync(roleId);
            permissions.AddRange(rolePermissions);
        }
        
        // 去重
        return permissions.DistinctBy(p => p.PermissionId).ToList();
    }
    
    /// <summary>
    /// 检查用户是否有权限
    /// </summary>
    public async Task<bool> HasPermissionAsync(string userId, string resource, string action)
    {
        var permissions = await GetUserPermissionsAsync(userId);
        
        // 检查通配符权限
        var hasWildcard = permissions.Any(p => 
            p.Resource == "*" || 
            (p.Resource == resource && p.Action == "*"));
        
        if (hasWildcard)
        {
            return true;
        }
        
        // 检查具体权限
        return permissions.Any(p => p.Resource == resource && p.Action == action);
    }
}
```

```csharp
using K3Cloud.System.Organization;

public class OrganizationService : IOrganizationService
{
    private readonly IOrganizationRepository _repository;
    private readonly ILogger<OrganizationService> _logger;
    
    public OrganizationService(
        IOrganizationRepository repository,
        ILogger<OrganizationService> logger)
    {
        _repository = repository;
        _logger = logger;
    }
    
    /// <summary>
    /// 创建组织
    /// </summary>
    public async Task<Organization> CreateOrganizationAsync(CreateOrganizationRequest request)
    {
        // 检查组织编码是否已存在
        if (await _repository.ExistsByCodeAsync(request.OrgCode))
        {
            throw new OrganizationExistsException("组织编码已存在");
        }
        
        // 获取父组织
        Organization parentOrg = null;
        if (!string.IsNullOrEmpty(request.ParentId))
        {
            parentOrg = await _repository.GetByIdAsync(request.ParentId);
            if (parentOrg == null)
            {
                throw new OrganizationNotFoundException($"父组织 {request.ParentId} 不存在");
            }
        }
        
        // 创建组织
        var org = new Organization
        {
            OrgId = Guid.NewGuid().ToString(),
            OrgName = request.OrgName,
            OrgCode = request.OrgCode,
            OrgType = request.OrgType,
            ParentId = request.ParentId,
            Path = parentOrg != null ? $"{parentOrg.Path}/{request.OrgCode}" : $"/{request.OrgCode}",
            Level = parentOrg != null ? parentOrg.Level + 1 : 1,
            Sort = request.Sort,
            Status = "ACTIVE",
            CreateTime = DateTime.UtcNow,
            CreateBy = request.CreatorId
        };
        
        // 保存组织
        await _repository.SaveAsync(org);
        
        _logger.LogInformation(
            "组织已创建: {OrgId}, 组织名称: {OrgName}, 组织编码: {OrgCode}, 父组织: {ParentId}",
            org.OrgId,
            org.OrgName,
            org.OrgCode,
            org.ParentId
        );
        
        return org;
    }
    
    /// <summary>
    /// 更新组织
    /// </summary>
    public async Task<Organization> UpdateOrganizationAsync(string orgId, UpdateOrganizationRequest request)
    {
        var org = await _repository.GetByIdAsync(orgId);
        if (org == null)
        {
            throw new OrganizationNotFoundException($"组织 {orgId} 不存在");
        }
        
        // 不能修改根组织的父组织
        if (org.OrgId == "root" && !string.IsNullOrEmpty(request.ParentId))
        {
            throw new OperationForbiddenException("不能修改根组织的父组织");
        }
        
        // 更新组织信息
        org.OrgName = request.OrgName;
        org.OrgType = request.OrgType;
        org.LeaderId = request.LeaderId;
        org.LeaderName = request.LeaderName;
        org.Sort = request.Sort;
        org.UpdateTime = DateTime.UtcNow;
        org.UpdateBy = request.UpdaterId;
        
        // 保存组织
        await _repository.UpdateAsync(org);
        
        _logger.LogInformation(
            "组织已更新: {OrgId}, 组织名称: {OrgName}",
            org.OrgId,
            org.OrgName
        );
        
        return org;
    }
    
    /// <summary>
    /// 移动组织（修改父组织）
    /// </summary>
    public async Task MoveOrganizationAsync(string orgId, string newParentId, string operatorId)
    {
        var org = await _repository.GetByIdAsync(orgId);
        if (org == null)
        {
            throw new OrganizationNotFoundException($"组织 {orgId} 不存在");
        }
        
        // 不能移动根组织
        if (org.OrgId == "root")
        {
            throw new OperationForbiddenException("不能移动根组织");
        }
        
        // 不能将组织移动到其子组织下
        if (await _repository.IsDescendantAsync(newParentId, orgId))
        {
            throw new OperationForbiddenException("不能将组织移动到其子组织下");
        }
        
        // 获取新父组织
        var newParentOrg = await _repository.GetByIdAsync(newParentId);
        if (newParentOrg == null)
        {
            throw new OrganizationNotFoundException($"新父组织 {newParentId} 不存在");
        }
        
        // 更新父组织和路径
        var oldPath = org.Path;
        org.ParentId = newParentId;
        org.Path = $"{newParentOrg.Path}/{org.OrgCode}";
        org.Level = newParentOrg.Level + 1;
        org.UpdateTime = DateTime.UtcNow;
        org.UpdateBy = operatorId;
        
        // 保存组织
        await _repository.UpdateAsync(org);
        
        // 更新所有子组织的路径
        await UpdateDescendantPathsAsync(orgId, oldPath, org.Path);
        
        _logger.LogInformation(
            "组织已移动: {OrgId}, 从 {OldPath} 到 {NewPath}, 操作人: {OperatorId}",
            orgId,
            oldPath,
            org.Path,
            operatorId
        );
    }
    
    /// <summary>
    /// 添加用户到组织
    /// </summary>
    public async Task AddUserToOrgAsync(string orgId, string userId, string position, string operatorId)
    {
        var org = await _repository.GetByIdAsync(orgId);
        if (org == null)
        {
            throw new OrganizationNotFoundException($"组织 {orgId} 不存在");
        }
        
        var user = await _repository.GetUserByIdAsync(userId);
        if (user == null)
        {
            throw new UserNotFoundException($"用户 {userId} 不存在");
        }
        
        // 检查是否已在组织中
        var exists = await _repository.UserInOrgAsync(orgId, userId);
        if (exists)
        {
            return;
        }
        
        // 添加用户到组织
        await _repository.AddUserToOrgAsync(orgId, userId, position, operatorId);
        
        _logger.LogInformation(
            "用户已添加到组织: {OrgId}, 用户: {UserId}, 职位: {Position}, 操作人: {OperatorId}",
            orgId,
            userId,
            position,
            operatorId
        );
    }
    
    /// <summary>
    /// 从组织移除用户
    /// </summary>
    public async Task RemoveUserFromOrgAsync(string orgId, string userId, string operatorId)
    {
        var org = await _repository.GetByIdAsync(orgId);
        if (org == null)
        {
            throw new OrganizationNotFoundException($"组织 {orgId} 不存在");
        }
        
        // 移除用户
        await _repository.RemoveUserFromOrgAsync(orgId, userId);
        
        _logger.LogInformation(
            "用户已从组织移除: {OrgId}, 用户: {UserId}, 操作人: {OperatorId}",
            orgId,
            userId,
            operatorId
        );
    }
    
    /// <summary>
    /// 获取组织树
    /// </summary>
    public async Task<List<OrganizationTree>> GetOrganizationTreeAsync(string rootId = null)
    {
        if (string.IsNullOrEmpty(rootId))
        {
            rootId = "root";
        }
        
        var org = await _repository.GetByIdAsync(rootId);
        if (org == null)
        {
            return new List<OrganizationTree>();
        }
        
        var tree = new OrganizationTree
        {
            OrgId = org.OrgId,
            OrgName = org.OrgName,
            OrgCode = org.OrgCode,
            OrgType = org.OrgType,
            ParentId = org.ParentId,
            Level = org.Level,
            Sort = org.Sort,
            Children = await GetChildrenAsync(org.OrgId)
        };
        
        return new List<OrganizationTree> { tree };
    }
    
    private async Task<List<OrganizationTree>> GetChildrenAsync(string parentId)
    {
        var children = await _repository.GetChildrenAsync(parentId);
        var result = new List<OrganizationTree>();
        
        foreach (var child in children)
        {
            var tree = new OrganizationTree
            {
                OrgId = child.OrgId,
                OrgName = child.OrgName,
                OrgCode = child.OrgCode,
                OrgType = child.OrgType,
                ParentId = child.ParentId,
                Level = child.Level,
                Sort = child.Sort,
                Children = await GetChildrenAsync(child.OrgId)
            };
            
            result.Add(tree);
        }
        
        return result;
    }
    
    private async Task UpdateDescendantPathsAsync(string orgId, string oldPathPrefix, string newPathPrefix)
    {
        var descendants = await _repository.GetDescendantsAsync(orgId);
        
        foreach (var descendant in descendants)
        {
            descendant.Path = descendant.Path.Replace(oldPathPrefix, newPathPrefix);
            descendant.Level = newPathPrefix.Count(c => c == '/') + 1;
            await _repository.UpdateAsync(descendant);
        }
    }
}
```

## 📝 最佳实践

### 1. 用户管理
- 最小权限原则：只授予必要的权限
- 定期审查：定期检查用户权限，及时清理
- 密码策略：强密码要求，定期更换
- 账号安全：启用MFA，监控异常登录

### 2. 角色权限
- 角色分离：职责分离，避免权限集中
- 权限细粒度：资源+动作的细粒度权限
- 权限审计：记录权限变更，定期审计
- 动态权限：根据业务需求动态调整

### 3. 组织架构
- 层级合理：组织层级不宜过深
- 职责明确：每个组织职责清晰
- 人员管理：及时更新组织人员信息
- 数据权限：基于组织的数据权限控制
