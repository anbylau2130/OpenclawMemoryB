## 📋 概述

```csharp
using K3Cloud.Reporting.DataSource;

public class ReportDataSource
{
    public string DataSourceId { get; set; }
    public DataSourceType Type { get; set; }  // SQL, API, Inline
    public string ConnectionId { get; set; }
    public ReportQuery Query { get; set; }
}

public class ReportQuery
{
    public string MainQuery { get; set; }
    public string SummaryQuery { get; set; }
    public string ChartQuery { get; set; }
    public Dictionary<string, string> Parameters { get; set; }
}

public class SqlDataSource : IDataSource
{
    private readonly IDbConnectionFactory _connectionFactory;
    
    public SqlDataSource(IDbConnectionFactory connectionFactory)
    {
        _connectionFactory = connectionFactory;
    }
    
    public async Task<DataTable> QueryAsync(string query, 
        Dictionary<string, object> parameters)
    {
        using var connection = await _connectionFactory.CreateConnectionAsync();
        using var command = connection.CreateCommand();
        command.CommandText = query;
        
        foreach (var param in parameters)
        {
            var parameter = command.CreateParameter();
            parameter.ParameterName = param.Key;
            parameter.Value = param.Value;
            command.Parameters.Add(parameter);
        }
        
        using var reader = await command.ExecuteReaderAsync();
        var table = new DataTable();
        table.Load(reader);
        
        return table;
    }
}
```

```csharp
using K3Cloud.Reporting.Designer;

public class ReportDesigner
{
    private readonly IComponentFactory _componentFactory;
    
    public ReportDesigner(IComponentFactory componentFactory)
    {
        _componentFactory = componentFactory;
    }
    
    public Report CreateReport(ReportDesign design)
    {
        var report = new Report(design.ReportId);
        
        // 创建画布
        var canvas = new ReportCanvas(design.Canvas);
        report.Canvas = canvas;
        
        // 创建各个区域
        foreach (var bandConfig in design.Bands)
        {
            var band = CreateBand(bandConfig);
            report.Bands.Add(band);
        }
        
        // 创建组件
        foreach (var componentConfig in design.Components)
        {
            var component = _componentFactory.Create(componentConfig);
            report.AddComponent(component);
        }
        
        return report;
    }
    
    private IBand CreateBand(BandConfig config)
    {
        switch (config.RepeatOn)
        {
            case "eachPage":
                return new PageHeaderBand(config);
            case "eachRow":
                return new DataBand(config);
            case "eachPage":
                return new PageFooterBand(config);
            default:
                return new StaticBand(config);
        }
    }
}
```

```csharp
public class GroupReport : ReportBase
{
    private readonly IDataSource _dataSource;
    
    public GroupReport(string reportId, IDataSource dataSource)
        : base(reportId)
    {
        _dataSource = dataSource;
    }
    
    public override async Task<byte[]> GenerateAsync(ReportParameters parameters)
    {
        var data = await _dataSource.QueryAsync(parameters.Query);
        
        var renderer = new GroupRenderer();
        renderer.GroupBy = parameters.GroupBy;
        renderer.Columns = parameters.Columns;
        
        return await renderer.RenderAsync(data);
    }
}

public class GroupRenderer : IReportRenderer
{
    public List<string> GroupBy { get; set; }
    public List<ColumnDefinition> Columns { get; set; }
    
    public async Task<byte[]> RenderAsync(DataTable data)
    {
        var document = new Document();
        
        var groupedData = data.GroupBy(row => 
            string.Join("|", GroupBy.Select(g => row[g]))
        );
        
        foreach (var group in groupedData)
        {
            // 渲染分组头
            var groupHeader = new Paragraph(group.Key);
            document.Add(groupHeader);
            
            // 渲染分组数据
            var table = CreateTable(Columns, group.ToList());
            document.Add(table);
            
            // 渲染分组汇总
            var summary = CreateSummary(group.ToList());
            document.Add(summary);
        }
        
        return await document.ExportAsync();
    }
}
```

```csharp
using K3Cloud.Reporting.Charts;

public class ChartReport : ReportBase
{
    private readonly IChartRenderer _chartRenderer;
    
    public ChartReport(string reportId, IChartRenderer chartRenderer)
        : base(reportId)
    {
        _chartRenderer = chartRenderer;
    }
    
    public override async Task<byte[]> GenerateAsync(ReportParameters parameters)
    {
        var charts = new List<Chart>();
        
        foreach (var chartConfig in parameters.Charts)
        {
            var data = await _dataSource.QueryAsync(chartConfig.DataSource.Query);
            
            var chart = CreateChart(chartConfig, data);
            charts.Add(chart);
        }
        
        return await _chartRenderer.RenderAsync(charts);
    }
    
    private IChart CreateChart(ChartConfig config, DataTable data)
    {
        switch (config.ChartType)
        {
            case "line":
                return new LineChart(config, data);
            case "bar":
                return new BarChart(config, data);
            case "pie":
                return new PieChart(config, data);
            default:
                throw new NotSupportedException($"不支持的图表类型: {config.ChartType}");
        }
    }
}
```

## 🔄 报表导出

### 1. Excel导出

```csharp
using K3Cloud.Reporting.Export;

public class ExcelExporter : IReportExporter
{
    public async Task<byte[]> ExportAsync(ReportData data)
    {
        using var package = new ExcelPackage();
        var worksheet = package.Workbook.Worksheets.Add("Data");
        
        // 写入表头
        for (int i = 0; i < data.Columns.Count; i++)
        {
            worksheet.Cells[1, i + 1].Value = data.Columns[i].Title;
            worksheet.Cells[1, i + 1].Style.Font.Bold = true;
        }
        
        // 写入数据
        for (int row = 0; row < data.Rows.Count; row++)
        {
            for (int col = 0; col < data.Columns.Count; col++)
            {
                var cellValue = data.Rows[row][col];
                worksheet.Cells[row + 2, col + 1].Value = cellValue;
                
                // 应用格式
                if (data.Columns[col].Format != null)
                {
                    worksheet.Cells[row + 2, col + 1].Style.Numberformat.Format = 
                        data.Columns[col].Format;
                }
            }
        }
        
        return await package.GetAsByteArrayAsync();
    }
}
```

### 2. PDF导出

```csharp
public class PdfExporter : IReportExporter
{
    public async Task<byte[]> ExportAsync(ReportData data)
    {
        using var document = new Document();
        
        var table = document.AddTable();
        table.AddColumn(data.Columns.Count);
        
        // 添加表头
        for (int i = 0; i < data.Columns.Count; i++)
        {
            var cell = table.AddCell();
            cell.AddParagraph(data.Columns[i].Title);
            cell.Style.Font.Bold = true;
            cell.Style.BackgroundColor = Colors.LightGray;
        }
        
        // 添加数据
        foreach (var row in data.Rows)
        {
            var row = table.AddRow();
            for (int i = 0; i < data.Columns.Count; i++)
            {
                var cell = row.Cells[i];
                cell.AddParagraph(row[i].ToString());
            }
        }
        
        return await document.ExportToPdfAsync();
    }
}
```

## 📊 性能优化

### 1. 数据分页

```csharp
public class PagedQuery
{
    public async Task<QueryResult> QueryPagedAsync(
        string query,
        int page,
        int pageSize)
    {
        var offset = (page - 1) * pageSize;
        var pagedQuery = $"{query} LIMIT {offset}, {pageSize}";
        
        var data = await ExecuteQueryAsync(pagedQuery);
        var total = await GetTotalCountAsync(query);
        
        return new QueryResult
        {
            Data = data,
            Total = total,
            Page = page,
            PageSize = pageSize,
            TotalPages = (int)Math.Ceiling((double)total / pageSize)
        };
    }
}
```

### 2. 报表缓存

```csharp
public class ReportCache
{
    private readonly IMemoryCache _cache;
    private readonly TimeSpan _cacheTtl;
    
    public ReportCache(IMemoryCache cache)
    {
        _cache = cache;
        _cacheTtl = TimeSpan.FromMinutes(30);
    }
    
    public async Task<byte[]> GetOrGenerateAsync(
        string reportId,
        ReportParameters parameters,
        Func<Task<byte[]>> generator)
    {
        var cacheKey = GenerateCacheKey(reportId, parameters);
        
        return await _cache.GetOrLoadAsync(cacheKey, async () =>
        {
            return await generator();
        }, _cacheTtl);
    }
    
    private string GenerateCacheKey(string reportId, ReportParameters parameters)
    {
        var keyBuilder = new StringBuilder();
        keyBuilder.Append(reportId);
        
        foreach (var param in parameters.GetAll())
        {
            keyBuilder.Append($"|{param.Key}={param.Value}");
        }
        
        return keyBuilder.ToString();
    }
}
```
