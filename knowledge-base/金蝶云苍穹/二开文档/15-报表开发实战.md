**Java实现多数据源查询**:
```java
@Service
public class MultiDataSourceReportService {
    
    @Autowired
    @Qualifier("salesDataSource")
    private DataSource salesDataSource;
    
    @Autowired
    @Qualifier("inventoryDataSource")
    private DataSource inventoryDataSource;
    
    @Autowired
    @Qualifier("financeDataSource")
    private DataSource financeDataSource;
    
    /**
     * 查询销售数据
     */
    public List<SalesData> querySalesData(Date startDate, Date endDate) {
        String sql = "SELECT bill_no, bill_date, customer_name, material_name, " +
                     "qty, amount, gross_profit " +
                     "FROM t_sales_bill s " +
                     "INNER JOIN t_customer c ON s.customer_id = c.customer_id " +
                     "INNER JOIN t_material m ON s.material_id = m.material_id " +
                     "WHERE s.bill_date BETWEEN ? AND ?";
        
        try (Connection conn = salesDataSource.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            
            stmt.setDate(1, new java.sql.Date(startDate.getTime()));
            stmt.setDate(2, new java.sql.Date(endDate.getTime()));
            
            List<SalesData> result = new ArrayList<>();
            try (ResultSet rs = stmt.executeQuery()) {
                while (rs.next()) {
                    SalesData data = new SalesData();
                    data.setBillNo(rs.getString("bill_no"));
                    data.setBillDate(rs.getDate("bill_date"));
                    data.setCustomerName(rs.getString("customer_name"));
                    data.setMaterialName(rs.getString("material_name"));
                    data.setQty(rs.getBigDecimal("qty"));
                    data.setAmount(rs.getBigDecimal("amount"));
                    data.setGrossProfit(rs.getBigDecimal("gross_profit"));
                    result.add(data);
                }
            }
            
            return result;
        } catch (SQLException e) {
            throw new RuntimeException("查询销售数据失败", e);
        }
    }
    
    /**
     * 查询库存数据
     */
    public List<InventoryData> queryInventoryData(List<String> materialCodes) {
        String sql = "SELECT material_code, material_name, stock_qty, " +
                     "available_qty, warehouse_name " +
                     "FROM t_inventory i " +
                     "INNER JOIN t_material m ON i.material_id = m.material_id " +
                     "INNER JOIN t_warehouse w ON i.warehouse_id = w.warehouse_id " +
                     "WHERE m.material_code IN (%s)";
        
        String placeholders = materialCodes.stream()
            .map(code -> "?")
            .collect(Collectors.joining(","));
        
        sql = String.format(sql, placeholders);
        
        try (Connection conn = inventoryDataSource.getConnection();
             PreparedStatement stmt = conn.prepareStatement(sql)) {
            
            for (int i = 0; i < materialCodes.size(); i++) {
                stmt.setString(i + 1, materialCodes.get(i));
            }
            
            List<InventoryData> result = new ArrayList<>();
            try (ResultSet rs = stmt.executeQuery()) {
                while (rs.next()) {
                    InventoryData data = new InventoryData();
                    data.setMaterialCode(rs.getString("material_code"));
                    data.setMaterialName(rs.getString("material_name"));
                    data.setStockQty(rs.getBigDecimal("stock_qty"));
                    data.setAvailableQty(rs.getBigDecimal("available_qty"));
                    data.setWarehouseName(rs.getString("warehouse_name"));
                    result.add(data);
                }
            }
            
            return result;
        } catch (SQLException e) {
            throw new RuntimeException("查询库存数据失败", e);
        }
    }
    
    /**
     * 合并数据
     */
    public List<CombinedReportData> mergeData(List<SalesData> salesData,
                                               List<InventoryData> inventoryData) {
        // 创建物料编码到库存的映射
        Map<String, InventoryData> inventoryMap = inventoryData.stream()
            .collect(Collectors.toMap(
                InventoryData::getMaterialCode,
                data -> data,
                (existing, replacement) -> existing
            ));
        
        // 合并数据
        return salesData.stream()
            .map(sales -> {
                CombinedReportData combined = new CombinedReportData();
                combined.setBillNo(sales.getBillNo());
                combined.setBillDate(sales.getBillDate());
                combined.setCustomerName(sales.getCustomerName());
                combined.setMaterialName(sales.getMaterialName());
                combined.setQty(sales.getQty());
                combined.setAmount(sales.getAmount());
                combined.setGrossProfit(sales.getGrossProfit());
                
                // 关联库存数据
                InventoryData inventory = inventoryMap.get(sales.getMaterialCode());
                if (inventory != null) {
                    combined.setStockQty(inventory.getStockQty());
                    combined.setAvailableQty(inventory.getAvailableQty());
                    combined.setWarehouseName(inventory.getWarehouseName());
                }
                
                return combined;
            })
            .collect(Collectors.toList());
    }
}
```

### 2. 交叉报表

**场景**：销售交叉分析报表（按客户×物料×月份）

**交叉报表实现**:
```java
@Service
public class CrossTableReportService {
    
    /**
     * 生成交叉报表
     */
    public CrossTableReport generateCrossReport(Date startDate, Date endDate) {
        // 1. 查询原始数据
        List<SalesDetail> details = querySalesDetails(startDate, endDate);
        
        // 2. 提取维度
        Set<String> customers = details.stream()
            .map(SalesDetail::getCustomerName)
            .collect(Collectors.toCollection(TreeSet::new));
        
        Set<String> materials = details.stream()
            .map(SalesDetail::getMaterialName)
            .collect(Collectors.toCollection(TreeSet::new));
        
        Set<String> months = details.stream()
            .map(d -> formatDate(d.getBillDate(), "yyyy-MM"))
            .collect(Collectors.toCollection(TreeSet::new));
        
        // 3. 构建交叉数据
        Map<String, BigDecimal> crossData = new HashMap<>();
        
        details.forEach(detail -> {
            String key = String.format("%s|%s|%s",
                detail.getCustomerName(),
                detail.getMaterialName(),
                formatDate(detail.getBillDate(), "yyyy-MM"));
            
            crossData.merge(key, detail.getAmount(), BigDecimal::add);
        });
        
        // 4. 生成报表
        CrossTableReport report = new CrossTableReport();
        report.setRowHeaders(new ArrayList<>(customers));
        report.setColumnHeaders(new ArrayList<>(materials));
        report.setData(crossData);
        
        return report;
    }
    
    /**
     * 导出Excel
     */
    public void exportToExcel(CrossTableReport report, OutputStream outputStream) 
            throws IOException {
        try (Workbook workbook = new XSSFWorkbook()) {
            Sheet sheet = workbook.createSheet("交叉报表");
            
            // 创建样式
            CellStyle headerStyle = createHeaderStyle(workbook);
            CellStyle dataStyle = createDataStyle(workbook);
            
            // 写入表头
            Row headerRow = sheet.createRow(0);
            headerRow.createCell(0).setCellValue("客户\\物料");
            headerRow.getCell(0).setCellStyle(headerStyle);
            
            int colIndex = 1;
            for (String material : report.getColumnHeaders()) {
                Cell cell = headerRow.createCell(colIndex++);
                cell.setCellValue(material);
                cell.setCellStyle(headerStyle);
            }
            
            // 写入数据
            int rowIndex = 1;
            for (String customer : report.getRowHeaders()) {
                Row dataRow = sheet.createRow(rowIndex++);
                
                // 客户列
                dataRow.createCell(0).setCellValue(customer);
                dataRow.getCell(0).setCellStyle(dataStyle);
                
                // 数据列
                colIndex = 1;
                for (String material : report.getColumnHeaders()) {
                    String key = customer + "|" + material;
                    BigDecimal amount = report.getData().get(key);
                    
                    Cell cell = dataRow.createCell(colIndex++);
                    if (amount != null) {
                        cell.setCellValue(amount.doubleValue());
                    } else {
                        cell.setCellValue(0);
                    }
                    cell.setCellStyle(dataStyle);
                }
            }
            
            // 自动调整列宽
            for (int i = 0; i <= report.getColumnHeaders().size(); i++) {
                sheet.autoSizeColumn(i);
            }
            
            workbook.write(outputStream);
        }
    }
    
    private CellStyle createHeaderStyle(Workbook workbook) {
        CellStyle style = workbook.createCellStyle();
        style.setFillForegroundColor(IndexedColors.GREY_25_PERCENT.getIndex());
        style.setFillPattern(FillPatternType.SOLID_FOREGROUND);
        style.setBorderTop(BorderStyle.THIN);
        style.setBorderBottom(BorderStyle.THIN);
        style.setBorderLeft(BorderStyle.THIN);
        style.setBorderRight(BorderStyle.THIN);
        style.setAlignment(HorizontalAlignment.CENTER);
        
        Font font = workbook.createFont();
        font.setBold(true);
        style.setFont(font);
        
        return style;
    }
    
    private CellStyle createDataStyle(Workbook workbook) {
        CellStyle style = workbook.createCellStyle();
        style.setBorderTop(BorderStyle.THIN);
        style.setBorderBottom(BorderStyle.THIN);
        style.setBorderLeft(BorderStyle.THIN);
        style.setBorderRight(BorderStyle.THIN);
        style.setAlignment(HorizontalAlignment.RIGHT);
        
        DataFormat format = workbook.createDataFormat();
        style.setDataFormat(format.getFormat("#,##0.00"));
        
        return style;
    }
}
```

```java
@Service
public class PivotTableService {
    
    /**
     * 生成数据透视表
     */
    public PivotTable generatePivotTable(PivotRequest request) {
        // 1. 查询原始数据
        List<Map<String, Object>> rawData = queryRawData(request);
        
        // 2. 构建透视表
        PivotTable pivotTable = new PivotTable();
        
        // 行维度
        List<String> rowDimensions = request.getRowDimensions();
        // 列维度
        List<String> columnDimensions = request.getColumnDimensions();
        // 值字段
        String valueField = request.getValueField();
        // 聚合方式
        AggregationType aggregation = request.getAggregation();
        
        // 3. 分组聚合
        Map<String, Map<String, List<BigDecimal>>> groupedData = new LinkedHashMap<>();
        Set<String> columnKeys = new TreeSet<>();
        
        for (Map<String, Object> row : rawData) {
            // 构建行键
            String rowKey = buildKey(row, rowDimensions);
            // 构建列键
            String colKey = buildKey(row, columnDimensions);
            
            columnKeys.add(colKey);
            
            groupedData.computeIfAbsent(rowKey, k -> new LinkedHashMap<>())
                .computeIfAbsent(colKey, k -> new ArrayList<>())
                .add(new BigDecimal(row.get(valueField).toString()));
        }
        
        // 4. 计算聚合值
        Map<String, Map<String, BigDecimal>> pivotData = new LinkedHashMap<>();
        
        for (Map.Entry<String, Map<String, List<BigDecimal>>> rowEntry : 
             groupedData.entrySet()) {
            String rowKey = rowEntry.getKey();
            Map<String, BigDecimal> rowData = new LinkedHashMap<>();
            
            for (String colKey : columnKeys) {
                List<BigDecimal> values = rowEntry.getValue().get(colKey);
                BigDecimal aggregatedValue = aggregate(values, aggregation);
                rowData.put(colKey, aggregatedValue);
            }
            
            pivotData.put(rowKey, rowData);
        }
        
        // 5. 设置结果
        pivotTable.setRowHeaders(new ArrayList<>(groupedData.keySet()));
        pivotTable.setColumnHeaders(new ArrayList<>(columnKeys));
        pivotTable.setData(pivotData);
        
        // 6. 计算行列合计
        calculateTotals(pivotTable, aggregation);
        
        return pivotTable;
    }
    
    /**
     * 构建维度键
     */
    private String buildKey(Map<String, Object> row, List<String> dimensions) {
        return dimensions.stream()
            .map(dim -> row.get(dim) != null ? row.get(dim).toString() : "")
            .collect(Collectors.joining("|"));
    }
    
    /**
     * 聚合计算
     */
    private BigDecimal aggregate(List<BigDecimal> values, AggregationType type) {
        if (values == null || values.isEmpty()) {
            return BigDecimal.ZERO;
        }
        
        switch (type) {
            case SUM:
                return values.stream()
                    .reduce(BigDecimal.ZERO, BigDecimal::add);
            
            case AVG:
                return values.stream()
                    .reduce(BigDecimal.ZERO, BigDecimal::add)
                    .divide(new BigDecimal(values.size()), 2, RoundingMode.HALF_UP);
            
            case COUNT:
                return new BigDecimal(values.size());
            
            case MAX:
                return values.stream()
                    .max(BigDecimal::compareTo)
                    .orElse(BigDecimal.ZERO);
            
            case MIN:
                return values.stream()
                    .min(BigDecimal::compareTo)
                    .orElse(BigDecimal.ZERO);
            
            default:
                return BigDecimal.ZERO;
        }
    }
    
    /**
     * 计算合计
     */
    private void calculateTotals(PivotTable table, AggregationType type) {
        // 行合计
        Map<String, BigDecimal> rowTotals = new LinkedHashMap<>();
        
        for (Map.Entry<String, Map<String, BigDecimal>> rowEntry : 
             table.getData().entrySet()) {
            BigDecimal total = rowEntry.getValue().values().stream()
                .reduce(BigDecimal.ZERO, BigDecimal::add);
            rowTotals.put(rowEntry.getKey(), total);
        }
        
        // 列合计
        Map<String, BigDecimal> columnTotals = new LinkedHashMap<>();
        
        for (String colKey : table.getColumnHeaders()) {
            BigDecimal total = table.getData().values().stream()
                .map(row -> row.get(colKey))
                .filter(Objects::nonNull)
                .reduce(BigDecimal.ZERO, BigDecimal::add);
            columnTotals.put(colKey, total);
        }
        
        // 总计
        BigDecimal grandTotal = columnTotals.values().stream()
            .reduce(BigDecimal.ZERO, BigDecimal::add);
        
        table.setRowTotals(rowTotals);
        table.setColumnTotals(columnTotals);
        table.setGrandTotal(grandTotal);
    }
}
```
