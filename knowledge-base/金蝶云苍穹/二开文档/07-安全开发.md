```csharp
using K3Cloud.Security.Authentication;
using K3Cloud.Security.Authorization;

public class AuthService
{
    private readonly IUserRepository _userRepository;
    private readonly IPasswordHasher _passwordHasher;
    private readonly IJwtTokenService _jwtTokenService;
    private readonly IMfaService _mfaService;
    private readonly IEventLogService _eventLog;
    
    public AuthService(
        IUserRepository userRepository,
        IPasswordHasher passwordHasher,
        IJwtTokenService jwtTokenService,
        IMfaService mfaService,
        IEventLogService eventLog)
    {
        _userRepository = userRepository;
        _passwordHasher = passwordHasher;
        _jwtTokenService = jwtTokenService;
        _mfaService = mfaService;
        _eventLog = eventLog;
    }
    
    public async Task<LoginResult> LoginAsync(LoginRequest request)
    {
        // 1. 验证用户名和密码
        var user = await _userRepository.GetByUsernameAsync(request.Username);
        if (user == null || !user.IsActive)
        {
            await _eventLog.LogLoginFailureAsync(request.Username, "用户不存在或已禁用");
            throw new AuthenticationException("用户名或密码错误");
        }
        
        var passwordValid = await _passwordHasher.VerifyAsync(
            request.Password, 
            user.PasswordHash, 
            user.PasswordSalt
        );
        
        if (!passwordValid)
        {
            await _eventLog.LogLoginFailureAsync(request.Username, "密码错误");
            throw new AuthenticationException("用户名或密码错误");
        }
        
        // 2. 检查是否需要MFA
        var mfaRequired = await CheckMfaRequiredAsync(user);
        if (mfaRequired)
        {
            var mfaToken = await _mfaService.GenerateMfaTokenAsync(user.Id);
            
            return new LoginResult
            {
                Success = false,
                RequireMfa = true,
                MfaToken = mfaToken,
                AvailableFactors = await _mfaService.GetAvailableFactorsAsync(user.Id)
            };
        }
        
        // 3. 生成访问令牌
        var accessToken = await _jwtTokenService.GenerateTokenAsync(user);
        var refreshToken = await _jwtTokenService.GenerateRefreshTokenAsync(user);
        
        // 4. 记录登录成功事件
        await _eventLog.LogLoginSuccessAsync(user.Id, request.IpAddress, request.UserAgent);
        
        return new LoginResult
        {
            Success = true,
            AccessToken = accessToken,
            RefreshToken = refreshToken,
            ExpiresIn = 3600,
            User = new UserInfo
            {
                UserId = user.Id,
                Username = user.Username,
                Roles = user.Roles,
                Permissions = user.Permissions
            }
        };
    }
    
    private async Task<bool> CheckMfaRequiredAsync(User user)
    {
        // 管理员必须MFA
        if (user.Roles.Contains("admin"))
        {
            return true;
        }
        
        // 异地登录需要MFA
        var lastLogin = await _eventLog.GetLastLoginAsync(user.Id);
        if (lastLogin != null && Is异地登录(lastLogin.IpAddress))
        {
            return true;
        }
        
        // 敏感操作需要MFA
        var hasSensitivePermissions = user.Permissions.Any(p => 
            p.StartsWith("sensitive"));
        
        return hasSensitivePermissions;
    }
    
    private bool Is异地登录(string lastIp, string currentIp)
    {
        // 简单的异地检测逻辑
        return lastIp != currentIp && !IsLocalIp(currentIp);
    }
    
    private bool IsLocalIp(string ip)
    {
        // 判断是否为本地IP
        return ip.StartsWith("192.168.") ||
               ip.StartsWith("10.") ||
               ip == "127.0.0.1" ||
               ip == "::1";
    }
}
```

**JWT令牌配置（C#）**:
```csharp
using System.IdentityModel.Tokens.Jwt;
using Microsoft.IdentityModel.Tokens;

public class JwtTokenService : IJwtTokenService
{
    private readonly JwtSecurityTokenHandler _tokenHandler;
    private readonly SigningCredentials _credentials;
    private readonly JwtOptions _options;
    
    public JwtTokenService(IOptions<JwtOptions> options)
    {
        _options = options.Value;
        
        var key = Encoding.ASCII.GetBytes(_options.Secret);
        _credentials = new SigningCredentials(
            new SymmetricSecurityKey(key),
            SecurityAlgorithms.HmacSha256
        );
        
        _tokenHandler = new JwtSecurityTokenHandler(_credentials);
    }
    
    public async Task<string> GenerateTokenAsync(User user)
    {
        var claims = new List<Claim>
        {
            new Claim(ClaimTypes.NameIdentifier, user.Id),
            new Claim(ClaimTypes.Name, user.Username),
            new Claim(JwtRegisteredClaimNames.Jti, Guid.NewGuid().ToString()),
            new Claim(JwtRegisteredClaimNames.Iat, DateTimeOffset.UtcNow.ToUnixTimeSeconds().ToString())
        };
        
        // 添加角色声明
        foreach (var role in user.Roles)
        {
            claims.Add(new Claim("role", role));
        }
        
        // 添加权限声明
        foreach (var permission in user.Permissions)
        {
            claims.Add(new Claim("permission", permission));
        }
        
        var tokenDescriptor = new SecurityTokenDescriptor
        {
            Subject = user.Id,
            Expires = DateTime.UtcNow.AddHours(_options.ExpirationHours),
            Issuer = _options.Issuer,
            Audience = _options.Audience,
            Claims = claims
        };
        
        var token = _tokenHandler.CreateToken(tokenDescriptor);
        return token;
    }
    
    public ClaimsPrincipal ValidateToken(string token)
    {
        var tokenHandler = new JwtSecurityTokenHandler(_credentials);
        
        try
        {
            var principal = tokenHandler.ValidateToken(token, 
                    new TokenValidationParameters
                    {
                        ValidateIssuer = true,
                        ValidIssuer = _options.Issuer,
                        ValidateAudience = true,
                        ValidAudience = _options.Audience,
                        ValidateLifetime = true,
                        Clock = DateTime.UtcNow
                    }, 
                    out var validatedToken);
            
            return principal;
        }
        catch (SecurityTokenExpiredException)
        {
            throw new SecurityTokenException("访问令牌已过期");
        }
        catch (SecurityTokenInvalidSignatureException)
        {
            throw new SecurityTokenException("访问令牌签名无效");
        }
    }
}

public class JwtOptions
{
    public string Secret { get; set; }
    public string Issuer { get; set; }
    public string Audience { get; set; }
    public int ExpirationHours { get; set; } = 24
}
```

**C# 权限服务**:
```csharp
using K3Cloud.Security.Authorization;

public class PermissionService : IPermissionService
{
    private readonly IPermissionRepository _repository;
    private readonly IRoleRepository _roleRepository;
    
    public PermissionService(
        IPermissionRepository repository,
        IRoleRepository roleRepository)
    {
        _repository = repository;
        _roleRepository = roleRepository;
    }
    
    public async Task<bool> HasPermissionAsync(string userId, string resource, string action)
    {
        // 获取用户角色
        var userRoles = await _repository.GetUserRolesAsync(userId);
        
        foreach (var roleId in userRoles)
        {
            var role = await _roleRepository.GetByIdAsync(roleId);
            
            // 检查是否是管理员
            if (role.IsSystem && role.Permissions.Contains("*"))
            {
                return true;
            }
            
            // 检查是否有权限
            var hasPermission = role.Permissions.Contains($"{resource}:{action}");
            if (hasPermission)
            {
                return true;
            }
        }
        
        return false;
    }
    
    public async Task<bool> HasAnyPermissionAsync(string userId, List<string> permissions)
    {
        var userRoles = await _repository.GetUserRolesAsync(userId);
        
        foreach (var roleId in userRoles)
        {
            var role = await _roleRepository.GetByIdAsync(roleId);
            
            if (role.IsSystem && role.Permissions.Contains("*"))
            {
                return true;
            }
            
            var hasAny = role.Permissions.Any(p => permissions.Contains(p));
            if (hasAny)
            {
                return true;
            }
        }
        
        return false;
    }
    
    public async Task<AuthorizationResult> CheckAuthorizationAsync(
        string userId, 
        string resource, 
        string action)
    {
        var hasPermission = await HasPermissionAsync(userId, resource, action);
        
        if (!hasPermission)
        {
            return new AuthorizationResult
            {
                Allowed = false,
                Reason = "权限不足",
                RequiredPermission = $"{resource}:{action}"
            };
        }
        
        return new AuthorizationResult { Allowed = true };
    }
}

public class AuthorizationResult
{
    public bool Allowed { get; set; }
    public string Reason { get; set; }
    public string RequiredPermission { get; set; }
}
```

**C# 数据权限过滤器**:
```csharp
using K3Cloud.Security.Data;

public class DataPermissionFilter
{
    private readonly ICurrentUserService _currentUser;
    
    public DataPermissionFilter(ICurrentUserService currentUser)
    {
        _currentUser = currentUser;
    }
    
    public IQueryable<T> ApplyDataPermissions<T>(
        IQueryable<T> query, 
        string resourceType) where T : class
    {
        var user = _currentUser.GetCurrentUser();
        var dataPermissions = user.GetDataPermissions(resourceType);
        
        if (!dataPermissions.Any())
        {
            return query;
        }
        
        // 应用行级权限
        foreach (var permission in dataPermissions.Where(p => p.Type == "row_level"))
        {
            query = ApplyRowLevelPermission(query, permission);
        }
        
        // 应用列级权限
        var columnsToHide = dataPermissions
            .Where(p => p.Type == "column_level")
            .Select(p => p.Field)
            .ToList();
        
        query = ApplyColumnLevelPermissions(query, columnsToHide);
        
        return query;
    }
    
    private IQueryable<T> ApplyRowLevelPermission<T>(
        IQueryable<T> query, 
        DataPermission permission) where T : class
    {
        // 使用动态表达式构建过滤器
        var parameter = Expression.Parameter(typeof(T));
        var property = Expression.Property(parameter, permission.Field);
        
        var value = _currentUser.GetPropertyValue(permission.Field);
        var comparison = Expression.Equal(property, Expression.Constant(value));
        
        var lambda = Expression.Lambda(comparison, parameter);
        return query.Where(lambda);
    }
    
    private IQueryable<T> ApplyColumnLevelPermissions<T>(
        IQueryable<T> query, 
        List<string> columnsToHide) where T : class
    {
        if (!columnsToHide.Any())
        {
            return query;
        }
        
        // 使用Select隐藏指定列
        var parameter = Expression.Parameter(typeof(T));
        var bindings = new List<MemberBinding>();
        
        // 获取所有属性
        var properties = typeof(T).GetProperties(BindingFlags.Public | BindingFlags.Instance)
            .Where(p => !columnsToHide.Contains(p.Name))
            .ToList();
        
        foreach (var property in properties)
        {
            var member = Expression.Property(parameter, property.Name);
            bindings.Add(member);
        }
        
        var selector = Expression.Lambda(Expression.New(typeof(T), bindings.ToArray()), parameter);
        return query.Select(selector);
    }
}
```

**C# 加密服务**:
```csharp
using K3Cloud.Security.Encryption;
using System.Security.Cryptography;

public class EncryptionService : IEncryptionService
{
    private readonly IConfiguration _config;
    
    public EncryptionService(IConfiguration config)
    {
        _config = config;
    }
    
    public async Task<string> EncryptAsync(string plainText)
    {
        var algorithm = _config["Encryption:DefaultAlgorithm"] ?? "AES-256-CBC";
        
        switch (algorithm)
        {
            case "AES-256-CBC":
                return await EncryptAes256CbcAsync(plainText);
            case "AES-256-CTR":
                return await EncryptAes256CtrAsync(plainText);
            default:
                throw new NotSupportedException($"不支持的加密算法: {algorithm}");
        }
    }
    
    private async Task<string> EncryptAes256CbcAsync(string plainText)
    {
        var key = GetEncryptionKey();
        var iv = GenerateIv();
        
        using var aes = Aes.Create();
        aes.Key = key;
        aes.IV = iv;
        aes.Mode = CipherMode.CBC;
        aes.Padding = PaddingMode.PKCS7;
        
        var encryptor = aes.CreateEncryptor();
        using var msEncrypt = new MemoryStream();
        {
            using (var csEncrypt = new CryptoStream(msEncrypt, encryptor, CryptoStreamMode.Write))
            {
                using (var swEncrypt = new StreamWriter(csEncrypt))
                {
                    swEncrypt.Write(plainText);
                }
            }
            
            return Convert.ToBase64String(msEncrypt.ToArray());
        }
    }
    
    private async Task<string> EncryptAes256CtrAsync(string plainText)
    {
        var key = GetEncryptionKey();
        
        using var aes = Aes.Create();
        aes.Key = key;
        aes.Mode = CipherMode.CTR;
        aes.Padding = PaddingMode.PKCS7;
        
        var encryptor = aes.CreateEncryptor();
        using var msEncrypt = new MemoryStream();
        {
            using (var csEncrypt = new CryptoStream(msEncrypt, encryptor, CryptoStreamMode.Write))
            {
                using (var swEncrypt = new StreamWriter(csEncrypt))
                {
                    swEncrypt.Write(plainText);
                }
            }
            
            var cipherText = new byte[msEncrypt.Length + 16]; // AES-CTR需要16字节的nonce
            Buffer.BlockCopy(msEncrypt.ToArray(), 0, cipherText, msEncrypt.Length);
            Buffer.BlockCopy(msEncrypt.ToArray(), 0, cipherText, msEncrypt.Length, 16, 16);
            
            return Convert.ToBase64String(cipherText);
        }
    }
    
    public async Task<string> DecryptAsync(string cipherText)
    {
        var iv = GetIvFromCipherText(cipherText);
        var key = GetEncryptionKey();
        
        using var aes = Aes.Create();
        aes.Key = key;
        aes.IV = iv;
        
        var decryptor = aes.CreateDecryptor();
        var cipherBytes = Convert.FromBase64String(cipherText);
        
        using var msDecrypt = new MemoryStream(cipherBytes);
        using (var csDecrypt = new CryptoStream(msDecrypt, decryptor, CryptoStreamMode.Read))
        {
            using (var srDecrypt = new StreamReader(csDecrypt))
            {
                return await srDecrypt.ReadToEndAsync();
            }
        }
    }
    
    private byte[] GetEncryptionKey()
    {
        var key = _config["Encryption:Key"];
        return Convert.FromBase64String(key);
    }
    
    private byte[] GenerateIv()
    {
        var iv = new byte[16];
        using (var rng = RandomNumberGenerator.Create())
        {
            rng.GetBytes(iv);
        }
        return iv;
    }
    
    private byte[] GetIvFromCipherText(string cipherText)
    {
        var bytes = Convert.FromBase64String(cipherText);
        var iv = new byte[16];
        Buffer.BlockCopy(bytes, bytes.Length - 16, iv, 0, 16);
        return iv;
    }
}
```

```csharp
    private readonly ICurrentUser _currentUser;
    
    public AuditService(IAuditLogRepository repository, ICurrentUser currentUser)
    {
        _repository = repository;
        _currentUser = currentUser;
    }
    
    public async Task LogAsync(AuditEvent auditEvent)
    {
        var log = new AuditLog
        {
            EventId = auditEvent.EventId,
            EventName = auditEvent.EventName,
            Level = auditEvent.Level,
            Categories = auditEvent.Categories,
            Timestamp = DateTime.UtcNow,
            IpAddress = _currentUser.GetIpAddress(),
            UserAgent = _currentUser.GetUserAgent(),
            UserId = _currentUser.GetUserId()
        };
        
        // 添加事件数据
        if (auditEvent.Data != null)
        {
            foreach (var item in auditEvent.Data)
            {
                log.SetData(item.Key, item.Value);
            }
        }
        
        await _repository.AddAsync(log);
    }
    
    public async Task LogLoginSuccessAsync(string userId, string ipAddress, string userAgent)
    {
        var auditEvent = new AuditEvent
        {
            EventId = "user_login",
            EventName = "用户登录成功",
            Level = "info",
            Categories = new List<string> { "authentication" },
            Data = new Dictionary<string, object>
            {
                { "login_status", "success" },
                { "login_time", DateTime.UtcNow }
            }
        };
        
        await LogAsync(auditEvent);
    }
    
    public async Task LogPermissionDeniedAsync(
        string userId, 
        string resource, 
        string action,
        string reason)
    {
        var auditEvent = new AuditEvent
        {
            EventId = "permission_denied",
            EventName = "权限拒绝",
            Level = "warning",
            Categories = new List<string> { "authorization" },
            Data = new Dictionary<string, object>
            {
                { "resource", resource },
                { "action", action },
                { "reason", reason }
            }
        };
        
        await LogAsync(auditEvent);
    }
    
    public async Task<List<AuditLog>> QueryAsync(AuditQuery query)
    {
        return await _repository.QueryAsync(query);
    }
}

public class AuditEvent
{
    public string EventId { get; set; }
    public string EventName { get; set; }
    public string Level { get; set; }  // info, warning, error, critical
    public List<string> Categories { get; set; }
    public Dictionary<string, object> Data { get; set; }
}
```

```csharp
using K3Cloud.Security.Middleware;
using Microsoft.AspNetCore.Http;

public class SecurityHeadersMiddleware
{
    private readonly RequestDelegate _next;
    
    public SecurityHeadersMiddleware(RequestDelegate next)
    {
        _next = next;
    }
    
    public async Task InvokeAsync(HttpContext context)
    {
        // 添加安全响应头
        context.Response.Headers.Add("X-Content-Type-Options", "nosniff");
        context.Response.Headers.Add("X-Frame-Options", "SAMEORIGIN");
        context.Response.Headers.Add("X-XSS-Protection", "1; mode=block");
        context.Response.Headers.Add("Content-Security-Policy", "default-src 'self'");
        
        await _next(context);
    }
}

public class AntiCsrfMiddleware
{
    private readonly IAntiCsrfService _antiCsrf;
    
    public AntiCsrfMiddleware(IAntiCsrfService antiCsrf)
    {
        _antiCsrf = antiCsrf;
    }
    
    public async Task InvokeAsync(HttpContext context)
    {
        // 验证POST/PUT/DELETE请求
        if (HttpMethods.IsPost(context.Request.Method) ||
            HttpMethods.IsPut(context.Request.Method) ||
            HttpMethods.IsDelete(context.Request.Method))
        {
            var token = context.Request.Headers["X-CSRF-TOKEN"];
            
            if (string.IsNullOrEmpty(token) || !_antiCsrf.ValidateToken(token))
            {
                context.Response.StatusCode = 403;
                await context.Response.WriteAsync("CSRF token validation failed");
                return;
            }
        }
        
        await _next(context);
    }
}
```

## 📝 最佳实践

### 1. 密码安全
- 最小长度：8个字符
- 复杂度：必须包含大小写字母、数字和特殊字符
- 定期更换：每90天更换一次
- 密码哈希：使用PBKDF2或Argon2

### 2. 会话管理
- 使用安全的会话ID
- 设置合理的超时时间（30分钟）
- 登出时立即销毁会话
- 防止会话固定

### 3. API安全
- 使用HTTPS
- 实施请求速率限制
- 验证输入参数
- 最小化数据暴露

### 4. 代码安全
- 避免硬编码密钥
- 使用最新的依赖库
- 定期进行安全审计
- 遵循安全编码规范
