**React Native认证组件**:
```jsx
import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, Alert, AsyncStorage } from 'react-native';
import { useDispatch } from 'react-redux';
import { login } from '@/store/modules/user';

const LoginScreen = ({ navigation }) => {
  const [username, setUsername] = useState('');
  const [password, setPassword] = useState('');
  const [loading, setLoading] = useState(false);
  const dispatch = useDispatch();

  const handleLogin = async () => {
    if (!username || !password) {
      Alert.alert('错误', '请输入用户名和密码');
      return;
    }

    setLoading(true);
    
    try {
      const result = await dispatch(login({ username, password }));
      
      if (result.payload.success) {
        // 保存token
        await AsyncStorage.setItem('token', result.payload.token);
        await AsyncStorage.setItem('userInfo', JSON.stringify(result.payload.user));
        
        // 跳转到首页
        navigation.reset({
          index: 0,
          routes: [{ name: 'Home' }],
        });
      } else {
        Alert.alert('登录失败', result.payload.message);
      }
    } catch (error) {
      console.error('登录失败:', error);
      Alert.alert('错误', '登录失败，请稍后重试');
    } finally {
      setLoading(false);
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>金蝶云苍穹</Text>
      
      <View style={styles.form}>
        <TextInput
          style={styles.input}
          placeholder="用户名"
          value={username}
          onChangeText={setUsername}
          autoCapitalize="none"
        />
        
        <TextInput
          style={styles.input}
          placeholder="密码"
          value={password}
          onChangeText={setPassword}
          secureTextEntry
        />
        
        <TouchableOpacity 
          style={[styles.button, loading && styles.buttonDisabled]}
          onPress={handleLogin}
          disabled={loading}
        >
          <Text style={styles.buttonText}>
            {loading ? '登录中...' : '登录'}
          </Text>
        </TouchableOpacity>
      </View>
    </View>
  );
};
```

```javascript
// src/store/modules/user.js
import { createSlice, createAsyncThunk } from '@reduxjs/toolkit';
import api from '@/services/api';

// 登录
export const login = createAsyncThunk(
  'user/login',
  async ({ username, password }) => {
    const response = await api.post('/api/auth/login', {
      username,
      password,
      platform: 'mobile',
    });
    return response.data;
  }
);

// 获取用户信息
export const getUserInfo = createAsyncThunk(
  'user/getUserInfo',
  async () => {
    const token = await AsyncStorage.getItem('token');
    const response = await api.get('/api/user/info', {
      headers: {
        'Authorization': `Bearer ${token}`
      }
    });
    return response.data;
  }
);

// 退出登录
export const logout = createAsyncThunk(
  'user/logout',
  async () => {
    await AsyncStorage.removeItem('token');
    await AsyncStorage.removeItem('userInfo');
    return {};
  }
);

const userSlice = createSlice({
  name: 'user',
  initialState: {
    token: null,
    userInfo: null,
    loading: false,
    error: null,
  },
  reducers: {
    setToken: (state, action) => {
      state.token = action.payload;
    },
    setUserInfo: (state, action) => {
      state.userInfo = action.payload;
    },
  },
  extraReducers: (builder) => {
    builder
      .addCase(login.pending, (state) => {
        state.loading = true;
      })
      .addCase(login.fulfilled, (state, action) => {
        state.loading = false;
        state.token = action.payload.token;
        state.userInfo = action.payload.user;
      })
      .addCase(login.rejected, (state, action) => {
        state.loading = false;
        state.error = action.error.message;
      });
  },
});

export const { setToken, setUserInfo } = userSlice.actions;
export default userSlice.reducer;
```

```jsx
import React, { useState } from 'react';
import { View, Text, StyleSheet } from 'react-native';
import { Camera, useCameraDevice, useCodeScanner } from 'react-native-vision-camera';

const QRScanner = ({ onScan }) => {
  const device = useCameraDevice('back');
  const [hasPermission, setHasPermission] = useState(false);
  const [scanned, setScanned] = useState(false);

  const codeScanner = useCodeScanner({
    codeTypes: ['qr', 'ean-13'],
    onCodeScanned: (codes) => {
      if (scanned || codes.length === 0) return;
      
      setScanned(true);
      const code = codes[0].value;
      onScan(code);
      
      // 2秒后允许重新扫描
      setTimeout(() => setScanned(false), 2000);
    },
  });

  useEffect(() => {
    (async () => {
      const permission = await Camera.requestCameraPermission();
      setHasPermission(permission === 'granted');
    })();
  }, []);

  if (!hasPermission) {
    return (
      <View style={styles.container}>
        <Text>需要相机权限才能使用扫码功能</Text>
      </View>
    );
  }

  if (!device) {
    return (
      <View style={styles.container}>
        <Text>设备不支持相机</Text>
      </View>
    );
  }

  return (
    <View style={styles.container}>
      <Camera
        style={styles.camera}
        device={device}
        isActive={true}
        codeScanner={codeScanner}
      />
      
      <View style={styles.overlay}>
        <View style={styles.scanFrame}>
          <View style={styles.scanLine} />
        </View>
        <Text style={styles.scanText}>将二维码放入框内即可扫描</Text>

    </View>
  );
};
```

**bills.js**:
```javascript
import { fetchBills } from '../../api/bill';
import { formatDate, formatMoney } from '../../utils/format';

Page({
  data: {
    bills: [],
    loading: false,
    refreshing: false,
    hasMore: true,
    page: 1,
    pageSize: 20,
  },

  onLoad() {
    this.loadBills();
  },

  onPullDownRefresh() {
    this.refreshBills();
  },

  onReachBottom() {
    if (this.data.hasMore && !this.data.loading) {
      this.loadMoreBills();
    }
  },

  async loadBills() {
    this.setData({ loading: true });

    try {
      const res = await fetchBills({
        page: this.data.page,
        pageSize: this.data.pageSize,
      });

      const bills = res.data.rows.map(item => ({
        ...item,
        statusText: this.getStatusText(item.status),
        billDate: formatDate(item.billDate),
        amount: formatMoney(item.amount),
      }));

      this.setData({
        bills: this.data.page === 1 ? bills : [...this.data.bills, ...bills],
        hasMore: bills.length >= this.data.pageSize,
      });
    } catch (error) {
      console.error('加载失败:', error);
      wx.showToast({
        title: '加载失败',
        icon: 'none',
      });
    } finally {
      this.setData({ loading: false });
      wx.stopPullDownRefresh();
    }
  },

  refreshBills() {
    this.setData({ page: 1, hasMore: true });
    this.loadBills();
  },

  loadMoreBills() {
    this.setData({ page: this.data.page + 1 });
    this.loadBills();
  },

  handleSearch(e) {
    const keyword = e.detail.value;
    this.setData({ keyword, page: 1 });
    this.loadBills();
  },

  handleBillDetail(e) {
    const bill = e.currentTarget.dataset.bill;
    wx.navigateTo({
      url: `/pages/billDetail/billDetail?billId=${bill.billId}`,
    });
  },

  getStatusText(status) {
    const statusTexts = {
      draft: '草稿',
      submitted: '已提交',
      approved: '已审批',
      rejected: '已驳回',
    };
    return statusTexts[status] || status;
  },
});
```

```javascript
import JPush from 'jpush-react-native';
import { Platform } from 'react-native';

class PushService {
  init() {
    JPush.init();
    
    if (Platform.OS === 'android') {
      JPush.notifyJSDidLoad(resultCode => {
        if (resultCode === 0) {
          console.log('JPush SDK 初始化成功');
        }
      });
    }
    
    this.setListener();
  }
  
  setListener() {
    // 连接状态监听
    this.connectionListener = JPush.addConnectionEventListener(result => {
      console.log('连接状态:', result.connectEnable);
    });
    
    // 通知点击监听
    this.notificationListener = JPush.addNotificationListener(result => {
      console.log('收到通知:', result);
      this.handleNotificationClick(result);
    });
    
    // 自定义消息监听
    this.customMessageListener = JPush.addCustomMessageListener(result => {
      console.log('收到自定义消息:', result);
      this.handleCustomMessage(result);
    });
    
    // 本地通知监听
    this.localNotificationListener = JPush.addLocalNotificationListener(result => {
      console.log('收到本地通知:', result);
    });
  }
  
  handleNotificationClick(result) {
    const extra = result.extras;
    
    if (extra.billId) {
      // 跳转到单据详情页
      NavigationService.navigate('BillDetail', {
        billId: extra.billId,
      });
    }
  }
  
  handleCustomMessage(result) {
    // 处理自定义消息
    const { type, data } = result.extras;
    
    switch (type) {
      case 'update':
        this.updateData(data);
        break;
      case 'logout':
        this.logout();
        break;
      default:
        break;
    }
  }
  
  setAlias(alias) {
    JPush.setAlias({
      sequence: Date.now(),
      alias: alias,
    });
  }
  
  addTags(tags) {
    JPush.addTags({
      sequence: Date.now(),
      tags: tags,
    });
  }
  
  clean() {
    this.connectionListener?.remove();
    this.notificationListener?.remove();
    this.customMessageListener?.remove();
    this.localNotificationListener?.remove();
  }
}

export default new PushService();
```

## 📊 离线存储

### 1. 数据同步策略

**数据同步管理器**:
```javascript
import AsyncStorage from '@react-native-async-storage/async-storage';
import NetInfo from '@react-native-community/netinfo';
import api from './api';

class SyncManager {
  constructor() {
    this.syncQueue = [];
    this.isSyncing = false;
    this.lastSyncTime = 0;
    this.syncInterval = 5 * 60 * 1000; // 5分钟
  }
  
  // 检查网络状态
  async isOnline() {
    const netInfo = await NetInfo.fetch();
    return netInfo.isConnected;
  }
  
  // 添加到同步队列
  addToSyncQueue(action) {
    this.syncQueue.push(action);
    this.persistSyncQueue();
  }
  
  // 持久化同步队列
  async persistSyncQueue() {
    try {
      await AsyncStorage.setItem(
        'syncQueue',
        JSON.stringify(this.syncQueue)
      );
    } catch (error) {
      console.error('持久化失败:', error);
    }
  }
  
  // 加载同步队列
  async loadSyncQueue() {
    try {
      const queue = await AsyncStorage.getItem('syncQueue');
      this.syncQueue = queue ? JSON.parse(queue) : [];
    } catch (error) {
      console.error('加载失败:', error);
      this.syncQueue = [];
    }
  }
  
  // 同步数据
  async sync() {
    if (this.isSyncing || this.syncQueue.length === 0) {
      return;
    }
    
    const online = await this.isOnline();
    if (!online) {
      console.log('网络未连接，跳过同步');
      return;
    }
    
    this.isSyncing = true;
    
    try {
      // 按顺序处理同步队列
      for (const action of this.syncQueue) {
        await this.syncAction(action);
      }
      
      // 清空同步队列
      this.syncQueue = [];
      await this.persistSyncQueue();
      
      // 更新最后同步时间
      this.lastSyncTime = Date.now();
      await AsyncStorage.setItem(
        'lastSyncTime',
        this.lastSyncTime.toString()
      );
      
    } catch (error) {
      console.error('同步失败:', error);
    } finally {
      this.isSyncing = false;
    }
  }
  
  // 同步单个操作
  async syncAction(action) {
    const { type, data, timestamp } = action;
    
    switch (type) {
      case 'create_bill':
        await api.post('/api/bills', data);
        break;
      case 'update_bill':
        await api.put(`/api/bills/${data.billId}`, data);
        break;
      case 'delete_bill':
        await api.delete(`/api/bills/${data.billId}`);
        break;
      default:
        console.warn('未知操作类型:', type);
    }
  }
  
  // 启动自动同步
  startAutoSync() {
    this.loadSyncQueue();
    
    // 立即同步一次
    this.sync();
    
    // 定时同步
    setInterval(() => {
      this.sync();
    }, this.syncInterval);
  }
  
  // 手动触发同步
  async manualSync() {
    return await this.sync();
  }
  
  // 获取同步状态
  getSyncStatus() {
    return {
      isSyncing: this.isSyncing,
      queueLength: this.syncQueue.length,
      lastSyncTime: this.lastSyncTime,
    };
  }
}

export default new SyncManager();
```
