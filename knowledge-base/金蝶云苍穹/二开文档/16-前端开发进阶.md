export default {
  name: 'K3Form',
  
  props: {
    // 表单数据
    value: {
      type: Object,
      default: () => ({})
    },
    // 字段配置
    fields: {
      type: Array,
      required: true
    },
    // 验证规则
    rules: {
      type: Object,
      default: () => ({})
    },
    // 标签宽度
    labelWidth: {
      type: String,
      default: '120px'
    },
    // 尺寸
    size: {
      type: String,
      default: 'medium'
    },
    // 是否显示按钮
    showButtons: {
      type: Boolean,
      default: true
    },
    // 提交按钮文本
    submitText: {
      type: String,
      default: '提交'
    },
    // 重置按钮文本
    resetText: {
      type: String,
      default: '重置'
    }
  },
  
  data() {
    return {
      formData: {}
    };
  },
  
  watch: {
    value: {
      handler(val) {
        this.formData = { ...val };
      },
      immediate: true,
      deep: true
    },
    
    formData: {
      handler(val) {
        this.$emit('input', val);
      },
      deep: true
    }
  },
  
  methods: {
    // 提交
    handleSubmit() {
      this.$refs.form.validate(valid => {
        if (valid) {
          this.$emit('submit', this.formData);
        } else {
          this.$message.error('请检查表单填写是否正确');
          return false;
        }
      });
    },
    
    // 重置
    handleReset() {
      this.$refs.form.resetFields();
      this.$emit('reset');
    },
    
    // 验证
    validate() {
      return this.$refs.form.validate();
    },
    
    // 验证字段
    validateField(prop) {
      return this.$refs.form.validateField(prop);
    },
    
    // 清除验证
    clearValidate(props) {
      this.$refs.form.clearValidate(props);
    }
  }
};
</script>

<template>
  <div class="k3-table">
    <!-- 工具栏 -->
    <div v-if="showToolbar" class="table-toolbar">
      <div class="toolbar-left">
        <slot name="toolbar-left"/>
      </div>
      <div class="toolbar-right">

            导出

          <el-button icon="el-icon-setting" @click="showColumnConfig = true">
            列配置
          </el-button>
        </el-button-group>
        <slot name="toolbar-right"/>
      </div>
    </div>
    
    <!-- 表格 -->
    <el-table
      ref="table"
      :data="tableData"
      :height="height"
      :max-height="maxHeight"
      :stripe="stripe"
      :border="border"

      :row-key="rowKey"
      :default-expand-all="defaultExpandAll"
      :highlight-current-row="highlightCurrentRow"

          :fixed="column.fixed"

          :header-align="column.headerAlign || 'center'">
          
          <template slot-scope="scope">
            <!-- 格式化显示 -->

            </template>

            <template v-else-if="column.enum">

<script>
export default {
  name: 'K3Table',
  
  props: {
    // 数据源
    data: {
      type: Array,
      default: () => []
    },
    // 列配置
    columns: {
      type: Array,
      required: true
    },
    // 是否显示工具栏
    showToolbar: {
      type: Boolean,
      default: true
    },
    // 是否显示选择列
    showSelection: {
      type: Boolean,
      default: false
    },
    // 是否显示序号列
    showIndex: {
      type: Boolean,
      default: true
    },
    // 是否显示分页
    showPagination: {
      type: Boolean,
      default: true
    },
    // 当前页
    currentPage: {
      type: Number,
      default: 1
    },
    // 每页条数
    pageSize: {
      type: Number,
      default: 20
    },
    // 总条数
    total: {
      type: Number,
      default: 0
    },
    // 分页选项
    pageSizes: {
      type: Array,
      default: () => [10, 20, 50, 100]
    }
  },
  
  data() {
    return {
      tableData: [],
      selectedColumns: [],
      showColumnConfig: false
    };
  },
  
  computed: {
    visibleColumns() {
      return this.columns.filter(col => 
        this.selectedColumns.includes(col.prop)
      );
    },
    
    paginationLayout() {
      return 'total, sizes, prev, pager, next, jumper';
    }
  },
  
  watch: {
    data: {
      handler(val) {
        this.tableData = val;
      },
      immediate: true
    },
    
    columns: {
      handler(val) {
        this.selectedColumns = val.map(col => col.prop);
      },
      immediate: true
    }
  },
  
  methods: {
    // 刷新
    handleRefresh() {
      this.$emit('refresh');
    },
    
    // 导出
    handleExport() {
      this.$emit('export');
    },
    
    // 选择变化
    handleSelectionChange(selection) {
      this.$emit('selection-change', selection);
    },
    
    // 排序变化
    handleSortChange({ column, prop, order }) {
      this.$emit('sort-change', { column, prop, order });
    },
    
    // 行点击
    handleRowClick(row, column, event) {
      this.$emit('row-click', row, column, event);
    },
    
    // 行双击
    handleRowDblclick(row, column, event) {
      this.$emit('row-dblclick', row, column, event);
    },
    
    // 分页大小变化
    handleSizeChange(val) {
      this.$emit('size-change', val);
    },
    
    // 当前页变化
    handleCurrentChange(val) {
      this.$emit('current-change', val);
    },
    
    // 列配置
    handleColumnConfig() {
      this.showColumnConfig = false;
      this.$emit('column-config', this.selectedColumns);
    },
    
    // 获取枚举标签
    getEnumLabel(enumData, value) {
      const item = enumData.find(e => e.value === value);
      return item ? item.label : value;
    },
    
    // 获取枚举标签类型
    getEnumTagType(enumData, value) {
      const item = enumData.find(e => e.value === value);
      return item ? item.type : 'info';
    },
    
    // 格式化金额
    formatAmount(value, precision = 2) {
      if (!value) return '0.00';
      return Number(value).toLocaleString('zh-CN', {
        minimumFractionDigits: precision,
        maximumFractionDigits: precision
      });
    },
    
    // 格式化日期
    formatDate(value, format = 'yyyy-MM-dd') {
      if (!value) return '';
      // 日期格式化逻辑
      return value;
    }
  }
};
</script>

// store/modules/user.js
const state = {
  token: localStorage.getItem('token') || '',
  userInfo: JSON.parse(localStorage.getItem('userInfo') || '{}'),
  roles: [],
  permissions: []
};

const mutations = {
  SET_TOKEN(state, token) {
    state.token = token;
    localStorage.setItem('token', token);
  },
  
  SET_USER_INFO(state, userInfo) {
    state.userInfo = userInfo;
    localStorage.setItem('userInfo', JSON.stringify(userInfo));
  },
  
  SET_ROLES(state, roles) {
    state.roles = roles;
  },
  
  SET_PERMISSIONS(state, permissions) {
    state.permissions = permissions;
  },
  
  CLEAR_USER(state) {
    state.token = '';
    state.userInfo = {};
    state.roles = [];
    state.permissions = [];
    localStorage.removeItem('token');
    localStorage.removeItem('userInfo');
  }
};

const actions = {
  // 登录
  async login({ commit }, userInfo) {
    const { username, password } = userInfo;
    const response = await login({ username, password });
    
    commit('SET_TOKEN', response.token);
    commit('SET_USER_INFO', response.userInfo);
    
    return response;
  },
  
  // 获取用户信息
  async getUserInfo({ commit }) {
    const response = await getUserInfo();
    
    commit('SET_ROLES', response.roles);
    commit('SET_PERMISSIONS', response.permissions);
    
    return response;
  },
  
  // 登出
  async logout({ commit }) {
    await logout();
    commit('CLEAR_USER');
  },
  
  // 刷新token
  async refreshToken({ commit, state }) {
    const response = await refreshToken(state.token);
    commit('SET_TOKEN', response.token);
    return response;
  }
};

const getters = {
  token: state => state.token,
  userInfo: state => state.userInfo,
  roles: state => state.roles,
  permissions: state => state.permissions,
  hasPermission: state => permission => {
    return state.permissions.includes(permission);
  }
};

export default {
  namespaced: true,
  state,
  mutations,
  actions,
  getters
};

### 2. 虚拟滚动

**虚拟列表组件**:
```vue
<template>
  <div class="virtual-list" @scroll="handleScroll">
    <div :style="{ height: totalHeight + 'px' }">
      <div :style="{ transform: `translateY(${offset}px)` }">
        <div
          v-for="item in visibleItems"
          :key="item[itemKey]"
          :style="{ height: itemHeight + 'px' }"
          class="list-item">
          <slot :item="item"/>
        </div>
      </div>
    </div>
  </div>
</template>

<script>
export default {
  name: 'VirtualList',
  
  props: {
    items: Array,
    itemHeight: {
      type: Number,
      default: 50
    },
    itemKey: {
      type: String,
      default: 'id'
    },
    visibleCount: {
      type: Number,
      default: 20
    }
  },
  
  data() {
    return {
      startIndex: 0,
      offset: 0
    };
  },
  
  computed: {
    totalHeight() {
      return this.items.length * this.itemHeight;
    },
    
    visibleItems() {
      const endIndex = Math.min(
        this.startIndex + this.visibleCount,
        this.items.length
      );
      return this.items.slice(this.startIndex, endIndex);
    }
  },
  
  methods: {
    handleScroll(e) {
      const scrollTop = e.target.scrollTop;
      this.startIndex = Math.floor(scrollTop / this.itemHeight);
      this.offset = scrollTop - (scrollTop % this.itemHeight);
    }
  }
};
</script>
```
