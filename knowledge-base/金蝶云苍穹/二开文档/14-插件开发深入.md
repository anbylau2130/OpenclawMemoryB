## 🔄 插件生命周期

### 1. 生命周期阶段

**插件生命周期流程**:
```
加载 → 注册 → 初始化 → 激活 → 运行 → 停用 → 销毁 → 卸载
```

**生命周期详解**:
```java
public class CustomPlugin extends AbstractPlugin {
    
    private static final Logger logger = LoggerFactory.getLogger(CustomPlugin.class);
    
    /**
     * 1. 加载阶段（Loading）
     * - 插件类被加载到JVM
     * - 静态代码块执行
     * - 类初始化
     */
    static {
        logger.info("Plugin class loaded");
    }
    
    /**
     * 2. 注册阶段（Registration）
     * - 插件被注册到插件管理器
     * - 读取插件元数据
     */
    @Override
    public void register(PluginContext context) {
        logger.info("Plugin registered: {}", context.getPluginId());
        // 注册扩展点
        context.registerExtensionPoint("custom.operation", CustomOperation.class);
    }
    
    /**
     * 3. 初始化阶段（Initialization）
     * - 创建插件实例
     * - 注入依赖
     * - 加载配置
     */
    @Override
    public void initialize(PluginContext context) throws PluginException {
        logger.info("Plugin initializing...");
        
        // 加载配置
        PluginConfig config = context.getConfig();
        String dbType = config.getString("database.type", "postgresql");
        
        // 初始化资源
        initializeResources(context);
        
        // 注册监听器
        registerEventListeners(context);
        
        logger.info("Plugin initialized successfully");
    }
    
    /**
     * 4. 激活阶段（Activation）
     * - 插件开始提供服务
     * - 可以处理业务请求
     */
    @Override
    public void activate(PluginContext context) {
        logger.info("Plugin activating...");
        
        // 启动定时任务
        startScheduledTasks(context);
        
        // 启动消息监听
        startMessageListeners(context);
        
        // 设置插件状态
        setActive(true);
        
        logger.info("Plugin activated successfully");
    }
    
    /**
     * 5. 运行阶段（Running）
     * - 插件正常运行
     * - 处理业务逻辑
     */
    @Override
    public Object execute(PluginContext context, PluginRequest request) 
            throws PluginException {
        logger.debug("Executing plugin: {}", request.getOperation());
        
        // 获取操作类型
        String operation = request.getOperation();
        
        // 根据操作类型分发处理
        switch (operation) {
            case "query":
                return handleQuery(context, request);
            case "create":
                return handleCreate(context, request);
            case "update":
                return handleUpdate(context, request);
            case "delete":
                return handleDelete(context, request);
            default:
                throw new PluginException("Unknown operation: " + operation);
        }
    }
    
    /**
     * 6. 停用阶段（Deactivation）
     * - 停止接收新请求
     * - 完成进行中的请求
     */
    @Override
    public void deactivate(PluginContext context) {
        logger.info("Plugin deactivating...");
        
        // 停止接收新请求
        setActive(false);
        
        // 等待进行中的请求完成
        waitForPendingRequests(30, TimeUnit.SECONDS);
        
        // 停止定时任务
        stopScheduledTasks();
        
        // 停止消息监听
        stopMessageListeners();
        
        logger.info("Plugin deactivated successfully");
    }
    
    /**
     * 7. 销毁阶段（Destruction）
     * - 释放资源
     * - 清理缓存
     * - 关闭连接
     */
    @Override
    public void destroy(PluginContext context) {
        logger.info("Plugin destroying...");
        
        // 关闭数据库连接
        closeDatabaseConnections();
        
        // 清理缓存
        clearCache();
        
        // 释放其他资源
        releaseResources();
        
        logger.info("Plugin destroyed successfully");
    }
    
    /**
     * 8. 卸载阶段（Unloading）
     * - 插件类被卸载
     * - 从插件管理器移除
     */
    @Override
    public void unregister(PluginContext context) {
        logger.info("Plugin unregistered: {}", context.getPluginId());
        // 清理注册的扩展点
        context.unregisterExtensionPoint("custom.operation");
    }
}
```

### 2. 状态转换

**插件状态机**:
```java
public enum PluginState {
    /**
     * 已加载：插件类已加载到JVM
     */
    LOADED,
    
    /**
     * 已注册：插件已注册到管理器
     */
    REGISTERED,
    
    /**
     * 已初始化：插件已初始化
     */
    INITIALIZED,
    
    /**
     * 已激活：插件已激活，可提供服务
     */
    ACTIVATED,
    
    /**
     * 运行中：插件正在执行
     */
    RUNNING,
    
    /**
     * 已停用：插件已停用
     */
    DEACTIVATED,
    
    /**
     * 已销毁：插件已销毁
     */
    DESTROYED,
    
    /**
     * 已卸载：插件已卸载
     */
    UNLOADED,
    
    /**
     * 错误：插件状态错误
     */
    ERROR
}

/**
 * 状态转换规则
 */
public class PluginStateTransition {
    
    private static final Map<PluginState, Set<PluginState>> TRANSITIONS = new HashMap<>();
    
    static {
        // 定义允许的状态转换
        TRANSITIONS.put(PluginState.LOADED, 
            Set.of(PluginState.REGISTERED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.REGISTERED, 
            Set.of(PluginState.INITIALIZED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.INITIALIZED, 
            Set.of(PluginState.ACTIVATED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.ACTIVATED, 
            Set.of(PluginState.RUNNING, PluginState.DEACTIVATED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.RUNNING, 
            Set.of(PluginState.ACTIVATED, PluginState.DEACTIVATED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.DEACTIVATED, 
            Set.of(PluginState.ACTIVATED, PluginState.DESTROYED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.DESTROYED, 
            Set.of(PluginState.UNLOADED, PluginState.ERROR));
        
        TRANSITIONS.put(PluginState.ERROR, 
            Set.of(PluginState.DESTROYED));
    }
    
    public static boolean canTransition(PluginState from, PluginState to) {
        Set<PluginState> allowedStates = TRANSITIONS.get(from);
        return allowedStates != null && allowedStates.contains(to);
    }
}

@Service
public class DynamicObjectService {
    
    @Autowired
    private DynamicObjectRepository repository;
    
    /**
     * 查询单据
     */
    public DynamicObject loadBill(String entityName, Object pkValue) {
        return repository.load(entityName, pkValue);
    }
    
    /**
     * 查询单据（带关联）
     */
    public DynamicObject loadBillWithEntries(String entityName, Object pkValue) {
        QueryParameter parameter = new QueryParameter();
        parameter.setEntityName(entityName);
        parameter.setPk(pkValue);
        parameter.setSelectFields("*", "entry.*");  // 加载明细
        
        return repository.queryOne(parameter);
    }
    
    /**
     * 查询单据列表
     */
    public List<DynamicObject> queryBillList(String entityName, 
                                              Map<String, Object> filter) {
        QueryParameter parameter = new QueryParameter();
        parameter.setEntityName(entityName);
        
        // 构建过滤条件
        filter.forEach((key, value) -> {
            parameter.addFilterItem(key, CompareType.EQUAL, value);
        });
        
        return repository.query(parameter);
    }
    
    /**
     * 分页查询
     */
    public PageResult<DynamicObject> queryBillPage(String entityName,
                                                    Map<String, Object> filter,
                                                    int pageIndex, int pageSize) {
        QueryParameter parameter = new QueryParameter();
        parameter.setEntityName(entityName);
        parameter.setPageIndex(pageIndex);
        parameter.setPageSize(pageSize);
        
        filter.forEach((key, value) -> {
            parameter.addFilterItem(key, CompareType.EQUAL, value);
        });
        
        return repository.queryPage(parameter);
    }
    
    /**
     * 保存单据
     */
    public Object saveBill(DynamicObject bill) {
        // 设置审计字段
        bill.set("create_time", new Date());
        bill.set("modify_time", new Date());
        
        return repository.save(bill);
    }
    
    /**
     * 批量保存
     */
    public List<Object> saveBillBatch(List<DynamicObject> bills) {
        return repository.saveBatch(bills);
    }
    
    /**
     * 删除单据
     */
    public void deleteBill(String entityName, Object pkValue) {
        repository.delete(entityName, pkValue);
    }
    
    /**
     * 批量删除

public class BillOperationService {

        context.setOperationKey("audit");
        context.addParameter("audit_result", auditResult);
        context.addParameter("audit_opinion", auditOpinion);
        
        return operationService.executeOperation(context);
    }
    
    /**
     * 反审核单据
     */
    public OperationResult unaudit(String entityName, Object billId) {
        OperationContext context = new OperationContext();
        context.setEntityName(entityName);
        context.setBillId(billId);
        context.setOperationKey("unaudit");
        
        return operationService.executeOperation(context);
    }
    
    /**
     * 关闭单据
     */
    public OperationResult close(String entityName, Object billId) {
        OperationContext context = new OperationContext();
        context.setEntityName(entityName);
        context.setBillId(billId);
        context.setOperationKey("close");
        
        return operationService.executeOperation(context);
    }
    
    /**
     * 执行自定义操作
     */
    public OperationResult executeCustomOperation(String entityName, 
                                                   Object billId,
                                                   String operationKey,
                                                   Map<String, Object> params) {
        OperationContext context = new OperationContext();
        context.setEntityName(entityName);
        context.setBillId(billId);
        context.setOperationKey(operationKey);
        
        if (params != null) {
            params.forEach(context::addParameter);
        }
        
        return operationService.executeOperation(context);
    }
    
    /**
     * 批量操作
     */
    public List<OperationResult> executeBatchOperation(String entityName,
                                                        List<Object> billIds,
                                                        String operationKey) {
        return billIds.parallelStream()
            .map(billId -> {
                OperationContext context = new OperationContext();
                context.setEntityName(entityName);
                context.setBillId(billId);
                context.setOperationKey(operationKey);
                return operationService.executeOperation(context);
            })
            .collect(Collectors.toList());
    }
}

@Slf4j
public class PluginLogger {
    
    /**
     * 记录插件执行日志
     */
    public static void logExecution(String pluginId, String operation, 
                                     long startTime, boolean success) {
        long executionTime = System.currentTimeMillis() - startTime;
        
        if (success) {
            log.info("[PLUGIN] {} - {} executed successfully in {}ms", 
                pluginId, operation, executionTime);
        } else {
            log.error("[PLUGIN] {} - {} execution failed after {}ms", 
                pluginId, operation, executionTime);
        }
    }
    
    /**
     * 记录参数日志
     */
    public static void logParameters(String operation, Map<String, Object> params) {
        if (log.isDebugEnabled()) {
            log.debug("[PLUGIN] {} parameters: {}", operation, 
                JsonUtils.toJson(params));
        }
    }
    
    /**
     * 记录结果日志
     */
    public static void logResult(String operation, Object result) {
        if (log.isDebugEnabled()) {
            log.debug("[PLUGIN] {} result: {}", operation, 
                JsonUtils.toJson(result));
        }
    }
    
    /**
     * 记录异常日志
     */
    public static void logException(String operation, Exception e) {
        log.error("[PLUGIN] {} exception: {}", operation, e.getMessage(), e);
    }
    
    /**
     * 记录性能日志
     */
    public static void logPerformance(String operation, long executionTime, 
                                       Map<String, Object> metrics) {
        if (executionTime > 1000) {  // 超过1秒记录
            log.warn("[PLUGIN-PERF] {} took {}ms, metrics: {}", 
                operation, executionTime, metrics);
        } else {
            log.debug("[PLUGIN-PERF] {} took {}ms", operation, executionTime);
        }
    }
}
```
