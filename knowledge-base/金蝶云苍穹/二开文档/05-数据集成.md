```csharp
using K3Cloud.Integration.Mapping;

public class DataMapper
{
    private readonly MappingConfiguration _config;
    private readonly IReferenceResolver _referenceResolver;
    
    public DataMapper(MappingConfiguration config, IReferenceResolver referenceResolver)
    {
        _config = config;
        _referenceResolver = referenceResolver;
    }
    
    public async Task<TTarget> MapAsync<TSource>(TSource source)
    {
        var target = Activator.CreateInstance<TTarget>();
        var mappings = _config.GetMappings<TSource, TTarget>();
        
        foreach (var mapping in mappings)
        {
            var sourceValue = GetPropertyValue(source, mapping.SourceField);
            var targetValue = await TransformAsync(sourceValue, mapping);
            
            SetPropertyValue(target, mapping.TargetField, targetValue);
        }
        
        return target;
    }
    
    private async Task<object> TransformAsync(object value, FieldMapping mapping)
    {
        // 无转换
        if (mapping.Transformation == "none")
        {
            return value;
        }
        
        // 格式化转换
        if (mapping.Transformation.StartsWith("format:"))
        {
            var format = mapping.Transformation.Split(":")[1];
            return FormatValue(value, format);
        }
        
        // 引用转换
        if (mapping.Transformation.StartsWith("reference:"))
        {
            var entityName = mapping.Transformation.Split(":")[1];
            return await _referenceResolver.ResolveAsync(entityName, value);
        }
        
        // 运算转换
        if (mapping.Transformation.StartsWith("multiply:"))
        {
            var factor = decimal.Parse(mapping.Transformation.Split(":")[1]);
            return Convert.ToDecimal(value) * factor;
        }
        
        // 查找表转换
        if (mapping.Transformation.StartsWith("lookup:"))
        {
            var lookupConfig = mapping.Transformation.Split(":")[1];
            return await LookupValueAsync(lookupConfig, value);
        }
        
        return value;
    }
}
```

```csharp
using K3Cloud.Integration.Sync;

public class DataSyncEngine
{
    private readonly ISyncStrategy _strategy;
    private readonly IChangeDetector _changeDetector;
    private readonly IDataTransformer _transformer;
    
    public DataSyncEngine(
        ISyncStrategy strategy,
        IChangeDetector changeDetector,
        IDataTransformer transformer)
    {
        _strategy = strategy;
        _changeDetector = changeDetector;
        _transformer = transformer;
    }
    
    public async Task<SyncResult> SyncAsync(SyncConfiguration config)
    {
        var changes = await DetectChangesAsync(config);
        
        var result = new SyncResult
        {
            StartTime = DateTime.UtcNow
        };
        
        foreach (var change in changes)
        {
            try
            {
                await ProcessChangeAsync(change);
                result.SuccessCount++;
            }
            catch (Exception ex)
            {
                result.Errors.Add(new SyncError
                {
                    ChangeId = change.ChangeId,
                    Error = ex.Message,
                    Timestamp = DateTime.UtcNow
                });
            }
        }
        
        result.EndTime = DateTime.UtcNow;
        result.Duration = result.EndTime - result.StartTime;
        
        return result;
    }
    
    private async Task<List<DataChange>> DetectChangesAsync(SyncConfiguration config)
    {
        var lastSyncTime = await GetLastSyncTimeAsync(config.SourceSystem);
        
        var changes = await _changeDetector.DetectChangesAsync(
            config.SourceSystem,
            config.TimestampField,
            lastSyncTime
        );
        
        return changes;
    }
    
    private async Task ProcessChangeAsync(DataChange change)
    {
        var transformed = await _transformer.TransformAsync(change.Data, change.Mapping);
        
        switch (change.ChangeType)
        {
            case ChangeType.Insert:
                await InsertAsync(transformed);
                break;
            case ChangeType.Update:
                await UpdateAsync(transformed);
                break;
            case ChangeType.Delete:
                await DeleteAsync(transformed);
                break;
        }
    }
}
```

```csharp
using K3Cloud.Integration.Api;

public class ApiClient : IApiClient
{
    private readonly HttpClient _httpClient;
    private readonly IAuthService _authService;
    private readonly IRetryPolicy _retryPolicy;
    
    public ApiClient(
        HttpClient httpClient,
        IAuthService authService,
        IRetryPolicy retryPolicy)
    {
        _httpClient = httpClient;
        _authService = authService;
        _retryPolicy = retryPolicy;
    }
    
    public async Task<T> GetAsync<T>(string endpoint, object parameters = null)
    {
        return await _retryPolicy.ExecuteAsync(async () =>
        {
            var token = await _authService.GetTokenAsync();
            var request = CreateRequest(HttpMethod.Get, endpoint, token, parameters);
            var response = await _httpClient.SendAsync(request);
            return await HandleResponseAsync<T>(response);
        });
    }
    
    public async Task<T> PostAsync<T>(string endpoint, object data)
    {
        return await _retryPolicy.ExecuteAsync(async () =>
        {
            var token = await _authService.GetTokenAsync();
            var request = CreateRequest(HttpMethod.Post, endpoint, token, data);
            var response = await _httpClient.SendAsync(request);
            return await HandleResponseAsync<T>(response);
        });
    }
    
    private HttpRequestMessage CreateRequest(
        HttpMethod method,
        string endpoint,
        string token,
        object data)
    {
        var request = new HttpRequestMessage(method, endpoint);
        request.Headers.Add("Authorization", $"Bearer {token}");
        request.Headers.Add("Content-Type", "application/json");
        
        if (data != null && (method == HttpMethod.Post || method == HttpMethod.Put))
        {
            request.Content = new StringContent(
                JsonSerializer.Serialize(data),
                Encoding.UTF8,
                "application/json");
        }
        
        return request;
    }
    
    private async Task<T> HandleResponseAsync<T>(HttpResponseMessage response)
    {
        if (response.IsSuccessStatusCode)
        {
            var content = await response.Content.ReadAsStringAsync();
            return JsonSerializer.Deserialize<T>(content);
        }
        else
        {
            var error = await response.Content.ReadAsStringAsync();
            throw new ApiException(
                $"API请求失败: {response.StatusCode}",
                response.StatusCode,
                error);
        }
    }
}
```

```csharp
using K3Cloud.Integration.Webhooks;

[ApiController]
[Route("api/webhook")]
public class WebhookController : ControllerBase
{
    private readonly IWebhookValidator _validator;
    private readonly IWebhookProcessor _processor;
    
    [HttpPost]
    [Route("order")]
    public async Task<IActionResult> HandleOrderWebhook(
        [FromHeader(Name = "X-Webhook-Secret")] string secret,
        [FromBody] WebhookPayload payload)
    {
        // 验证webhook密钥
        if (!_validator.ValidateSecret(secret))
        {
            return Unauthorized(new { error = "Invalid webhook secret" });
        }
        
        // 验证事件签名
        if (!_validator.ValidateSignature(payload))
        {
            return BadRequest(new { error = "Invalid signature" });
        }
        
        // 处理webhook事件
        await _processor.ProcessAsync(payload);
        
        return Ok(new { status = "success" });
    }
}
```

### 1. 数据库连接池

```csharp
using K3Cloud.Integration.Database;

public class DatabaseConnectionPool
{
    private readonly ConcurrentBag<IDbConnection> _connections;
    private readonly string _connectionString;
    private readonly int _maxPoolSize;
    private readonly Semaphore _semaphore;
    
    public DatabaseConnectionPool(string connectionString, int maxPoolSize = 20)
    {
        _connectionString = connectionString;
        _maxPoolSize = maxPoolSize;
        _connections = new ConcurrentBag<IDbConnection>();
        _semaphore = new Semaphore(maxPoolSize, maxPoolSize);
    }
    
    public async Task<IDbConnection> GetConnectionAsync()
    {
        await _semaphore.WaitAsync();
        
        if (_connections.TryTake(out var connection))
        {
            if (IsConnectionValid(connection))
            {
                return connection;
            }
            else
            {
                await connection.CloseAsync();
            }
        }
        
        // 创建新连接
        connection = CreateConnection();
        return connection;
    }
    
    public void ReturnConnection(IDbConnection connection)
    {
        if (IsConnectionValid(connection))
        {
            _connections.Add(connection);
        }
        else
        {
            connection.Close();
        }
        
        _semaphore.Release();
    }
}
```

### 2. 批量数据处理

```csharp
public class BatchDataProcessor
{
    private readonly DatabaseConnectionPool _connectionPool;
    private readonly int _batchSize;
    
    public BatchDataProcessor(DatabaseConnectionPool connectionPool, int batchSize = 1000)
    {
        _connectionPool = connectionPool;
        _batchSize = batchSize;
    }
    
    public async Task ProcessBatchAsync<T>(IEnumerable<T> data, Func<T, Task> processor)
    {
        var batch = new List<T>();
        var processed = 0;
        var total = data.Count();
        
        using var connection = await _connectionPool.GetConnectionAsync()
        {
            using var transaction = connection.BeginTransaction();
            
            try
            {
                foreach (var item in data)
                {
                    batch.Add(item);
                    
                    if (batch.Count >= _batchSize)
                    {
                        await ProcessBatchInternalAsync(connection, transaction, batch, processor);
                        processed += batch.Count;
                        batch.Clear();
                        
                        // 更新进度
                        UpdateProgress(processed, total);
                    }
                }
                
                // 处理剩余数据
                if (batch.Count > 0)
                {
                    await ProcessBatchInternalAsync(connection, transaction, batch, processor);
                    processed += batch.Count;
                }
                
                transaction.Commit();
            }
            catch
            {
                transaction.Rollback();
                throw;
            }
            finally
            {
                _connectionPool.ReturnConnection(connection);
            }
        }
    }
    
    private async Task ProcessBatchInternalAsync<T>(
        IDbConnection connection,
        IDbTransaction transaction,
        List<T> batch,
        Func<T, Task> processor)
    {
        foreach (var item in batch)
        {
            await processor(item);
        }
    }
}
```

```csharp
using K3Cloud.Integration.Messaging;

public class MessageProducer : IMessageProducer
{
    private readonly IMessageQueue _queue;
    
    public MessageProducer(IMessageQueue queue)
    {
        _queue = queue;
    }
    
    public async Task SendAsync<T>(string topic, T message, MessageOptions options = null)
    {
        var envelope = new MessageEnvelope<T>
        {
            Topic = topic,
            Payload = message,
            Headers = options?.Headers ?? new Dictionary<string, string>(),
            Timestamp = DateTime.UtcNow,
            MessageId = Guid.NewGuid().ToString()
        };
        
        var serialized = JsonSerializer.Serialize(envelope);
        await _queue.PublishAsync(topic, serialized);
    }
    
    public async Task SendBatchAsync<T>(string topic, IEnumerable<T> messages)
    {
        var envelopes = messages.Select(m => new MessageEnvelope<T>
        {
            Topic = topic,
            Payload = m,
            Timestamp = DateTime.UtcNow,
            MessageId = Guid.NewGuid().ToString()
        });
        
        var serialized = JsonSerializer.Serialize(envelopes);
        await _queue.PublishBatchAsync(topic, serialized);
    }
}
```

```csharp
public class MessageConsumer : IMessageConsumer
{
    private readonly IMessageQueue _queue;
    private readonly IMessageHandler _handler;
    private readonly CancellationTokenSource _cts;
    
    public MessageConsumer(IMessageQueue queue, IMessageHandler handler)
    {
        _queue = queue;
        _handler = handler;
        _cts = new CancellationTokenSource();
    }
    
    public async Task StartAsync(string topic, string subscription)
    {
        await _queue.SubscribeAsync(topic, subscription, async (message) =>
        {
            try
            {
                var envelope = JsonSerializer.Deserialize<MessageEnvelope>(message);
                await _handler.HandleAsync(envelope);
            }
            catch (Exception ex)
            {
                await _handler.HandleErrorAsync(message, ex);
            }
        }, _cts.Token);
    }
    
    public void Stop()
    {
        _cts.Cancel();
    }
}
```

## 🛠️ 错误处理

### 1. 重试机制

```csharp
public class ExponentialBackoffRetryPolicy : IRetryPolicy
{
    private readonly int _maxRetries;
    private readonly int _baseBackoffMs;
    
    public ExponentialBackoffRetryPolicy(int maxRetries = 3, int baseBackoffMs = 1000)
    {
        _maxRetries = maxRetries;
        _baseBackoffMs = baseBackoffMs;
    }
    
    public async Task ExecuteAsync(Func<Task> action)
    {
        int retry = 0;
        int backoffMs = _baseBackoffMs;
        
        while (retry <= _maxRetries)
        {
            try
            {
                return await action();
            }
            catch (Exception ex) when (IsTransientError(ex))
            {
                retry++;
                if (retry > _maxRetries)
                {
                    throw;
                }
                
                await Task.Delay(backoffMs);
                backoffMs *= 2;  // 指数退避
            }
        }
    }
    
    private bool IsTransientError(Exception ex)
    {
        // 判断是否为瞬时错误（网络超时、限流等）
        return ex is TimeoutException || 
               ex is HttpRequestException ||
               ex.Message.Contains("rate limit");
    }
}
```
